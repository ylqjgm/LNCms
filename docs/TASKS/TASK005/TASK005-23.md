### 2. 实现UpdateUserStatus方法

```go
// UpdateUserStatus 更新用户状态
// @Summary 更新用户状态
// @Description 更新用户激活状态
// @Tags 用户管理
// @Accept json
// @Produce json
// @Security BearerAuth
// @Param id path int true "用户ID"
// @Param request body dto.UpdateUserStatusRequest true "更新用户状态请求"
// @Success 200 {object} response.Envelope{data=dto.UserResponse}
// @Failure 400 {object} response.Envelope
// @Failure 401 {object} response.Envelope
// @Failure 404 {object} response.Envelope
// @Failure 422 {object} response.Envelope
// @Failure 500 {object} response.Envelope
// @Router /api/v1/users/{id}/status [put]
func (h *userHandler) UpdateUserStatus(c *gin.Context) {
    // 1. 获取路径参数
    idStr := c.Param("id")
    id, err := strconv.ParseInt(idStr, 10, 64)
    if err != nil {
        response.BadRequest(c, "无效的用户ID", err)
        return
    }
    
    // 2. 绑定请求参数
    var req dto.UpdateUserStatusRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        response.BadRequest(c, "请求参数错误", err)
        return
    }
    
    // 3. 调用UserService.UpdateStatus
    user, err := h.userService.UpdateStatus(c.Request.Context(), id, req.IsActive)
    if err != nil {
        if isNotFoundError(err) {
            response.NotFound(c, "用户不存在", err)
            return
        }
        if isValidationError(err) {
            response.UnprocessableEntity(c, "验证失败", err)
            return
        }
        response.InternalServerError(c, "更新用户状态失败", err)
        return
    }
    
    // 4. 转换为响应DTO
    userResponse := dto.UserResponseFromModel(user)
    
    // 5. 格式化响应（Envelope格式）
    response.OK(c, userResponse, "更新用户状态成功")
}
```

### 5. 注意事项

- **请求参数绑定**：使用ShouldBindJSON绑定JSON请求体
- **路径参数解析**：使用c.Param获取路径参数，并转换为int64类型
- **参数验证**：使用binding标签进行参数验证，is_active为必填字段
- **错误处理**：区分不同的错误类型（未找到、验证失败、服务器错误）
- **响应格式**：使用Envelope格式，HTTP状态码200 OK
- **权限控制**：需要认证和管理员权限，在路由配置中应用JWT认证中间件和权限中间件
- **状态更新**：只更新is_active字段，不影响其他字段
