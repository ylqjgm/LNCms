```go
// AuthorHandler 作者控制器接口（扩展）
// 提供作者管理的HTTP API处理
type AuthorHandler interface {
    // Register 作者注册
    // POST /api/v1/authors/register

    // Login 作者登录
    // POST /api/v1/authors/login

```

### 2. 实现Login方法

在authorHandler实现中添加Login方法：

```go
// Login 作者登录
// @Summary 作者登录
// @Description 作者登录接口，支持作者名或邮箱登录
// @Tags 作者管理
// @Accept json
// @Produce json
// @Param request body dto.AuthorLoginRequest true "登录请求"
// @Success 200 {object} response.Envelope{data=dto.AuthorLoginResponse}
// @Failure 400 {object} response.Envelope
// @Failure 401 {object} response.Envelope
// @Failure 422 {object} response.Envelope
// @Failure 500 {object} response.Envelope
// @Router /api/v1/authors/login [post]
func (h *authorHandler) Login(c *gin.Context) {
    // 1. 绑定请求参数
    var req dto.AuthorLoginRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        response.BadRequest(c, "请求参数错误", err)
        return
    }
    
    // 2. 获取客户端IP和User-Agent
    req.IPAddress = c.ClientIP()
    req.UserAgent = c.GetHeader("User-Agent")
    
    // 3. 调用AuthService.AuthorLogin
    resp, err := h.authService.AuthorLogin(c.Request.Context(), &req)
    if err != nil {
        // 根据错误类型返回不同的HTTP状态码
        if strings.Contains(err.Error(), "不存在") || strings.Contains(err.Error(), "密码错误") {
            response.Unauthorized(c, err.Error(), nil)
            return
        }
        if strings.Contains(err.Error(), "未激活") || strings.Contains(err.Error(), "已锁定") || strings.Contains(err.Error(), "未验证") {
            response.Forbidden(c, err.Error(), nil)
            return
        }
        response.InternalServerError(c, "登录失败", err)
        return
    }
    
    // 4. 格式化响应（Envelope格式）
    response.Success(c, http.StatusOK, resp)
}
```

### 3. 注意事项

- **请求验证**：使用Gin的binding标签进行参数验证
- **错误处理**：根据错误类型返回合适的HTTP状态码（401未授权、403禁止访问等）
- **响应格式**：使用Envelope格式返回响应
- **错误消息**：所有错误消息必须使用中文
- **API文档**：添加Swagger注释
