### 2. 实现DeleteAuthor方法

在authorHandler实现中添加DeleteAuthor方法：

```go
// DeleteAuthor 删除作者
// @Summary 删除作者
// @Description 删除作者（管理员操作）
// @Tags 作者管理
// @Accept json
// @Produce json
// @Security BearerAuth
// @Param id path int true "作者ID"
// @Success 200 {object} response.Envelope
// @Failure 400 {object} response.Envelope
// @Failure 404 {object} response.Envelope
// @Failure 500 {object} response.Envelope
// @Router /api/v1/authors/{id} [delete]
func (h *authorHandler) DeleteAuthor(c *gin.Context) {
    // 1. 获取路径参数
    authorID, err := strconv.ParseInt(c.Param("id"), 10, 64)
    if err != nil {
        response.BadRequest(c, "无效的作者ID", err)
        return
    }
    
    // 2. 调用AuthorService.DeleteAuthor
    if err := h.authorService.DeleteAuthor(c.Request.Context(), authorID); err != nil {
        if strings.Contains(err.Error(), "不存在") {
            response.NotFound(c, err.Error(), nil)
            return
        }
        response.InternalServerError(c, "删除作者失败", err)
        return
    }
    
    // 3. 格式化响应（Envelope格式）
    response.Success(c, http.StatusOK, map[string]string{
        "message": "作者删除成功",
    })
}
```
