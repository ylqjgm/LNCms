### 2. 实现BanAuthor方法

在authorHandler实现中添加BanAuthor方法：

```go
// BanAuthor 封禁作者
// @Summary 封禁作者
// @Description 封禁作者，可以设置封禁原因和封禁期限
// @Tags 作者管理
// @Accept json
// @Produce json
// @Security BearerAuth
// @Param id path int true "作者ID"
// @Param request body dto.BanAuthorRequest true "封禁请求"
// @Success 200 {object} response.Envelope
// @Failure 400 {object} response.Envelope
// @Failure 404 {object} response.Envelope
// @Failure 500 {object} response.Envelope
// @Router /api/v1/authors/{id}/ban [put]
func (h *authorHandler) BanAuthor(c *gin.Context) {
    // 1. 获取路径参数
    authorID, err := strconv.ParseInt(c.Param("id"), 10, 64)
    if err != nil {
        response.BadRequest(c, "无效的作者ID", err)
        return
    }
    
    // 2. 绑定请求参数
    var req dto.BanAuthorRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        response.BadRequest(c, "请求参数错误", err)
        return
    }
    
    // 3. 调用AuthorService.BanAuthor
    var duration *time.Duration
    if req.Duration != nil {
        duration = req.Duration
    }
    
    if err := h.authorService.BanAuthor(c.Request.Context(), authorID, req.Reason, duration); err != nil {
        if strings.Contains(err.Error(), "不存在") {
            response.NotFound(c, err.Error(), nil)
            return
        }
        if strings.Contains(err.Error(), "已被封禁") {
            response.Conflict(c, err.Error(), nil)
            return
        }
        response.InternalServerError(c, "封禁作者失败", err)
        return
    }
    
    // 4. 格式化响应（Envelope格式）
    response.Success(c, http.StatusOK, map[string]string{
        "message": "作者封禁成功",
    })
}
```

### 3. 实现UnbanAuthor方法

在authorHandler实现中添加UnbanAuthor方法：

```go
// UnbanAuthor 解封作者
// @Summary 解封作者
// @Description 解封作者，恢复作者正常状态
// @Tags 作者管理
// @Accept json
// @Produce json
// @Security BearerAuth
// @Param id path int true "作者ID"
// @Success 200 {object} response.Envelope
// @Failure 400 {object} response.Envelope
// @Failure 404 {object} response.Envelope
// @Failure 500 {object} response.Envelope
// @Router /api/v1/authors/{id}/unban [put]
func (h *authorHandler) UnbanAuthor(c *gin.Context) {
    // 1. 获取路径参数
    authorID, err := strconv.ParseInt(c.Param("id"), 10, 64)
    if err != nil {
        response.BadRequest(c, "无效的作者ID", err)
        return
    }
    
    // 2. 调用AuthorService.UnbanAuthor
    if err := h.authorService.UnbanAuthor(c.Request.Context(), authorID); err != nil {
        if strings.Contains(err.Error(), "不存在") {
            response.NotFound(c, err.Error(), nil)
            return
        }
        if strings.Contains(err.Error(), "未被封禁") {
            response.Conflict(c, err.Error(), nil)
            return
        }
        response.InternalServerError(c, "解封作者失败", err)
        return
    }
    
    // 3. 格式化响应（Envelope格式）
    response.Success(c, http.StatusOK, map[string]string{
        "message": "作者解封成功",
    })
}
```
