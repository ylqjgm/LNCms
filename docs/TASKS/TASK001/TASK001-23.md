    // 1200-1299: 缓存错误
    ErrCodeCacheError          = "1005" // 缓存错误
    ErrCodeCacheConnection     = "1006" // 缓存连接错误
    
    // 1300-1399: 搜索服务错误
    ErrCodeSearchError         = "1007" // 搜索服务错误
