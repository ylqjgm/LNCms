package repository

import (
    "context"

    "github.com/stretchr/testify/assert"
    "github.com/stretchr/testify/require"
    "github.com/stretchr/testify/suite"
    "gorm.io/driver/sqlite"
    "gorm.io/gorm"
    "yourproject/internal/pkg/model"
)

// RoleRepositoryTestSuite 角色仓储测试套件
type RoleRepositoryTestSuite struct {
    suite.Suite
    db     *gorm.DB
    repo   RoleRepository
    ctx    context.Context
}

// SetupTest 每个测试前的设置
func (s *RoleRepositoryTestSuite) SetupTest() {
    // 使用SQLite内存数据库进行测试
    db, err := gorm.Open(sqlite.Open(":memory:"), &gorm.Config{})
    require.NoError(s.T(), err)
    
    // 自动迁移表结构
    err = db.AutoMigrate(&model.Role{})
    require.NoError(s.T(), err)
    
    s.db = db
    s.repo = NewRoleRepository(db)
    s.ctx = context.Background()
}

// TearDownTest 每个测试后的清理
func (s *RoleRepositoryTestSuite) TearDownTest() {
    if s.db != nil {
        sqlDB, _ := s.db.DB()
        if sqlDB != nil {
            sqlDB.Close()
        }
    }
}

// TestGetByID 测试GetByID方法
func (s *RoleRepositoryTestSuite) TestGetByID() {
    // 创建测试数据
    role := &model.Role{
        RoleName:        "测试角色",
        RoleDescription: stringPtr("测试角色描述"),
        IsActive:        true,
        IsSystem:        false,
    }
    err := s.repo.Create(s.ctx, role)
    require.NoError(s.T(), err)
    
    tests := []struct {
        name    string
        id      int64
        wantErr bool
        errMsg  string
    }{

            id:      role.ID,
            wantErr: false,
        },
        {
            name:    "异常情况：获取不存在的角色",
            id:      99999,
            wantErr: true,
            errMsg:  "角色不存在",
        },
    }
    
    for _, tt := range tests {
        s.T().Run(tt.name, func(t *testing.T) {
            got, err := s.repo.GetByID(s.ctx, tt.id)
            if tt.wantErr {
                assert.Error(t, err)
                if tt.errMsg != "" {
                    assert.Contains(t, err.Error(), tt.errMsg)
                }
                assert.Nil(t, got)
            } else {
                assert.NoError(t, err)
                assert.NotNil(t, got)
                assert.Equal(t, tt.id, got.ID)
            }
        })
    }
}

func (s *RoleRepositoryTestSuite) TestGetAll() {
    // 创建测试数据
    roles := []*model.Role{
        {
            RoleName:        "角色1",
            RoleDescription: stringPtr("角色1描述"),
            IsActive:        true,
        },
        {
            RoleName:        "角色2",
            RoleDescription: stringPtr("角色2描述"),
            IsActive:        true,
        },
    }
    for _, role := range roles {
        err := s.repo.Create(s.ctx, role)
        require.NoError(s.T(), err)
    }
    
    // 测试获取所有角色
    got, err := s.repo.GetAll(s.ctx)
    assert.NoError(s.T(), err)
    assert.GreaterOrEqual(s.T(), len(got), len(roles))
    
    // 验证返回的角色数据
    roleMap := make(map[int64]*model.Role)
    for _, role := range got {
        roleMap[role.ID] = role
    }
    for _, expected := range roles {
        actual, exists := roleMap[expected.ID]
        assert.True(s.T(), exists, "角色 %d 应该存在", expected.ID)
        assert.Equal(s.T(), expected.RoleName, actual.RoleName)
    }
}

// TestCreate 测试Create方法
func (s *RoleRepositoryTestSuite) TestCreate() {
    tests := []struct {
        name    string
        role    *model.Role
        wantErr bool
    }{
        {
            name: "正常情况：创建角色",
            role: &model.Role{
                RoleName:        "新角色",
                RoleDescription: stringPtr("新角色描述"),
                IsActive:        true,
                IsSystem:        false,
            },
            wantErr: false,
        },
        {
            name: "异常情况：创建重复名称的角色",
            role: &model.Role{
                RoleName: "重复角色",
            },
            wantErr: false, // 第一次创建应该成功
        },
    }
    
    for i, tt := range tests {
        s.T().Run(tt.name, func(t *testing.T) {
            err := s.repo.Create(s.ctx, tt.role)
            if tt.wantErr {
                assert.Error(t, err)
            } else {
                assert.NoError(t, err)
                assert.NotZero(t, tt.role.ID)
                
                // 第二次创建相同名称的角色应该失败
                if i == 1 {
                    duplicate := &model.Role{
                        RoleName: tt.role.RoleName,
                    }
                    err := s.repo.Create(s.ctx, duplicate)
                    assert.Error(t, err, "创建重复名称的角色应该失败")
                }
            }
        })
    }
}
