```go
package repository

import (
    "context"
    "testing"
    
    "github.com/stretchr/testify/assert"
    "github.com/stretchr/testify/require"
    "github.com/stretchr/testify/suite"
    "gorm.io/driver/sqlite"
    "gorm.io/gorm"
    "yourproject/internal/pkg/model"
)

// UserGroupRepositoryTestSuite 用户组仓储测试套件
type UserGroupRepositoryTestSuite struct {
    suite.Suite
    db     *gorm.DB
    repo   UserGroupRepository
    ctx    context.Context
}

// SetupTest 每个测试前的设置
func (s *UserGroupRepositoryTestSuite) SetupTest() {
    // 使用SQLite内存数据库进行测试
    db, err := gorm.Open(sqlite.Open(":memory:"), &gorm.Config{})
    require.NoError(s.T(), err)
    
    // 自动迁移表结构
    err = db.AutoMigrate(&model.UserGroup{})
    require.NoError(s.T(), err)
    
    s.db = db
    s.repo = NewUserGroupRepository(db)
    s.ctx = context.Background()
}

// TearDownTest 每个测试后的清理
func (s *UserGroupRepositoryTestSuite) TearDownTest() {
    if s.db != nil {
        sqlDB, _ := s.db.DB()
        if sqlDB != nil {
            sqlDB.Close()
        }
    }
}
```

```go
// TestGetByID 测试GetByID方法
func (s *UserGroupRepositoryTestSuite) TestGetByID() {
    // 创建测试数据
    userGroup := &model.UserGroup{
        GroupName:        "测试用户组",
        GroupDescription: stringPtr("测试用户组描述"),
        IsActive:         true,
    }
    err := s.repo.Create(s.ctx, userGroup)
    require.NoError(s.T(), err)
    
    tests := []struct {
        name    string
        id      int64
        wantErr bool
        errMsg  string
    }{
        {
            name:    "正常情况：获取存在的用户组",
            id:      userGroup.ID,
            wantErr: false,
        },
        {
            name:    "异常情况：获取不存在的用户组",
            id:      99999,
            wantErr: true,
            errMsg:  "用户组不存在",
        },
    }
    
    for _, tt := range tests {
        s.T().Run(tt.name, func(t *testing.T) {
            got, err := s.repo.GetByID(s.ctx, tt.id)
            if tt.wantErr {
                assert.Error(t, err)
                if tt.errMsg != "" {
                    assert.Contains(t, err.Error(), tt.errMsg)
                }
                assert.Nil(t, got)
            } else {
                assert.NoError(t, err)
                assert.NotNil(t, got)
                assert.Equal(t, tt.id, got.ID)
            }
        })
    }
}
```

```go
func (s *UserGroupRepositoryTestSuite) TestGetAll() {
    // 创建测试数据
    userGroups := []*model.UserGroup{
        {
            GroupName:        "用户组1",
            GroupDescription: stringPtr("用户组1描述"),
            IsActive:         true,
        },
        {
            GroupName:        "用户组2",
            GroupDescription: stringPtr("用户组2描述"),
            IsActive:         true,
        },
    }
    for _, userGroup := range userGroups {
        err := s.repo.Create(s.ctx, userGroup)
        require.NoError(s.T(), err)
    }
    
    // 测试获取所有用户组
    got, err := s.repo.GetAll(s.ctx)
    assert.NoError(s.T(), err)
    assert.GreaterOrEqual(s.T(), len(got), len(userGroups))
    
    // 验证返回的用户组数据
    userGroupMap := make(map[int64]*model.UserGroup)
    for _, userGroup := range got {
        userGroupMap[userGroup.ID] = userGroup
    }
    for _, expected := range userGroups {
        actual, exists := userGroupMap[expected.ID]
        assert.True(s.T(), exists, "用户组 %d 应该存在", expected.ID)
        assert.Equal(s.T(), expected.GroupName, actual.GroupName)
    }
}
```
