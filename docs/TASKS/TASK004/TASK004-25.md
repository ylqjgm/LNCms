// TestGetAll 测试GetAll方法
func (s *AuthorTypeRepositoryTestSuite) TestGetAll() {
    // 创建测试数据，使用不同的sort_order值
    authorTypes := []*model.AuthorType{
        {
            Code:      "reprint",
            Name:      "转载",
            SortOrder: 2,
        },
        {
            Code:      "original",
            Name:      "原创",
            SortOrder: 1,
        },
        {
            Code:      "translate",
            Name:      "翻译",
            SortOrder: 3,
        },
    }
    for _, authorType := range authorTypes {
        err := s.repo.Create(s.ctx, authorType)
        require.NoError(s.T(), err)
    }
    
    // 测试获取所有作者类型
    got, err := s.repo.GetAll(s.ctx)
    assert.NoError(s.T(), err)
    assert.GreaterOrEqual(s.T(), len(got), len(authorTypes))
    
    // 验证排序：应该按sort_order ASC排序
    for i := 1; i < len(got); i++ {
        assert.GreaterOrEqual(s.T(), got[i].SortOrder, got[i-1].SortOrder,
            "作者类型应该按sort_order升序排列")
        if got[i].SortOrder == got[i-1].SortOrder {
            // 如果sort_order相同，应该按id升序排列
            assert.Greater(s.T(), got[i].ID, got[i-1].ID,
                "sort_order相同时，应该按id升序排列")
        }
    }
    
    // 验证返回的作者类型数据
    authorTypeMap := make(map[int64]*model.AuthorType)
    for _, authorType := range got {
        authorTypeMap[authorType.ID] = authorType
    }
    for _, expected := range authorTypes {
        actual, exists := authorTypeMap[expected.ID]
        assert.True(s.T(), exists, "作者类型 %d 应该存在", expected.ID)
        assert.Equal(s.T(), expected.Code, actual.Code)
        assert.Equal(s.T(), expected.Name, actual.Name)
    }
}
