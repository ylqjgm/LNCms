func (r *authorLevelRepository) GetByID(ctx context.Context, id int64) (*model.AuthorLevel, error) {
    var authorLevel model.AuthorLevel
    err := r.db.WithContext(ctx).
        Where("id = ?", id).
        First(&authorLevel).Error
    if err != nil {
        if err == gorm.ErrRecordNotFound {
            return nil, fmt.Errorf("作者等级不存在")
