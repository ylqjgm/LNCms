### 4. 实现ComicChapterRepository接口

```go
// comicChapterRepository ComicChapterRepository接口实现
type comicChapterRepository struct {
    db *gorm.DB
}

// NewComicChapterRepository 创建ComicChapterRepository实例
func NewComicChapterRepository(db *gorm.DB) ComicChapterRepository {
    return &comicChapterRepository{db: db}
}

// Create 创建漫画章节
func (r *comicChapterRepository) Create(ctx context.Context, chapter *models.ComicChapter) error {
    if err := r.db.WithContext(ctx).Create(chapter).Error; err != nil {
        return fmt.Errorf("创建漫画章节失败: %w", err)
    }
    return nil
}

// GetByID 根据ID获取漫画章节
func (r *comicChapterRepository) GetByID(ctx context.Context, id int64) (*models.ComicChapter, error) {
    var chapter models.ComicChapter
    if err := r.db.WithContext(ctx).Where("id = ?", id).First(&chapter).Error; err != nil {
        if errors.Is(err, gorm.ErrRecordNotFound) {
            return nil, fmt.Errorf("漫画章节不存在")
        }
        return nil, fmt.Errorf("查询漫画章节失败: %w", err)
    }
    return &chapter, nil
}

// GetByComicID 根据漫画ID获取章节列表
func (r *comicChapterRepository) GetByComicID(ctx context.Context, comicID int64, params *ComicChapterQueryParams) ([]*models.ComicChapter, int64, error) {
    var chapters []*models.ComicChapter
    var total int64
    
    // 构建查询
    query := r.db.WithContext(ctx).Model(&models.ComicChapter{}).
        Where("comic_id = ?", comicID)
    
    // 应用筛选条件
    if params.Status != "" {
        query = query.Where("status = ?", params.Status)
    }
    if params.IsVip != nil {
        query = query.Where("is_vip = ?", *params.IsVip)
    }
    if params.IsActive != nil {
        query = query.Where("is_active = ?", *params.IsActive)
    }
    
    // 获取总数
    if err := query.Count(&total).Error; err != nil {
        return nil, 0, fmt.Errorf("查询章节总数失败: %w", err)
    }
    
    // 应用排序
    orderBy := "sort_order"
    if params.OrderBy != "" {
        // 验证排序字段，防止SQL注入
        validOrderBy := map[string]bool{
            "sort_order": true,
            "created_at": true,
            "updated_at": true,
        }
        if validOrderBy[params.OrderBy] {
            orderBy = params.OrderBy
        }
    }
    orderDir := "asc"
    if params.OrderDir == "desc" {
        orderDir = "desc"
    }
    query = query.Order(orderBy + " " + orderDir)
    
    // 应用分页
    page := params.Page
    if page < 1 {
        page = 1
    }
    limit := params.Limit
    if limit < 1 {
        limit = 20
    }
    if limit > 100 {
        limit = 100
    }
    offset := (page - 1) * limit
    query = query.Offset(offset).Limit(limit)
    
    // 执行查询
    if err := query.Find(&chapters).Error; err != nil {
        return nil, 0, fmt.Errorf("查询章节列表失败: %w", err)
    }
    
    return chapters, total, nil
}

// Update 更新漫画章节
func (r *comicChapterRepository) Update(ctx context.Context, chapter *models.ComicChapter) error {
    if err := r.db.WithContext(ctx).Save(chapter).Error; err != nil {
        return fmt.Errorf("更新漫画章节失败: %w", err)
    }
    return nil
}

// Delete 删除漫画章节
func (r *comicChapterRepository) Delete(ctx context.Context, id int64) error {
    result := r.db.WithContext(ctx).Delete(&models.ComicChapter{}, id)
    if result.Error != nil {
        return fmt.Errorf("删除漫画章节失败: %w", result.Error)
    }
    if result.RowsAffected == 0 {
        return fmt.Errorf("漫画章节不存在")
    }
    return nil
}

// UpdateStatus 更新章节状态
func (r *comicChapterRepository) UpdateStatus(ctx context.Context, id int64, status string) error {
    result := r.db.WithContext(ctx).Model(&models.ComicChapter{}).
        Where("id = ?", id).
        Update("status", status)
    if result.Error != nil {
        return fmt.Errorf("更新漫画章节状态失败: %w", result.Error)
    }
    if result.RowsAffected == 0 {
        return fmt.Errorf("漫画章节不存在")
    }
    return nil
}
```
