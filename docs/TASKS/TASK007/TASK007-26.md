### 4. 实现ComicRepository接口

```go
// comicRepository ComicRepository接口实现
type comicRepository struct {
    db *gorm.DB
}

// NewComicRepository 创建ComicRepository实例
func NewComicRepository(db *gorm.DB) ComicRepository {
    return &comicRepository{db: db}
}

// Create 创建漫画
func (r *comicRepository) Create(ctx context.Context, comic *models.Comic) error {
    if err := r.db.WithContext(ctx).Create(comic).Error; err != nil {
        return fmt.Errorf("创建漫画失败: %w", err)
    }
    return nil
}

// GetByID 根据ID获取漫画
func (r *comicRepository) GetByID(ctx context.Context, id int64) (*models.Comic, error) {
    var comic models.Comic
    if err := r.db.WithContext(ctx).Where("id = ?", id).First(&comic).Error; err != nil {
        if errors.Is(err, gorm.ErrRecordNotFound) {
            return nil, fmt.Errorf("漫画不存在")
        }
        return nil, fmt.Errorf("查询漫画失败: %w", err)
    }
    return &comic, nil
}

// GetAll 获取漫画列表（支持分页、筛选、排序）
func (r *comicRepository) GetAll(ctx context.Context, params *ComicQueryParams) ([]*models.Comic, int64, error) {
    var comics []*models.Comic
    var total int64
    
    // 构建查询
    query := r.db.WithContext(ctx).Model(&models.Comic{})
    
    // 应用筛选条件
    if params.Title != "" {
        query = query.Where("title ILIKE ?", "%"+params.Title+"%")
    }
    if params.AuthorID != nil {
        query = query.Where("author_id = ?", *params.AuthorID)
    }
    if params.RegionID != nil {
        query = query.Where("region_id = ?", *params.RegionID)
    }
    if len(params.TagIDs) > 0 {
        // PostgreSQL数组包含查询
        query = query.Where("tag_ids && ?", pq.Array(params.TagIDs))
    }
    if params.Status != "" {
        query = query.Where("status = ?", params.Status)
    }
    if params.SerialStatus != "" {
        query = query.Where("serial_status = ?", params.SerialStatus)
    }
    if params.IsActive != nil {
        query = query.Where("is_active = ?", *params.IsActive)
    }
    
    // 获取总数
    if err := query.Count(&total).Error; err != nil {
        return nil, 0, fmt.Errorf("查询漫画总数失败: %w", err)
    }
    
    // 应用排序
    orderBy := "created_at"
    if params.OrderBy != "" {
        // 验证排序字段，防止SQL注入
        validOrderBy := map[string]bool{
            "title":      true,
            "created_at": true,
            "updated_at": true,
            "rating":     true,
        }
        if validOrderBy[params.OrderBy] {
            orderBy = params.OrderBy
        }
    }
    orderDir := "desc"
    if params.OrderDir == "asc" {
        orderDir = "asc"
    }
    query = query.Order(orderBy + " " + orderDir)
    
    // 应用分页
    page := params.Page
    if page < 1 {
        page = 1
    }
    limit := params.Limit
    if limit < 1 {
        limit = 20
    }
    if limit > 100 {
        limit = 100
    }
    offset := (page - 1) * limit
    query = query.Offset(offset).Limit(limit)
    
    // 执行查询
    if err := query.Find(&comics).Error; err != nil {
        return nil, 0, fmt.Errorf("查询漫画列表失败: %w", err)
    }
    
    return comics, total, nil
}

// Update 更新漫画
func (r *comicRepository) Update(ctx context.Context, comic *models.Comic) error {
    if err := r.db.WithContext(ctx).Save(comic).Error; err != nil {
        return fmt.Errorf("更新漫画失败: %w", err)
    }
    return nil
}

// Delete 删除漫画
func (r *comicRepository) Delete(ctx context.Context, id int64) error {
    result := r.db.WithContext(ctx).Delete(&models.Comic{}, id)
    if result.Error != nil {
        return fmt.Errorf("删除漫画失败: %w", result.Error)
    }
    if result.RowsAffected == 0 {
        return fmt.Errorf("漫画不存在")
    }
    return nil
}

// GetByAuthorID 根据作者ID获取漫画列表
func (r *comicRepository) GetByAuthorID(ctx context.Context, authorID int64, params *ComicQueryParams) ([]*models.Comic, int64, error) {
    params.AuthorID = &authorID
    return r.GetAll(ctx, params)
}

// GetByRegionID 根据地区ID获取漫画列表
func (r *comicRepository) GetByRegionID(ctx context.Context, regionID int64, params *ComicQueryParams) ([]*models.Comic, int64, error) {
    params.RegionID = &regionID
    return r.GetAll(ctx, params)
}

// GetByTagID 根据标签ID获取漫画列表
func (r *comicRepository) GetByTagID(ctx context.Context, tagID int64, params *ComicQueryParams) ([]*models.Comic, int64, error) {
    params.TagIDs = []int64{tagID}
    return r.GetAll(ctx, params)
}

// UpdateStatus 更新漫画状态
func (r *comicRepository) UpdateStatus(ctx context.Context, id int64, status string) error {
    result := r.db.WithContext(ctx).Model(&models.Comic{}).
        Where("id = ?", id).
        Update("status", status)
    if result.Error != nil {
        return fmt.Errorf("更新漫画状态失败: %w", result.Error)
    }
    if result.RowsAffected == 0 {
        return fmt.Errorf("漫画不存在")
    }
    return nil
}
```
