import (

// VideoSeasonService 视频分季服务接口

### 2. 实现VideoSeasonService接口

在同一个文件中实现VideoSeasonService接口：

```go
// videoSeasonService VideoSeasonService的实现
type videoSeasonService struct {
    seasonRepo repository.VideoSeasonRepository
    videoRepo  repository.VideoRepository
    videoSvc   VideoService
}

// NewVideoSeasonService 创建VideoSeasonService实例
func NewVideoSeasonService(
    seasonRepo repository.VideoSeasonRepository,
    videoRepo repository.VideoRepository,
    videoSvc VideoService,
) VideoSeasonService {
    return &videoSeasonService{
        seasonRepo: seasonRepo,
        videoRepo:  videoRepo,
        videoSvc:   videoSvc,
    }
}

// GetSeason 根据ID获取分季
func (s *videoSeasonService) GetSeason(ctx context.Context, id int64) (*models.VideoSeason, error) {
    if id <= 0 {
        return nil, errors.New("分季ID无效")
    }
    
    season, err := s.seasonRepo.GetByID(ctx, id)
    if err != nil {
        return nil, fmt.Errorf("获取分季失败: %w", err)
    }
    
    return season, nil
}

// GetSeasons 获取分季列表
func (s *videoSeasonService) GetSeasons(ctx context.Context, videoID int64) ([]*models.VideoSeason, error) {
    if videoID <= 0 {
        return nil, errors.New("视频ID无效")
    }
    
    // 验证视频是否存在
    _, err := s.videoRepo.GetByID(ctx, videoID)
    if err != nil {
        return nil, errors.New("视频不存在")
    }
    
    seasons, err := s.seasonRepo.GetByVideoID(ctx, videoID)
    if err != nil {
        return nil, fmt.Errorf("获取分季列表失败: %w", err)
    }
    
    return seasons, nil
}

// CreateSeason 创建分季
func (s *videoSeasonService) CreateSeason(ctx context.Context, season *models.VideoSeason) error {
    // 业务验证
    if err := s.validateSeason(season); err != nil {
        return err
    }
    
    // 验证视频是否存在
    _, err := s.videoRepo.GetByID(ctx, season.VideoID)
    if err != nil {
        return errors.New("视频不存在")
    }
    
    // 设置默认值
    if season.Status == "" {
        season.Status = "draft"
    }
    if season.SortOrder == 0 {
        // 获取当前最大排序值
        seasons, err := s.seasonRepo.GetByVideoID(ctx, season.VideoID)
        if err == nil && len(seasons) > 0 {
            maxOrder := 0
            for _, s := range seasons {
                if s.SortOrder > maxOrder {
                    maxOrder = s.SortOrder
                }
            }
            season.SortOrder = maxOrder + 1
        } else {
            season.SortOrder = 1
        }
    }
    
    // 创建分季
    if err := s.seasonRepo.Create(ctx, season); err != nil {
        return fmt.Errorf("创建分季失败: %w", err)
    }
    
    return nil
}

// UpdateSeason 更新分季
func (s *videoSeasonService) UpdateSeason(ctx context.Context, season *models.VideoSeason) error {
    // 业务验证
    if err := s.validateSeason(season); err != nil {
        return err
    }
    
    // 检查分季是否存在
    existing, err := s.seasonRepo.GetByID(ctx, season.ID)
    if err != nil {
        return errors.New("分季不存在")
    }
    
    // 验证视频ID是否改变（不允许改变）
    if existing.VideoID != season.VideoID {
        return errors.New("不能修改分季所属的视频")
    }
    
    // 更新分季
    if err := s.seasonRepo.Update(ctx, season); err != nil {
        return fmt.Errorf("更新分季失败: %w", err)
    }
    
    return nil
}

// DeleteSeason 删除分季
func (s *videoSeasonService) DeleteSeason(ctx context.Context, id int64) error {
    if id <= 0 {
        return errors.New("分季ID无效")
    }
    
    // 检查分季是否存在
    _, err := s.seasonRepo.GetByID(ctx, id)
    if err != nil {
        return errors.New("分季不存在")
    }
    
    // 删除分季
    if err := s.seasonRepo.Delete(ctx, id); err != nil {
        return fmt.Errorf("删除分季失败: %w", err)
    }
    
    return nil
}

// UpdateStatus 更新分季状态
func (s *videoSeasonService) UpdateStatus(ctx context.Context, id int64, status string) error {
    if id <= 0 {
        return errors.New("分季ID无效")
    }
    
    // 验证状态
    validStatuses := []string{"draft", "reviewing", "published", "archived"}
    isValid := false
    for _, validStatus := range validStatuses {
        if status == validStatus {
            isValid = true
            break
        }
    }
    if !isValid {
        return errors.New("无效的分季状态")
    }
    
    // 检查分季是否存在
    _, err := s.seasonRepo.GetByID(ctx, id)
    if err != nil {
        return errors.New("分季不存在")
    }
    
    // 更新状态
    if err := s.seasonRepo.UpdateStatus(ctx, id, status); err != nil {
        return fmt.Errorf("更新分季状态失败: %w", err)
    }
    
    return nil
}

// validateSeason 验证分季数据
func (s *videoSeasonService) validateSeason(season *models.VideoSeason) error {
    if season.VideoID <= 0 {
        return errors.New("视频ID无效")
    }
    
    if season.SeasonName == "" {
        return errors.New("分季名称不能为空")
    }
    if len(season.SeasonName) > 200 {
        return errors.New("分季名称长度不能超过200个字符")
    }
    
    // 验证状态
    if season.Status != "" {
        validStatuses := []string{"draft", "reviewing", "published", "archived"}
        isValid := false
        for _, validStatus := range validStatuses {
            if season.Status == validStatus {
                isValid = true
                break
            }
        }
        if !isValid {
            return errors.New("无效的分季状态")
        }
    }
    
    return nil
}
```
