### 1. 创建AudioService接口文件

在 `internal/service/` 目录下创建 `audio_service.go` 文件，定义AudioService接口：

```go
package service

import (
    "context"
    "yourproject/internal/models"
)

// AudioService 音频服务接口
// 提供音频管理的业务逻辑
type AudioService interface {
    // GetAudio 根据ID获取音频
    GetAudio(ctx context.Context, id int64) (*models.Audio, error)
    
    // GetAudios 获取音频列表（支持分页、筛选、排序）
    GetAudios(ctx context.Context, params *AudioQueryParams) ([]*models.Audio, int64, error)
    
    // CreateAudio 创建音频
    CreateAudio(ctx context.Context, audio *models.Audio) error
    
    // UpdateAudio 更新音频
    UpdateAudio(ctx context.Context, audio *models.Audio) error
    
    // DeleteAudio 删除音频
    DeleteAudio(ctx context.Context, id int64) error
}

// AudioQueryParams 音频查询参数
type AudioQueryParams struct {
    Page          int     // 页码，从1开始
    Limit         int     // 每页数量
    Title         string  // 标题筛选（模糊匹配）
    AuthorID      *int64  // 作者ID筛选
    CategoryIDs   []int64 // 分类ID筛选
    TagIDs        []int64 // 标签ID筛选
    Status        string  // 状态筛选
    SerialStatus  string  // 连载状态筛选
    IsActive      *bool   // 是否启用筛选
    OrderBy       string  // 排序字段（title、created_at、updated_at、duration、rating）
    OrderDir      string  // 排序方向（asc、desc）
}
```

### 2. 实现AudioService接口

在同一个文件中实现AudioService接口：

```go
// audioService AudioService的实现
type audioService struct {
    audioRepo   repository.AudioRepository
    categorySvc CategoryService
    tagSvc      TagService
}

// NewAudioService 创建AudioService实例
func NewAudioService(
    audioRepo repository.AudioRepository,
    categorySvc CategoryService,
    tagSvc TagService,
) AudioService {
    return &audioService{
        audioRepo:   audioRepo,
        categorySvc: categorySvc,
        tagSvc:      tagSvc,
    }
}

// GetAudio 根据ID获取音频
func (s *audioService) GetAudio(ctx context.Context, id int64) (*models.Audio, error) {
    if id <= 0 {
        return nil, errors.New("音频ID无效")
    }
    
    audio, err := s.audioRepo.GetByID(ctx, id)
    if err != nil {
        return nil, fmt.Errorf("获取音频失败: %w", err)
    }
    
    return audio, nil
}

// GetAudios 获取音频列表
func (s *audioService) GetAudios(ctx context.Context, params *AudioQueryParams) ([]*models.Audio, int64, error) {
    // 构建查询参数
    queryParams := &repository.AudioQueryParams{
        Page:         params.Page,
        Limit:        params.Limit,
        Title:        params.Title,
        AuthorID:     params.AuthorID,
        CategoryIDs:  params.CategoryIDs,
        TagIDs:       params.TagIDs,
        Status:       params.Status,
        SerialStatus: params.SerialStatus,
        IsActive:     params.IsActive,
        OrderBy:      params.OrderBy,
        OrderDir:     params.OrderDir,
    }
    
    // 验证和设置默认值
    if queryParams.Page < 1 {
        queryParams.Page = 1
    }
    if queryParams.Limit < 1 {
        queryParams.Limit = 20
    }
    if queryParams.Limit > 100 {
        queryParams.Limit = 100
    }
    if queryParams.OrderBy == "" {
        queryParams.OrderBy = "created_at"
    }
    if queryParams.OrderDir == "" {
        queryParams.OrderDir = "desc"
    }
    
    audios, total, err := s.audioRepo.GetAll(ctx, queryParams)
    if err != nil {
        return nil, 0, fmt.Errorf("获取音频列表失败: %w", err)
    }
    
    return audios, total, nil
}

// CreateAudio 创建音频
func (s *audioService) CreateAudio(ctx context.Context, audio *models.Audio) error {
    // 业务验证
    if err := s.validateAudio(audio); err != nil {
        return err
    }
    
    // 验证分类ID
    if len(audio.CategoryIDs) > 0 {
        for _, categoryID := range audio.CategoryIDs {
            category, err := s.categorySvc.GetCategory(ctx, categoryID)
            if err != nil {
                return fmt.Errorf("分类ID %d 不存在", categoryID)
            }
            // 验证分类类型
            if category.CategoryType != "audio" {
                return fmt.Errorf("分类ID %d 不是音频分类", categoryID)
            }
        }
    }
    
    // 验证标签ID
    if len(audio.TagIDs) > 0 {
        for _, tagID := range audio.TagIDs {
            _, err := s.tagSvc.GetTag(ctx, tagID)
            if err != nil {
                return fmt.Errorf("标签ID %d 不存在", tagID)
            }
        }
    }
    
    // 设置默认值
    if audio.Status == "" {
        audio.Status = "draft"
    }
    if audio.SerialStatus == "" {
        audio.SerialStatus = "serializing"
    }
    if audio.Duration == 0 {
        audio.Duration = 0
    }
    if audio.Rating == 0 {
        audio.Rating = 0
    }
    if audio.IsActive == nil {
        active := true
        audio.IsActive = &active
    }
    
    // 创建音频
    if err := s.audioRepo.Create(ctx, audio); err != nil {
        return fmt.Errorf("创建音频失败: %w", err)
    }
    
    return nil
}

// UpdateAudio 更新音频
func (s *audioService) UpdateAudio(ctx context.Context, audio *models.Audio) error {
    // 业务验证
    if err := s.validateAudio(audio); err != nil {
        return err
    }
    
    // 检查音频是否存在
    existing, err := s.audioRepo.GetByID(ctx, audio.ID)
    if err != nil {
        return errors.New("音频不存在")
    }
    
    // 验证分类ID
    if len(audio.CategoryIDs) > 0 {
        for _, categoryID := range audio.CategoryIDs {
            category, err := s.categorySvc.GetCategory(ctx, categoryID)
            if err != nil {
                return fmt.Errorf("分类ID %d 不存在", categoryID)
            }
            // 验证分类类型
            if category.CategoryType != "audio" {
                return fmt.Errorf("分类ID %d 不是音频分类", categoryID)
            }
        }
    }
    
    // 验证标签ID
    if len(audio.TagIDs) > 0 {
        for _, tagID := range audio.TagIDs {
            _, err := s.tagSvc.GetTag(ctx, tagID)
            if err != nil {
                return fmt.Errorf("标签ID %d 不存在", tagID)
            }
        }
    }
    
    // 更新音频
    if err := s.audioRepo.Update(ctx, audio); err != nil {
        return fmt.Errorf("更新音频失败: %w", err)
    }
    
    return nil
}

// DeleteAudio 删除音频
func (s *audioService) DeleteAudio(ctx context.Context, id int64) error {
    if id <= 0 {
        return errors.New("音频ID无效")
    }
    
    // 检查音频是否存在
    audio, err := s.audioRepo.GetByID(ctx, id)
    if err != nil {
        return errors.New("音频不存在")
    }
    
    // 删除音频
    if err := s.audioRepo.Delete(ctx, id); err != nil {
        return fmt.Errorf("删除音频失败: %w", err)
    }
    
    return nil
}

// validateAudio 验证音频数据
func (s *audioService) validateAudio(audio *models.Audio) error {
    if audio.Title == "" {
        return errors.New("音频标题不能为空")
    }
    if len(audio.Title) > 200 {
        return errors.New("音频标题长度不能超过200个字符")
    }
    
    if audio.AuthorID <= 0 {
        return errors.New("作者ID无效")
    }
    
    // 验证状态
    validStatuses := []string{"draft", "reviewing", "published", "archived"}
    if audio.Status != "" {
        isValid := false
        for _, validStatus := range validStatuses {
            if audio.Status == validStatus {
                isValid = true
                break
            }
        }
        if !isValid {
            return errors.New("无效的音频状态")
        }
    }
    
    // 验证连载状态
    validSerialStatuses := []string{"serializing", "completed", "suspended"}
    if audio.SerialStatus != "" {
        isValid := false
        for _, validStatus := range validSerialStatuses {
            if audio.SerialStatus == validStatus {
                isValid = true
                break
            }
        }
        if !isValid {
            return errors.New("无效的连载状态")
        }
    }
    
    return nil
}
```
