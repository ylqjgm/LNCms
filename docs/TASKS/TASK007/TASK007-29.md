### 4. 实现AudioRepository接口

```go
// audioRepository AudioRepository接口实现
type audioRepository struct {
    db *gorm.DB
}

// NewAudioRepository 创建AudioRepository实例
func NewAudioRepository(db *gorm.DB) AudioRepository {
    return &audioRepository{db: db}
}

// Create 创建音频
func (r *audioRepository) Create(ctx context.Context, audio *models.Audio) error {
    if err := r.db.WithContext(ctx).Create(audio).Error; err != nil {
        return fmt.Errorf("创建音频失败: %w", err)
    }
    return nil
}

// GetByID 根据ID获取音频
func (r *audioRepository) GetByID(ctx context.Context, id int64) (*models.Audio, error) {
    var audio models.Audio
    if err := r.db.WithContext(ctx).Where("id = ?", id).First(&audio).Error; err != nil {
        if errors.Is(err, gorm.ErrRecordNotFound) {
            return nil, fmt.Errorf("音频不存在")
        }
        return nil, fmt.Errorf("查询音频失败: %w", err)
    }
    return &audio, nil
}

// GetAll 获取音频列表（支持分页、筛选、排序）
func (r *audioRepository) GetAll(ctx context.Context, params *AudioQueryParams) ([]*models.Audio, int64, error) {
    var audios []*models.Audio
    var total int64
    
    // 构建查询
    query := r.db.WithContext(ctx).Model(&models.Audio{})
    
    // 应用筛选条件
    if params.Title != "" {
        query = query.Where("title ILIKE ?", "%"+params.Title+"%")
    }
    if params.AuthorID != nil {
        query = query.Where("author_id = ?", *params.AuthorID)
    }
    if len(params.CategoryIDs) > 0 {
        // PostgreSQL数组包含查询
        query = query.Where("category_ids && ?", pq.Array(params.CategoryIDs))
    }
    if len(params.TagIDs) > 0 {
        // PostgreSQL数组包含查询
        query = query.Where("tag_ids && ?", pq.Array(params.TagIDs))
    }
    if params.Status != "" {
        query = query.Where("status = ?", params.Status)
    }
    if params.SerialStatus != "" {
        query = query.Where("serial_status = ?", params.SerialStatus)
    }
    if params.IsActive != nil {
        query = query.Where("is_active = ?", *params.IsActive)
    }
    
    // 获取总数
    if err := query.Count(&total).Error; err != nil {
        return nil, 0, fmt.Errorf("查询音频总数失败: %w", err)
    }
    
    // 应用排序
    orderBy := "created_at"
    if params.OrderBy != "" {
        // 验证排序字段，防止SQL注入
        validOrderBy := map[string]bool{
            "title":      true,
            "created_at": true,
            "updated_at": true,
            "duration":   true,
            "rating":     true,
        }
        if validOrderBy[params.OrderBy] {
            orderBy = params.OrderBy
        }
    }
    orderDir := "desc"
    if params.OrderDir == "asc" {
        orderDir = "asc"
    }
    query = query.Order(orderBy + " " + orderDir)
    
    // 应用分页
    page := params.Page
    if page < 1 {
        page = 1
    }
    limit := params.Limit
    if limit < 1 {
        limit = 20
    }
    if limit > 100 {
        limit = 100
    }
    offset := (page - 1) * limit
    query = query.Offset(offset).Limit(limit)
    
    // 执行查询
    if err := query.Find(&audios).Error; err != nil {
        return nil, 0, fmt.Errorf("查询音频列表失败: %w", err)
    }
    
    return audios, total, nil
}

// Update 更新音频
func (r *audioRepository) Update(ctx context.Context, audio *models.Audio) error {
    if err := r.db.WithContext(ctx).Save(audio).Error; err != nil {
        return fmt.Errorf("更新音频失败: %w", err)
    }
    return nil
}

// Delete 删除音频
func (r *audioRepository) Delete(ctx context.Context, id int64) error {
    result := r.db.WithContext(ctx).Delete(&models.Audio{}, id)
    if result.Error != nil {
        return fmt.Errorf("删除音频失败: %w", result.Error)
    }
    if result.RowsAffected == 0 {
        return fmt.Errorf("音频不存在")
    }
    return nil
}

// GetByAuthorID 根据作者ID获取音频列表
func (r *audioRepository) GetByAuthorID(ctx context.Context, authorID int64, params *AudioQueryParams) ([]*models.Audio, int64, error) {
    params.AuthorID = &authorID
    return r.GetAll(ctx, params)
}

// GetByCategoryID 根据分类ID获取音频列表
func (r *audioRepository) GetByCategoryID(ctx context.Context, categoryID int64, params *AudioQueryParams) ([]*models.Audio, int64, error) {
    params.CategoryIDs = []int64{categoryID}
    return r.GetAll(ctx, params)
}

// GetByTagID 根据标签ID获取音频列表
func (r *audioRepository) GetByTagID(ctx context.Context, tagID int64, params *AudioQueryParams) ([]*models.Audio, int64, error) {
    params.TagIDs = []int64{tagID}
    return r.GetAll(ctx, params)
}

// UpdateStatus 更新音频状态
func (r *audioRepository) UpdateStatus(ctx context.Context, id int64, status string) error {
    result := r.db.WithContext(ctx).Model(&models.Audio{}).
        Where("id = ?", id).
        Update("status", status)
    if result.Error != nil {
        return fmt.Errorf("更新音频状态失败: %w", result.Error)
    }
    if result.RowsAffected == 0 {
        return fmt.Errorf("音频不存在")
    }
    return nil
}

// UpdateDuration 更新音频时长
func (r *audioRepository) UpdateDuration(ctx context.Context, id int64, duration time.Duration) error {
    // 将Duration转换为秒数（或根据数据库设计转换为其他格式）
    durationSeconds := int64(duration.Seconds())
    result := r.db.WithContext(ctx).Model(&models.Audio{}).
        Where("id = ?", id).
        Update("duration", durationSeconds)
    if result.Error != nil {
        return fmt.Errorf("更新音频时长失败: %w", result.Error)
    }
    if result.RowsAffected == 0 {
        return fmt.Errorf("音频不存在")
    }
    return nil
}
```
