### 3. 实现VideoFileRepository接口

在同一个文件中实现VideoFileRepository接口：

```go
// videoFileRepository VideoFileRepository的实现
type videoFileRepository struct {
    db *gorm.DB
}

// NewVideoFileRepository 创建VideoFileRepository实例
func NewVideoFileRepository(db *gorm.DB) VideoFileRepository {
    return &videoFileRepository{db: db}
}

// Create 创建视频文件
func (r *videoFileRepository) Create(ctx context.Context, videoFile *models.VideoFile) error {
    if err := r.db.WithContext(ctx).Create(videoFile).Error; err != nil {
        return fmt.Errorf("创建视频文件失败: %w", err)
    }
    return nil
}

// GetByID 根据ID获取视频文件
func (r *videoFileRepository) GetByID(ctx context.Context, id int64) (*models.VideoFile, error) {
    var videoFile models.VideoFile
    if err := r.db.WithContext(ctx).Where("id = ?", id).First(&videoFile).Error; err != nil {
        if errors.Is(err, gorm.ErrRecordNotFound) {
            return nil, errors.New("视频文件不存在")
        }
        return nil, fmt.Errorf("查询视频文件失败: %w", err)
    }
    return &videoFile, nil
}

// GetByEpisodeID 根据分集ID获取视频文件列表
func (r *videoFileRepository) GetByEpisodeID(ctx context.Context, episodeID int64) ([]*models.VideoFile, error) {
    var videoFiles []*models.VideoFile
    if err := r.db.WithContext(ctx).
        Where("episode_id = ?", episodeID).
        Order("id ASC").
        Find(&videoFiles).Error; err != nil {
        return nil, fmt.Errorf("查询视频文件列表失败: %w", err)
    }
    return videoFiles, nil
}

// Update 更新视频文件
func (r *videoFileRepository) Update(ctx context.Context, videoFile *models.VideoFile) error {
    if err := r.db.WithContext(ctx).Save(videoFile).Error; err != nil {
        return fmt.Errorf("更新视频文件失败: %w", err)
    }
    return nil
}

// Delete 删除视频文件
func (r *videoFileRepository) Delete(ctx context.Context, id int64) error {
    result := r.db.WithContext(ctx).Delete(&models.VideoFile{}, id)
    if result.Error != nil {
        return fmt.Errorf("删除视频文件失败: %w", result.Error)
    }
    if result.RowsAffected == 0 {
        return errors.New("视频文件不存在")
    }
    return nil
}
```
