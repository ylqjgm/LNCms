import (

// RegionRepository 地区仓储接口
// 提供地区数据的CRUD操作和树形结构查询
type RegionRepository interface {
    // Create 创建地区
    Create(ctx context.Context, region *models.Region) error
    
    // GetByID 根据ID获取地区
    GetByID(ctx context.Context, id int64) (*models.Region, error)

    // Update 更新地区

    // Delete 删除地区

### 2. 创建Repository实现文件

在 `internal/repository/` 目录下创建 `region_repository_impl.go` 文件，实现RegionRepository接口：

```go
package repository

import (
    "context"
    "errors"
    "fmt"
    "gorm.io/gorm"
    "yourproject/internal/models"
)

// regionRepository RegionRepository的实现
type regionRepository struct {
    db *gorm.DB
}

// NewRegionRepository 创建RegionRepository实例
func NewRegionRepository(db *gorm.DB) RegionRepository {
    return &regionRepository{
        db: db,
    }
}

// Create 创建地区
func (r *regionRepository) Create(ctx context.Context, region *models.Region) error {
    if err := r.db.WithContext(ctx).Create(region).Error; err != nil {
        return fmt.Errorf("创建地区失败: %w", err)
    }
    return nil
}

// GetByID 根据ID获取地区
func (r *regionRepository) GetByID(ctx context.Context, id int64) (*models.Region, error) {
    var region models.Region
    if err := r.db.WithContext(ctx).Where("id = ?", id).First(&region).Error; err != nil {
        if errors.Is(err, gorm.ErrRecordNotFound) {
            return nil, errors.New("地区不存在")
        }
        return nil, fmt.Errorf("查询地区失败: %w", err)
    }
    return &region, nil
}

// GetAll 获取所有地区
func (r *regionRepository) GetAll(ctx context.Context) ([]*models.Region, error) {
    var regions []*models.Region
    if err := r.db.WithContext(ctx).Order("sort_order ASC, id ASC").Find(&regions).Error; err != nil {
        return nil, fmt.Errorf("查询地区列表失败: %w", err)
    }
    return regions, nil
}

// Update 更新地区
func (r *regionRepository) Update(ctx context.Context, region *models.Region) error {
    if err := r.db.WithContext(ctx).Save(region).Error; err != nil {
        return fmt.Errorf("更新地区失败: %w", err)
    }
    return nil
}

// Delete 删除地区
func (r *regionRepository) Delete(ctx context.Context, id int64) error {
    // 检查是否存在子地区
    var count int64
    if err := r.db.WithContext(ctx).Model(&models.Region{}).Where("parent_id = ?", id).Count(&count).Error; err != nil {
        return fmt.Errorf("检查子地区失败: %w", err)
    }
    if count > 0 {
        return errors.New("存在子地区，无法删除")
    }
    
    if err := r.db.WithContext(ctx).Delete(&models.Region{}, id).Error; err != nil {
        return fmt.Errorf("删除地区失败: %w", err)
    }
    return nil
}

// GetTree 获取地区树（所有地区的树形结构）
func (r *regionRepository) GetTree(ctx context.Context) ([]*models.Region, error) {
    // 获取所有地区
    allRegions, err := r.GetAll(ctx)
    if err != nil {
        return nil, err
    }
    
    // 构建地区树
    return r.buildTree(allRegions, nil), nil
}

// GetChildren 获取指定地区的所有子地区（递归）
func (r *regionRepository) GetChildren(ctx context.Context, parentID int64) ([]*models.Region, error) {
    var children []*models.Region
    
    // 递归查询所有子地区
    if err := r.getChildrenRecursive(ctx, parentID, &children); err != nil {
        return nil, err
    }
    
    return children, nil
}

// getChildrenRecursive 递归获取子地区
func (r *regionRepository) getChildrenRecursive(ctx context.Context, parentID int64, result *[]*models.Region) error {
    var directChildren []*models.Region
    if err := r.db.WithContext(ctx).Where("parent_id = ?", parentID).Order("sort_order ASC, id ASC").Find(&directChildren).Error; err != nil {
        return fmt.Errorf("查询子地区失败: %w", err)
    }
    
    for _, child := range directChildren {
        *result = append(*result, child)
        // 递归查询子地区的子地区
        if err := r.getChildrenRecursive(ctx, child.ID, result); err != nil {
            return err
        }
    }
    
    return nil
}

// GetByParentID 根据父地区ID获取直接子地区（非递归）
func (r *regionRepository) GetByParentID(ctx context.Context, parentID *int64) ([]*models.Region, error) {
    var regions []*models.Region
    query := r.db.WithContext(ctx)
    
    if parentID == nil {
        query = query.Where("parent_id IS NULL")
    } else {
        query = query.Where("parent_id = ?", *parentID)
    }
    
    if err := query.Order("sort_order ASC, id ASC").Find(&regions).Error; err != nil {
        return nil, fmt.Errorf("查询子地区列表失败: %w", err)
    }
    return regions, nil
}

// buildTree 构建地区树
func (r *regionRepository) buildTree(regions []*models.Region, parentID *int64) []*models.Region {
    var tree []*models.Region
    
    for _, region := range regions {
        var regParentID *int64
        if region.ParentID != nil {
            regParentID = region.ParentID
        }
        
        // 判断是否为当前父节点的子节点
        if (parentID == nil && regParentID == nil) || (parentID != nil && regParentID != nil && *parentID == *regParentID) {
            // 递归构建子树
            region.Children = r.buildTree(regions, &region.ID)
            tree = append(tree, region)
        }
    }
    
    return tree
}
```
