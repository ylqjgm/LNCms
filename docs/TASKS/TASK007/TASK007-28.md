### 3. 实现ComicChapterImageRepository接口

```go
// comicChapterImageRepository ComicChapterImageRepository接口实现
type comicChapterImageRepository struct {
    db *gorm.DB
}

// NewComicChapterImageRepository 创建ComicChapterImageRepository实例
func NewComicChapterImageRepository(db *gorm.DB) ComicChapterImageRepository {
    return &comicChapterImageRepository{db: db}
}

// Create 创建漫画章节图片
func (r *comicChapterImageRepository) Create(ctx context.Context, image *models.ComicChapterImage) error {
    if err := r.db.WithContext(ctx).Create(image).Error; err != nil {
        return fmt.Errorf("创建漫画章节图片失败: %w", err)
    }
    return nil
}

// GetByID 根据ID获取漫画章节图片
func (r *comicChapterImageRepository) GetByID(ctx context.Context, id int64) (*models.ComicChapterImage, error) {
    var image models.ComicChapterImage
    if err := r.db.WithContext(ctx).Where("id = ?", id).First(&image).Error; err != nil {
        if errors.Is(err, gorm.ErrRecordNotFound) {
            return nil, fmt.Errorf("漫画章节图片不存在")
        }
        return nil, fmt.Errorf("查询漫画章节图片失败: %w", err)
    }
    return &image, nil
}

// GetByChapterID 根据章节ID获取图片列表
func (r *comicChapterImageRepository) GetByChapterID(ctx context.Context, chapterID int64) ([]*models.ComicChapterImage, error) {
    var images []*models.ComicChapterImage
    if err := r.db.WithContext(ctx).
        Where("chapter_id = ?", chapterID).
        Order("sort_order ASC").
        Find(&images).Error; err != nil {
        return nil, fmt.Errorf("查询漫画章节图片列表失败: %w", err)
    }
    return images, nil
}

// Update 更新漫画章节图片
func (r *comicChapterImageRepository) Update(ctx context.Context, image *models.ComicChapterImage) error {
    if err := r.db.WithContext(ctx).Save(image).Error; err != nil {
        return fmt.Errorf("更新漫画章节图片失败: %w", err)
    }
    return nil
}

// Delete 删除漫画章节图片
func (r *comicChapterImageRepository) Delete(ctx context.Context, id int64) error {
    result := r.db.WithContext(ctx).Delete(&models.ComicChapterImage{}, id)
    if result.Error != nil {
        return fmt.Errorf("删除漫画章节图片失败: %w", result.Error)
    }
    if result.RowsAffected == 0 {
        return fmt.Errorf("漫画章节图片不存在")
    }
    return nil
}

// UpdateSortOrder 更新图片排序
func (r *comicChapterImageRepository) UpdateSortOrder(ctx context.Context, id int64, sortOrder int) error {
    result := r.db.WithContext(ctx).Model(&models.ComicChapterImage{}).
        Where("id = ?", id).
        Update("sort_order", sortOrder)
    if result.Error != nil {
        return fmt.Errorf("更新漫画章节图片排序失败: %w", result.Error)
    }
    if result.RowsAffected == 0 {
        return fmt.Errorf("漫画章节图片不存在")
    }
    return nil
}
```
