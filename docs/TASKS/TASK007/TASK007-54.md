```go
package service

import (
    "context"
    "yourproject/internal/models"
)

// VideoEpisodeService 视频分集服务接口
// 提供视频分集管理的业务逻辑
type VideoEpisodeService interface {
    // GetEpisode 根据ID获取分集
    GetEpisode(ctx context.Context, id int64) (*models.VideoEpisode, error)
    
    // GetEpisodes 获取分集列表（支持分页、排序）
    GetEpisodes(ctx context.Context, params *VideoEpisodeQueryParams) ([]*models.VideoEpisode, int64, error)
    
    // CreateEpisode 创建分集
    CreateEpisode(ctx context.Context, episode *models.VideoEpisode) error
    
    // UpdateEpisode 更新分集
    UpdateEpisode(ctx context.Context, episode *models.VideoEpisode) error
    
    // DeleteEpisode 删除分集
    DeleteEpisode(ctx context.Context, id int64) error
    
    // UpdateStatus 更新分集状态
    UpdateStatus(ctx context.Context, id int64, status string) error
}

// VideoEpisodeQueryParams 视频分集查询参数
type VideoEpisodeQueryParams struct {
    VideoID  int64  // 视频ID（必填）
    SeasonID *int64 // 分季ID（可选）
    Page     int    // 页码，从1开始
    Limit    int    // 每页数量
    Status   string // 状态筛选
    IsVip    *bool  // 是否VIP筛选
    OrderBy  string // 排序字段（sort_order、created_at、updated_at）
    OrderDir string // 排序方向（asc、desc）
}
```

### 2. 实现VideoEpisodeService接口

在同一个文件中实现VideoEpisodeService接口：

```go
// videoEpisodeService VideoEpisodeService的实现
type videoEpisodeService struct {
    episodeRepo repository.VideoEpisodeRepository
    videoFileRepo repository.VideoFileRepository
    videoRepo    repository.VideoRepository
    seasonRepo   repository.VideoSeasonRepository
    videoSvc     VideoService
}

// NewVideoEpisodeService 创建VideoEpisodeService实例
func NewVideoEpisodeService(
    episodeRepo repository.VideoEpisodeRepository,
    videoFileRepo repository.VideoFileRepository,
    videoRepo repository.VideoRepository,
    seasonRepo repository.VideoSeasonRepository,
    videoSvc VideoService,
) VideoEpisodeService {
    return &videoEpisodeService{
        episodeRepo:  episodeRepo,
        videoFileRepo: videoFileRepo,
        videoRepo:    videoRepo,
        seasonRepo:   seasonRepo,
        videoSvc:     videoSvc,
    }
}

// GetEpisode 根据ID获取分集
func (s *videoEpisodeService) GetEpisode(ctx context.Context, id int64) (*models.VideoEpisode, error) {
    if id <= 0 {
        return nil, errors.New("分集ID无效")
    }
    
    episode, err := s.episodeRepo.GetByID(ctx, id)
    if err != nil {
        return nil, fmt.Errorf("获取分集失败: %w", err)
    }
    
    return episode, nil
}

// GetEpisodes 获取分集列表
func (s *videoEpisodeService) GetEpisodes(ctx context.Context, params *VideoEpisodeQueryParams) ([]*models.VideoEpisode, int64, error) {
    if params.VideoID <= 0 {
        return nil, 0, errors.New("视频ID无效")
    }
    
    // 验证视频是否存在
    _, err := s.videoRepo.GetByID(ctx, params.VideoID)
    if err != nil {
        return nil, 0, errors.New("视频不存在")
    }
    
    // 构建查询参数
    queryParams := &repository.VideoEpisodeQueryParams{
        Page:     params.Page,
        Limit:    params.Limit,
        Status:   params.Status,
        IsVip:    params.IsVip,
        OrderBy:  params.OrderBy,
        OrderDir: params.OrderDir,
    }
    
    // 验证和设置默认值
    if queryParams.Page < 1 {
        queryParams.Page = 1
    }
    if queryParams.Limit < 1 {
        queryParams.Limit = 20
    }
    if queryParams.Limit > 100 {
        queryParams.Limit = 100
    }
    if queryParams.OrderBy == "" {
        queryParams.OrderBy = "sort_order"
    }
    if queryParams.OrderDir == "" {
        queryParams.OrderDir = "asc"
    }
    
    var episodes []*models.VideoEpisode
    var total int64
    var err2 error
    
    if params.SeasonID != nil {
        episodes, total, err2 = s.episodeRepo.GetBySeasonID(ctx, *params.SeasonID, queryParams)
    } else {
        episodes, total, err2 = s.episodeRepo.GetByVideoID(ctx, params.VideoID, queryParams)
    }
    
    if err2 != nil {
        return nil, 0, fmt.Errorf("获取分集列表失败: %w", err2)
    }
    
    return episodes, total, nil
}

// CreateEpisode 创建分集
func (s *videoEpisodeService) CreateEpisode(ctx context.Context, episode *models.VideoEpisode) error {
    // 业务验证
    if err := s.validateEpisode(episode); err != nil {
        return err
    }
    
    // 验证视频是否存在
    _, err := s.videoRepo.GetByID(ctx, episode.VideoID)
    if err != nil {
        return errors.New("视频不存在")
    }
    
    // 验证分季是否存在（如果指定了分季）
    if episode.SeasonID != nil {
        _, err := s.seasonRepo.GetByID(ctx, *episode.SeasonID)
        if err != nil {
            return errors.New("分季不存在")
        }
    }
    
    // 设置默认值
    if episode.Status == "" {
        episode.Status = "draft"
    }
    if episode.SortOrder == 0 {
        // 获取当前最大排序值
        episodes, _, err := s.episodeRepo.GetByVideoID(ctx, episode.VideoID, &repository.VideoEpisodeQueryParams{
            Page:     1,
            Limit:    10000,
            OrderBy:  "sort_order",
            OrderDir: "desc",
        })
        if err == nil && len(episodes) > 0 {
            episode.SortOrder = episodes[0].SortOrder + 1
        } else {
            episode.SortOrder = 1
        }
    }
    
    // 创建分集
    if err := s.episodeRepo.Create(ctx, episode); err != nil {
        return fmt.Errorf("创建分集失败: %w", err)
    }
    
    // 如果分集已发布，更新视频总时长
    if episode.Status == "published" {
        if err := s.videoSvc.UpdateDuration(ctx, episode.VideoID); err != nil {
            // 记录错误但不中断流程
        }
    }
    
    return nil
}

// UpdateEpisode 更新分集
func (s *videoEpisodeService) UpdateEpisode(ctx context.Context, episode *models.VideoEpisode) error {
    // 业务验证
    if err := s.validateEpisode(episode); err != nil {
        return err
    }
    
    // 检查分集是否存在
    existing, err := s.episodeRepo.GetByID(ctx, episode.ID)
    if err != nil {
        return errors.New("分集不存在")
    }
    
    // 验证视频ID是否改变（不允许改变）
    if existing.VideoID != episode.VideoID {
        return errors.New("不能修改分集所属的视频")
    }
    
    // 更新分集
    if err := s.episodeRepo.Update(ctx, episode); err != nil {
        return fmt.Errorf("更新分集失败: %w", err)
    }
    
    // 如果状态变为已发布或从已发布变为其他状态，更新视频总时长
    if (episode.Status == "published" && existing.Status != "published") || 
       (existing.Status == "published" && episode.Status != "published") {
        if err := s.videoSvc.UpdateDuration(ctx, episode.VideoID); err != nil {
            // 记录错误但不中断流程
        }
    }
    
    return nil
}

// DeleteEpisode 删除分集
func (s *videoEpisodeService) DeleteEpisode(ctx context.Context, id int64) error {
    if id <= 0 {
        return errors.New("分集ID无效")
    }
    
    // 检查分集是否存在
    episode, err := s.episodeRepo.GetByID(ctx, id)
    if err != nil {
        return errors.New("分集不存在")
    }
    
    videoID := episode.VideoID
    
    // 删除分集
    if err := s.episodeRepo.Delete(ctx, id); err != nil {
        return fmt.Errorf("删除分集失败: %w", err)
    }
    
    // 如果分集已发布，更新视频总时长
    if episode.Status == "published" {
        if err := s.videoSvc.UpdateDuration(ctx, videoID); err != nil {
            // 记录错误但不中断流程
        }
    }
    
    return nil
}

// UpdateStatus 更新分集状态
func (s *videoEpisodeService) UpdateStatus(ctx context.Context, id int64, status string) error {
    if id <= 0 {
        return errors.New("分集ID无效")
    }
    
    // 验证状态
    validStatuses := []string{"draft", "reviewing", "published", "archived"}
    isValid := false
    for _, validStatus := range validStatuses {
        if status == validStatus {
            isValid = true
            break
        }
    }
    if !isValid {
        return errors.New("无效的分集状态")
    }
    
    // 检查分集是否存在
    episode, err := s.episodeRepo.GetByID(ctx, id)
    if err != nil {
        return errors.New("分集不存在")
    }
    
    oldStatus := episode.Status
    
    // 更新状态
    if err := s.episodeRepo.UpdateStatus(ctx, id, status); err != nil {
        return fmt.Errorf("更新分集状态失败: %w", err)
    }
    
    // 如果状态变为已发布或从已发布变为其他状态，更新视频总时长
    if (status == "published" && oldStatus != "published") || 
       (oldStatus == "published" && status != "published") {
        if err := s.videoSvc.UpdateDuration(ctx, episode.VideoID); err != nil {
            // 记录错误但不中断流程
        }
    }
    
    return nil
}

// validateEpisode 验证分集数据
func (s *videoEpisodeService) validateEpisode(episode *models.VideoEpisode) error {
    if episode.VideoID <= 0 {
        return errors.New("视频ID无效")
    }
    
    if episode.EpisodeTitle == "" {
        return errors.New("分集标题不能为空")
    }
    if len(episode.EpisodeTitle) > 200 {
        return errors.New("分集标题长度不能超过200个字符")
    }
    
    // 验证状态
    if episode.Status != "" {
        validStatuses := []string{"draft", "reviewing", "published", "archived"}
        isValid := false
        for _, validStatus := range validStatuses {
            if episode.Status == validStatus {
                isValid = true
                break
            }
        }
        if !isValid {
            return errors.New("无效的分集状态")
        }
    }
    
    return nil
}
```
