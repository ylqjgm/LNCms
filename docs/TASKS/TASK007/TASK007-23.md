### 3. 实现NovelVolumeRepository接口

```go
// novelVolumeRepository NovelVolumeRepository接口实现
type novelVolumeRepository struct {
    db *gorm.DB
}

// NewNovelVolumeRepository 创建NovelVolumeRepository实例
func NewNovelVolumeRepository(db *gorm.DB) NovelVolumeRepository {
    return &novelVolumeRepository{db: db}
}

// Create 创建小说分卷
func (r *novelVolumeRepository) Create(ctx context.Context, volume *models.NovelVolume) error {
    if err := r.db.WithContext(ctx).Create(volume).Error; err != nil {
        return fmt.Errorf("创建小说分卷失败: %w", err)
    }
    return nil
}

// GetByID 根据ID获取小说分卷
func (r *novelVolumeRepository) GetByID(ctx context.Context, id int64) (*models.NovelVolume, error) {
    var volume models.NovelVolume
    if err := r.db.WithContext(ctx).Where("id = ?", id).First(&volume).Error; err != nil {
        if errors.Is(err, gorm.ErrRecordNotFound) {
            return nil, fmt.Errorf("小说分卷不存在")
        }
        return nil, fmt.Errorf("查询小说分卷失败: %w", err)
    }
    return &volume, nil
}

// GetByNovelID 根据小说ID获取分卷列表
func (r *novelVolumeRepository) GetByNovelID(ctx context.Context, novelID int64) ([]*models.NovelVolume, error) {
    var volumes []*models.NovelVolume
    if err := r.db.WithContext(ctx).
        Where("novel_id = ?", novelID).
        Order("sort_order ASC").
        Find(&volumes).Error; err != nil {
        return nil, fmt.Errorf("查询小说分卷列表失败: %w", err)
    }
    return volumes, nil
}

// Update 更新小说分卷
func (r *novelVolumeRepository) Update(ctx context.Context, volume *models.NovelVolume) error {
    if err := r.db.WithContext(ctx).Save(volume).Error; err != nil {
        return fmt.Errorf("更新小说分卷失败: %w", err)
    }
    return nil
}

// Delete 删除小说分卷
func (r *novelVolumeRepository) Delete(ctx context.Context, id int64) error {
    result := r.db.WithContext(ctx).Delete(&models.NovelVolume{}, id)
    if result.Error != nil {
        return fmt.Errorf("删除小说分卷失败: %w", result.Error)
    }
    if result.RowsAffected == 0 {
        return fmt.Errorf("小说分卷不存在")
    }
    return nil
}

// UpdateStatus 更新分卷状态
func (r *novelVolumeRepository) UpdateStatus(ctx context.Context, id int64, status string) error {
    result := r.db.WithContext(ctx).Model(&models.NovelVolume{}).
        Where("id = ?", id).
        Update("status", status)
    if result.Error != nil {
        return fmt.Errorf("更新小说分卷状态失败: %w", result.Error)
    }
    if result.RowsAffected == 0 {
        return fmt.Errorf("小说分卷不存在")
    }
    return nil
}
```
