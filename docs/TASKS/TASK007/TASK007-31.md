### 3. 实现AudioEpisodeRepository接口

在同一个文件中实现AudioEpisodeRepository接口：

```go
// audioEpisodeRepository AudioEpisodeRepository的实现
type audioEpisodeRepository struct {
    db *gorm.DB
}

// NewAudioEpisodeRepository 创建AudioEpisodeRepository实例
func NewAudioEpisodeRepository(db *gorm.DB) AudioEpisodeRepository {
    return &audioEpisodeRepository{db: db}
}

// Create 创建音频分集
func (r *audioEpisodeRepository) Create(ctx context.Context, episode *models.AudioEpisode) error {
    if err := r.db.WithContext(ctx).Create(episode).Error; err != nil {
        return fmt.Errorf("创建音频分集失败: %w", err)
    }
    return nil
}

// GetByID 根据ID获取音频分集
func (r *audioEpisodeRepository) GetByID(ctx context.Context, id int64) (*models.AudioEpisode, error) {
    var episode models.AudioEpisode
    if err := r.db.WithContext(ctx).Where("id = ?", id).First(&episode).Error; err != nil {
        if errors.Is(err, gorm.ErrRecordNotFound) {
            return nil, errors.New("音频分集不存在")
        }
        return nil, fmt.Errorf("查询音频分集失败: %w", err)
    }
    return &episode, nil
}

// GetByAudioID 根据音频ID获取分集列表
func (r *audioEpisodeRepository) GetByAudioID(ctx context.Context, audioID int64, params *AudioEpisodeQueryParams) ([]*models.AudioEpisode, int64, error) {
    var episodes []*models.AudioEpisode
    var total int64
    
    query := r.db.WithContext(ctx).Model(&models.AudioEpisode{}).Where("audio_id = ?", audioID)
    
    // 应用筛选条件
    if params.Status != "" {
        query = query.Where("status = ?", params.Status)
    }
    if params.IsVip != nil {
        query = query.Where("is_vip = ?", *params.IsVip)
    }
    
    // 获取总数
    if err := query.Count(&total).Error; err != nil {
        return nil, 0, fmt.Errorf("查询音频分集总数失败: %w", err)
    }
    
    // 应用排序
    orderBy := "sort_order"
    if params.OrderBy != "" {
        validOrderBy := map[string]bool{
            "sort_order": true,
            "created_at": true,
            "updated_at": true,
        }
        if validOrderBy[params.OrderBy] {
            orderBy = params.OrderBy
        }
    }
    orderDir := "asc"
    if params.OrderDir == "desc" {
        orderDir = "desc"
    }
    query = query.Order(orderBy + " " + orderDir)
    
    // 应用分页
    page := params.Page
    if page < 1 {
        page = 1
    }
    limit := params.Limit
    if limit < 1 {
        limit = 20
    }
    if limit > 100 {
        limit = 100
    }
    offset := (page - 1) * limit
    query = query.Offset(offset).Limit(limit)
    
    // 执行查询
    if err := query.Find(&episodes).Error; err != nil {
        return nil, 0, fmt.Errorf("查询音频分集列表失败: %w", err)
    }
    
    return episodes, total, nil
}

// Update 更新音频分集
func (r *audioEpisodeRepository) Update(ctx context.Context, episode *models.AudioEpisode) error {
    if err := r.db.WithContext(ctx).Save(episode).Error; err != nil {
        return fmt.Errorf("更新音频分集失败: %w", err)

    if result.Error != nil {
        return fmt.Errorf("删除音频分集失败: %w", result.Error)
    }
    if result.RowsAffected == 0 {
        return errors.New("音频分集不存在")
    }
    return nil
}

// UpdateStatus 更新音频分集状态
func (r *audioEpisodeRepository) UpdateStatus(ctx context.Context, id int64, status string) error {

        Update("status", status)

    }
    if result.RowsAffected == 0 {
```
