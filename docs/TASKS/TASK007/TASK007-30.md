### 3. 实现AudioSeasonRepository接口

```go
// audioSeasonRepository AudioSeasonRepository接口实现
type audioSeasonRepository struct {
    db *gorm.DB
}

// NewAudioSeasonRepository 创建AudioSeasonRepository实例
func NewAudioSeasonRepository(db *gorm.DB) AudioSeasonRepository {
    return &audioSeasonRepository{db: db}
}

// Create 创建音频分季
func (r *audioSeasonRepository) Create(ctx context.Context, season *models.AudioSeason) error {
    if err := r.db.WithContext(ctx).Create(season).Error; err != nil {
        return fmt.Errorf("创建音频分季失败: %w", err)
    }
    return nil
}

// GetByID 根据ID获取音频分季
func (r *audioSeasonRepository) GetByID(ctx context.Context, id int64) (*models.AudioSeason, error) {
    var season models.AudioSeason
    if err := r.db.WithContext(ctx).Where("id = ?", id).First(&season).Error; err != nil {
        if errors.Is(err, gorm.ErrRecordNotFound) {
            return nil, fmt.Errorf("音频分季不存在")
        }
        return nil, fmt.Errorf("查询音频分季失败: %w", err)
    }
    return &season, nil
}

// GetByAudioID 根据音频ID获取分季列表
func (r *audioSeasonRepository) GetByAudioID(ctx context.Context, audioID int64) ([]*models.AudioSeason, error) {
    var seasons []*models.AudioSeason
    if err := r.db.WithContext(ctx).
        Where("audio_id = ?", audioID).
        Order("sort_order ASC").
        Find(&seasons).Error; err != nil {
        return nil, fmt.Errorf("查询音频分季列表失败: %w", err)
    }
    return seasons, nil
}

// Update 更新音频分季
func (r *audioSeasonRepository) Update(ctx context.Context, season *models.AudioSeason) error {
    if err := r.db.WithContext(ctx).Save(season).Error; err != nil {
        return fmt.Errorf("更新音频分季失败: %w", err)
    }
    return nil
}

// Delete 删除音频分季
func (r *audioSeasonRepository) Delete(ctx context.Context, id int64) error {
    result := r.db.WithContext(ctx).Delete(&models.AudioSeason{}, id)
    if result.Error != nil {
        return fmt.Errorf("删除音频分季失败: %w", result.Error)
    }
    if result.RowsAffected == 0 {
        return fmt.Errorf("音频分季不存在")
    }
    return nil
}

// UpdateStatus 更新分季状态
func (r *audioSeasonRepository) UpdateStatus(ctx context.Context, id int64, status string) error {
    result := r.db.WithContext(ctx).Model(&models.AudioSeason{}).
        Where("id = ?", id).
        Update("status", status)
    if result.Error != nil {
        return fmt.Errorf("更新音频分季状态失败: %w", result.Error)
    }
    if result.RowsAffected == 0 {
        return fmt.Errorf("音频分季不存在")
    }
    return nil
}
```
