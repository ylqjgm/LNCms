### 2. 创建AudioEpisode模型文件

在 `internal/models/` 目录下创建 `audio_episode.go` 文件，定义AudioEpisode模型：

```go
package models

import (
    "time"
    "gorm.io/gorm"
)

// AudioEpisode 音频分集模型
// 存储音频作品的分集信息，每个分集对应一个音频文件
// 一个音频作品可以包含多个分季，每个分季包含多个分集
type AudioEpisode struct {
    ID            int64          `gorm:"primaryKey;column:id;type:BIGSERIAL" json:"id"`
    AudioID       int64          `gorm:"column:audio_id;type:BIGINT;not null;index:idx_audio_episodes_audio_id;index:idx_audio_episodes_audio_sort" json:"audio_id"`
    SeasonID      *int64         `gorm:"column:season_id;type:BIGINT;index:idx_audio_episodes_season_id;index:idx_audio_episodes_season_sort" json:"season_id,omitempty"`
    Title         string         `gorm:"column:title;type:VARCHAR(200);not null" json:"title"`
    Description   *string        `gorm:"column:description;type:TEXT" json:"description,omitempty"`
    SortOrder     int            `gorm:"column:sort_order;type:INTEGER;not null;default:0;index:idx_audio_episodes_sort_order;index:idx_audio_episodes_audio_sort;index:idx_audio_episodes_season_sort" json:"sort_order"`
    Status        string         `gorm:"column:status;type:VARCHAR(50);not null;default:'draft';index:idx_audio_episodes_status" json:"status"`
    IsVIP         bool           `gorm:"column:is_vip;type:BOOLEAN;default:false;index:idx_audio_episodes_is_vip" json:"is_vip"`
    Duration      int            `gorm:"column:duration;type:INTEGER;default:0" json:"duration"`
    ViewCount     int64          `gorm:"column:view_count;type:BIGINT;default:0" json:"view_count"`
    LikeCount     int64          `gorm:"column:like_count;type:BIGINT;default:0" json:"like_count"`
    FavoriteCount int64          `gorm:"column:favorite_count;type:BIGINT;default:0" json:"favorite_count"`
    CommentCount  int64          `gorm:"column:comment_count;type:BIGINT;default:0" json:"comment_count"`
    CreatedAt     time.Time      `gorm:"column:created_at;type:TIMESTAMPTZ;default:CURRENT_TIMESTAMP" json:"created_at"`
    UpdatedAt     time.Time      `gorm:"column:updated_at;type:TIMESTAMPTZ;default:CURRENT_TIMESTAMP" json:"updated_at"`
    
    // 关联关系
    Audio  *Audio       `gorm:"foreignKey:AudioID;references:ID" json:"audio,omitempty"`
    Season *AudioSeason `gorm:"foreignKey:SeasonID;references:ID" json:"season,omitempty"`
}

// TableName 指定表名
func (AudioEpisode) TableName() string {
    return "audio_episodes"
}

// BeforeCreate 创建前钩子
func (ae *AudioEpisode) BeforeCreate(tx *gorm.DB) error {
    now := time.Now()
    if ae.CreatedAt.IsZero() {
        ae.CreatedAt = now
    }
    if ae.UpdatedAt.IsZero() {
        ae.UpdatedAt = now
    }
    return nil
}

// BeforeUpdate 更新前钩子
func (ae *AudioEpisode) BeforeUpdate(tx *gorm.DB) error {
    ae.UpdatedAt = time.Now()
    return nil
}
```

### 3. 添加模型注释

在模型文件中添加详细的中文注释，说明每个字段的用途：

```go
// AudioEpisode 音频分集模型
// 存储音频作品的分集信息，每个分集对应一个音频文件
// 一个音频作品可以包含多个分季，每个分季包含多个分集
type AudioEpisode struct {
    // ID 分集ID，自增主键
    ID int64 `gorm:"primaryKey;column:id;type:BIGSERIAL" json:"id"`
    
    // AudioID 音频ID，外键关联audios表，必填字段
    AudioID int64 `gorm:"column:audio_id;type:BIGINT;not null;index:idx_audio_episodes_audio_id;index:idx_audio_episodes_audio_sort" json:"audio_id"`
    
    // SeasonID 分季ID，外键关联audio_seasons表，可选字段，NULL表示不属于任何分季
    SeasonID *int64 `gorm:"column:season_id;type:BIGINT;index:idx_audio_episodes_season_id;index:idx_audio_episodes_season_sort" json:"season_id,omitempty"`
    
    // Title 分集标题，必填字段，最大长度200字符
    Title string `gorm:"column:title;type:VARCHAR(200);not null" json:"title"`
    
    // Description 分集描述，说明分集的内容和特点，可选字段
    Description *string `gorm:"column:description;type:TEXT" json:"description,omitempty"`
    
    // SortOrder 排序顺序，数字越小越靠前，用于排序显示，默认值为0
    SortOrder int `gorm:"column:sort_order;type:INTEGER;not null;default:0;index:idx_audio_episodes_sort_order;index:idx_audio_episodes_audio_sort;index:idx_audio_episodes_season_sort" json:"sort_order"`
    
    // Status 分集状态：draft-草稿、published-已发布、archived-已归档
    // 必须符合检查约束，默认值为draft
    Status string `gorm:"column:status;type:VARCHAR(50);not null;default:'draft';index:idx_audio_episodes_status" json:"status"`
    
    // IsVIP 是否VIP专享，TRUE-需要VIP权限，FALSE-免费，默认值为FALSE
    IsVIP bool `gorm:"column:is_vip;type:BOOLEAN;default:false;index:idx_audio_episodes_is_vip" json:"is_vip"`
    
    // Duration 时长（秒），默认值为0
    Duration int `gorm:"column:duration;type:INTEGER;default:0" json:"duration"`
    
    // ViewCount 播放次数，默认值为0
    ViewCount int64 `gorm:"column:view_count;type:BIGINT;default:0" json:"view_count"`
    
    // LikeCount 点赞次数，默认值为0
    LikeCount int64 `gorm:"column:like_count;type:BIGINT;default:0" json:"like_count"`
    
    // FavoriteCount 收藏次数，默认值为0
    FavoriteCount int64 `gorm:"column:favorite_count;type:BIGINT;default:0" json:"favorite_count"`
    
    // CommentCount 评论次数，默认值为0
    CommentCount int64 `gorm:"column:comment_count;type:BIGINT;default:0" json:"comment_count"`
    
    // CreatedAt 创建时间，自动设置为当前时间
    CreatedAt time.Time `gorm:"column:created_at;type:TIMESTAMPTZ;default:CURRENT_TIMESTAMP" json:"created_at"`
    
    // UpdatedAt 更新时间，每次更新时自动更新
    UpdatedAt time.Time `gorm:"column:updated_at;type:TIMESTAMPTZ;default:CURRENT_TIMESTAMP" json:"updated_at"`
    
    // 关联关系
    // Audio 音频作品信息，外键关联audios表
    Audio *Audio `gorm:"foreignKey:AudioID;references:ID" json:"audio,omitempty"`
    
    // Season 分季信息，外键关联audio_seasons表
    Season *AudioSeason `gorm:"foreignKey:SeasonID;references:ID" json:"season,omitempty"`
}
```

### 4. 定义模型方法

为AudioEpisode模型添加常用的辅助方法：

```go
// IsPublished 检查分集是否已发布
func (ae *AudioEpisode) IsPublished() bool {
    return ae.Status == "published"
}

// IsVIPContent 检查是否为VIP内容
func (ae *AudioEpisode) IsVIPContent() bool {
    return ae.IsVIP
}

// ValidateStatus 验证状态值是否有效
func (ae *AudioEpisode) ValidateStatus() bool {
    validStatuses := []string{"draft", "published", "archived"}
    for _, validStatus := range validStatuses {
        if ae.Status == validStatus {
            return true
        }
    }
    return false
}
```

### 5. 注意事项

- **字段类型映射**：确保GORM标签中的类型与PostgreSQL数据库类型一致
- **索引定义**：在GORM标签中定义索引，包括audio_id、season_id、sort_order、status、is_vip，以及复合索引
- **外键关联**：使用GORM的foreignKey和references标签定义与Audio和AudioSeason的关联关系
- **JSON标签**：所有字段都应有JSON标签，可选字段使用omitempty
- **指针类型**：可选字段（如SeasonID、Description）使用指针类型，支持NULL值
- **时间字段**：使用time.Time类型，支持PostgreSQL的TIMESTAMPTZ类型
- **默认值**：在GORM标签中设置默认值，与数据库设计保持一致
- **检查约束**：status字段需要在应用层进行验证，确保符合数据库约束
- **排序字段**：sort_order用于控制分集的显示顺序
- **统计字段**：view_count、like_count、favorite_count、comment_count等统计字段使用BIGINT类型
