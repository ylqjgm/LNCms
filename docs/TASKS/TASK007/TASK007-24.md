### 4. 实现NovelChapterRepository接口

```go
// novelChapterRepository NovelChapterRepository接口实现
type novelChapterRepository struct {
    db *gorm.DB
}

// NewNovelChapterRepository 创建NovelChapterRepository实例
func NewNovelChapterRepository(db *gorm.DB) NovelChapterRepository {
    return &novelChapterRepository{db: db}
}

// Create 创建小说章节
func (r *novelChapterRepository) Create(ctx context.Context, chapter *models.NovelChapter) error {
    if err := r.db.WithContext(ctx).Create(chapter).Error; err != nil {
        return fmt.Errorf("创建小说章节失败: %w", err)
    }
    return nil
}

// GetByID 根据ID获取小说章节
func (r *novelChapterRepository) GetByID(ctx context.Context, id int64) (*models.NovelChapter, error) {
    var chapter models.NovelChapter
    if err := r.db.WithContext(ctx).Where("id = ?", id).First(&chapter).Error; err != nil {
        if errors.Is(err, gorm.ErrRecordNotFound) {
            return nil, fmt.Errorf("小说章节不存在")
        }
        return nil, fmt.Errorf("查询小说章节失败: %w", err)
    }
    return &chapter, nil
}

// GetByNovelID 根据小说ID获取章节列表
func (r *novelChapterRepository) GetByNovelID(ctx context.Context, novelID int64, params *NovelChapterQueryParams) ([]*models.NovelChapter, int64, error) {
    params.NovelID = &novelID
    return r.queryChapters(ctx, params)
}

// GetByVolumeID 根据分卷ID获取章节列表
func (r *novelChapterRepository) GetByVolumeID(ctx context.Context, volumeID int64, params *NovelChapterQueryParams) ([]*models.NovelChapter, int64, error) {
    params.VolumeID = &volumeID
    return r.queryChapters(ctx, params)
}

// queryChapters 查询章节列表（内部方法）
func (r *novelChapterRepository) queryChapters(ctx context.Context, params *NovelChapterQueryParams) ([]*models.NovelChapter, int64, error) {
    var chapters []*models.NovelChapter
    var total int64
    
    // 构建查询
    query := r.db.WithContext(ctx).Model(&models.NovelChapter{})
    
    // 应用筛选条件
    if params.NovelID != nil {
        query = query.Where("novel_id = ?", *params.NovelID)
    }
    if params.VolumeID != nil {
        query = query.Where("volume_id = ?", *params.VolumeID)
    }
    if params.Status != "" {
        query = query.Where("status = ?", params.Status)
    }
    if params.IsVip != nil {
        query = query.Where("is_vip = ?", *params.IsVip)
    }
    if params.IsActive != nil {
        query = query.Where("is_active = ?", *params.IsActive)
    }
    
    // 获取总数
    if err := query.Count(&total).Error; err != nil {
        return nil, 0, fmt.Errorf("查询章节总数失败: %w", err)
    }
    
    // 应用排序
    orderBy := "sort_order"
    if params.OrderBy != "" {
        // 验证排序字段，防止SQL注入
        validOrderBy := map[string]bool{
            "sort_order": true,
            "created_at": true,
            "updated_at": true,
        }
        if validOrderBy[params.OrderBy] {
            orderBy = params.OrderBy
        }
    }
    orderDir := "asc"
    if params.OrderDir == "desc" {
        orderDir = "desc"
    }
    query = query.Order(orderBy + " " + orderDir)
    
    // 应用分页
    page := params.Page
    if page < 1 {
        page = 1
    }
    limit := params.Limit
    if limit < 1 {
        limit = 20
    }
    if limit > 100 {
        limit = 100
    }
    offset := (page - 1) * limit
    query = query.Offset(offset).Limit(limit)
    
    // 执行查询
    if err := query.Find(&chapters).Error; err != nil {
        return nil, 0, fmt.Errorf("查询章节列表失败: %w", err)
    }
    
    return chapters, total, nil
}

// Update 更新小说章节
func (r *novelChapterRepository) Update(ctx context.Context, chapter *models.NovelChapter) error {
    if err := r.db.WithContext(ctx).Save(chapter).Error; err != nil {
        return fmt.Errorf("更新小说章节失败: %w", err)
    }
    return nil
}

// Delete 删除小说章节
func (r *novelChapterRepository) Delete(ctx context.Context, id int64) error {
    result := r.db.WithContext(ctx).Delete(&models.NovelChapter{}, id)
    if result.Error != nil {
        return fmt.Errorf("删除小说章节失败: %w", result.Error)
    }
    if result.RowsAffected == 0 {
        return fmt.Errorf("小说章节不存在")
    }
    return nil
}

// UpdateStatus 更新章节状态
func (r *novelChapterRepository) UpdateStatus(ctx context.Context, id int64, status string) error {
    result := r.db.WithContext(ctx).Model(&models.NovelChapter{}).
        Where("id = ?", id).
        Update("status", status)
    if result.Error != nil {
        return fmt.Errorf("更新小说章节状态失败: %w", result.Error)
    }
    if result.RowsAffected == 0 {
        return fmt.Errorf("小说章节不存在")
    }
    return nil
}

// UpdateWordCount 更新章节字数
func (r *novelChapterRepository) UpdateWordCount(ctx context.Context, id int64, wordCount int64) error {
    result := r.db.WithContext(ctx).Model(&models.NovelChapter{}).
        Where("id = ?", id).
        Update("word_count", wordCount)
    if result.Error != nil {
        return fmt.Errorf("更新小说章节字数失败: %w", result.Error)
    }
    if result.RowsAffected == 0 {
        return fmt.Errorf("小说章节不存在")
    }
    return nil
}
```
