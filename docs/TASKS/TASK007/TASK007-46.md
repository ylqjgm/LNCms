```go
package service

import (
    "context"
    "yourproject/internal/models"
)

// ComicChapterService 漫画章节服务接口
// 提供漫画章节管理的业务逻辑
type ComicChapterService interface {
    // GetChapter 根据ID获取章节
    GetChapter(ctx context.Context, id int64) (*models.ComicChapter, error)
    
    // GetChapters 获取章节列表（支持分页、排序）
    GetChapters(ctx context.Context, params *ComicChapterQueryParams) ([]*models.ComicChapter, int64, error)
    
    // CreateChapter 创建章节
    CreateChapter(ctx context.Context, chapter *models.ComicChapter) error
    
    // UpdateChapter 更新章节
    UpdateChapter(ctx context.Context, chapter *models.ComicChapter) error
    
    // DeleteChapter 删除章节
    DeleteChapter(ctx context.Context, id int64) error
    
    // UpdateStatus 更新章节状态
    UpdateStatus(ctx context.Context, id int64, status string) error
}

// ComicChapterQueryParams 漫画章节查询参数
type ComicChapterQueryParams struct {
    ComicID  int64  // 漫画ID（必填）
    Page     int    // 页码，从1开始
    Limit    int    // 每页数量
    Status   string // 状态筛选
    IsVip    *bool  // 是否VIP筛选
    OrderBy  string // 排序字段（sort_order、created_at、updated_at）
    OrderDir string // 排序方向（asc、desc）
}
```

### 2. 实现ComicChapterService接口

在同一个文件中实现ComicChapterService接口：

```go
// comicChapterService ComicChapterService的实现
type comicChapterService struct {
    chapterRepo repository.ComicChapterRepository
    comicRepo   repository.ComicRepository
}

// NewComicChapterService 创建ComicChapterService实例
func NewComicChapterService(
    chapterRepo repository.ComicChapterRepository,
    comicRepo repository.ComicRepository,
) ComicChapterService {
    return &comicChapterService{
        chapterRepo: chapterRepo,
        comicRepo:   comicRepo,
    }
}

// GetChapter 根据ID获取章节
func (s *comicChapterService) GetChapter(ctx context.Context, id int64) (*models.ComicChapter, error) {
    if id <= 0 {
        return nil, errors.New("章节ID无效")
    }
    
    chapter, err := s.chapterRepo.GetByID(ctx, id)
    if err != nil {
        return nil, fmt.Errorf("获取章节失败: %w", err)
    }
    
    return chapter, nil
}

// GetChapters 获取章节列表
func (s *comicChapterService) GetChapters(ctx context.Context, params *ComicChapterQueryParams) ([]*models.ComicChapter, int64, error) {
    if params.ComicID <= 0 {
        return nil, 0, errors.New("漫画ID无效")
    }
    
    // 验证漫画是否存在
    _, err := s.comicRepo.GetByID(ctx, params.ComicID)
    if err != nil {
        return nil, 0, errors.New("漫画不存在")
    }
    
    // 构建查询参数
    queryParams := &repository.ComicChapterQueryParams{
        Page:     params.Page,
        Limit:    params.Limit,
        Status:   params.Status,
        IsVip:    params.IsVip,
        OrderBy:  params.OrderBy,
        OrderDir: params.OrderDir,
    }
    
    // 验证和设置默认值
    if queryParams.Page < 1 {
        queryParams.Page = 1
    }
    if queryParams.Limit < 1 {
        queryParams.Limit = 20
    }
    if queryParams.Limit > 100 {
        queryParams.Limit = 100
    }
    if queryParams.OrderBy == "" {
        queryParams.OrderBy = "sort_order"
    }
    if queryParams.OrderDir == "" {
        queryParams.OrderDir = "asc"
    }
    
    chapters, total, err := s.chapterRepo.GetByComicID(ctx, params.ComicID, queryParams)
    if err != nil {
        return nil, 0, fmt.Errorf("获取章节列表失败: %w", err)
    }
    
    return chapters, total, nil
}

// CreateChapter 创建章节
func (s *comicChapterService) CreateChapter(ctx context.Context, chapter *models.ComicChapter) error {
    // 业务验证
    if err := s.validateChapter(chapter); err != nil {
        return err
    }
    
    // 验证漫画是否存在
    _, err := s.comicRepo.GetByID(ctx, chapter.ComicID)
    if err != nil {
        return errors.New("漫画不存在")
    }
    
    // 设置默认值
    if chapter.Status == "" {
        chapter.Status = "draft"
    }
    if chapter.SortOrder == 0 {
        // 获取当前最大排序值
        chapters, _, err := s.chapterRepo.GetByComicID(ctx, chapter.ComicID, &repository.ComicChapterQueryParams{
            Page:     1,
            Limit:    10000,
            OrderBy:  "sort_order",
            OrderDir: "desc",
        })
        if err == nil && len(chapters) > 0 {
            chapter.SortOrder = chapters[0].SortOrder + 1
        } else {
            chapter.SortOrder = 1
        }
    }
    
    // 创建章节
    if err := s.chapterRepo.Create(ctx, chapter); err != nil {
        return fmt.Errorf("创建章节失败: %w", err)
    }
    
    return nil
}

// UpdateChapter 更新章节
func (s *comicChapterService) UpdateChapter(ctx context.Context, chapter *models.ComicChapter) error {
    // 业务验证
    if err := s.validateChapter(chapter); err != nil {
        return err
    }
    
    // 检查章节是否存在
    existing, err := s.chapterRepo.GetByID(ctx, chapter.ID)
    if err != nil {
        return errors.New("章节不存在")
    }
    
    // 验证漫画ID是否改变（不允许改变）
    if existing.ComicID != chapter.ComicID {
        return errors.New("不能修改章节所属的漫画")
    }
    
    // 更新章节
    if err := s.chapterRepo.Update(ctx, chapter); err != nil {
        return fmt.Errorf("更新章节失败: %w", err)
    }
    
    return nil
}

// DeleteChapter 删除章节
func (s *comicChapterService) DeleteChapter(ctx context.Context, id int64) error {
    if id <= 0 {
        return errors.New("章节ID无效")
    }
    
    // 检查章节是否存在
    chapter, err := s.chapterRepo.GetByID(ctx, id)
    if err != nil {
        return errors.New("章节不存在")
    }
    
    // 删除章节
    if err := s.chapterRepo.Delete(ctx, id); err != nil {
        return fmt.Errorf("删除章节失败: %w", err)
    }
    
    return nil
}

// UpdateStatus 更新章节状态
func (s *comicChapterService) UpdateStatus(ctx context.Context, id int64, status string) error {
    if id <= 0 {
        return errors.New("章节ID无效")
    }
    
    // 验证状态
    validStatuses := []string{"draft", "reviewing", "published", "archived"}
    isValid := false
    for _, validStatus := range validStatuses {
        if status == validStatus {
            isValid = true
            break
        }
    }
    if !isValid {
        return errors.New("无效的章节状态")
    }
    
    // 检查章节是否存在
    _, err := s.chapterRepo.GetByID(ctx, id)
    if err != nil {
        return errors.New("章节不存在")
    }
    
    // 更新状态
    if err := s.chapterRepo.UpdateStatus(ctx, id, status); err != nil {
        return fmt.Errorf("更新章节状态失败: %w", err)
    }
    
    return nil
}

// validateChapter 验证章节数据
func (s *comicChapterService) validateChapter(chapter *models.ComicChapter) error {
    if chapter.ComicID <= 0 {
        return errors.New("漫画ID无效")
    }
    
    if chapter.ChapterTitle == "" {
        return errors.New("章节标题不能为空")
    }
    if len(chapter.ChapterTitle) > 200 {
        return errors.New("章节标题长度不能超过200个字符")
    }
    
    // 验证状态
    if chapter.Status != "" {
        validStatuses := []string{"draft", "reviewing", "published", "archived"}
        isValid := false
        for _, validStatus := range validStatuses {
            if chapter.Status == validStatus {
                isValid = true
                break
            }
        }
        if !isValid {
            return errors.New("无效的章节状态")
        }
    }
    
    return nil
}
```
