### 2. 创建NovelVolume模型文件

在 `internal/models/` 目录下创建 `novel_volume.go` 文件，定义NovelVolume模型：

```go
package models

import (
    "time"
    "gorm.io/gorm"
)

// NovelVolume 小说分卷模型
// 存储小说分卷信息，包括分卷名称、描述、排序等
// 关联小说信息
type NovelVolume struct {
    ID                int64          `gorm:"primaryKey;column:id;type:BIGSERIAL" json:"id"`
    NovelID           int64          `gorm:"column:novel_id;type:BIGINT;not null;index:idx_novel_volumes_novel_id;uniqueIndex:uk_novel_volumes_novel_id_volume_order" json:"novel_id"`
    VolumeName        string         `gorm:"column:volume_name;type:VARCHAR(200);not null" json:"volume_name"`
    VolumeDescription *string        `gorm:"column:volume_description;type:TEXT" json:"volume_description,omitempty"`
    VolumeOrder       int            `gorm:"column:volume_order;type:INTEGER;not null;default:0;index:idx_novel_volumes_volume_order;uniqueIndex:uk_novel_volumes_novel_id_volume_order" json:"volume_order"`
    IsPublished       bool           `gorm:"column:is_published;type:BOOLEAN;default:false;index:idx_novel_volumes_is_published" json:"is_published"`
    PublishedAt       *time.Time     `gorm:"column:published_at;type:TIMESTAMPTZ" json:"published_at,omitempty"`
    CreatedAt         time.Time      `gorm:"column:created_at;type:TIMESTAMPTZ;default:CURRENT_TIMESTAMP" json:"created_at"`
    UpdatedAt         time.Time      `gorm:"column:updated_at;type:TIMESTAMPTZ;default:CURRENT_TIMESTAMP" json:"updated_at"`
    
    // 关联关系
    Novel *Novel `gorm:"foreignKey:NovelID;references:ID" json:"novel,omitempty"`
}

// TableName 指定表名
func (NovelVolume) TableName() string {
    return "novel_volumes"
}

// BeforeCreate 创建前钩子
func (nv *NovelVolume) BeforeCreate(tx *gorm.DB) error {
    now := time.Now()
    if nv.CreatedAt.IsZero() {
        nv.CreatedAt = now
    }
    if nv.UpdatedAt.IsZero() {
        nv.UpdatedAt = now
    }
    return nil
}

// BeforeUpdate 更新前钩子
func (nv *NovelVolume) BeforeUpdate(tx *gorm.DB) error {
    nv.UpdatedAt = time.Now()
    return nil
}
```

### 3. 添加模型注释

在模型文件中添加详细的中文注释，说明每个字段的用途：

```go
// NovelVolume 小说分卷模型
// 存储小说分卷信息，包括分卷名称、描述、排序等
// 关联小说信息
// 同一小说的分卷排序必须唯一
type NovelVolume struct {
    // ID 分卷ID，自增主键
    ID int64 `gorm:"primaryKey;column:id;type:BIGSERIAL" json:"id"`
    
    // NovelID 小说ID，外键关联novels表，必填字段
    NovelID int64 `gorm:"column:novel_id;type:BIGINT;not null;index:idx_novel_volumes_novel_id;uniqueIndex:uk_novel_volumes_novel_id_volume_order" json:"novel_id"`
    
    // VolumeName 分卷名称，必填字段
    VolumeName string `gorm:"column:volume_name;type:VARCHAR(200);not null" json:"volume_name"`
    
    // VolumeDescription 分卷描述，可选字段
    VolumeDescription *string `gorm:"column:volume_description;type:TEXT" json:"volume_description,omitempty"`
    
    // VolumeOrder 分卷排序，数字越小越靠前，用于排序显示，默认值为0
    // 同一小说的分卷排序必须唯一
    VolumeOrder int `gorm:"column:volume_order;type:INTEGER;not null;default:0;index:idx_novel_volumes_volume_order;uniqueIndex:uk_novel_volumes_novel_id_volume_order" json:"volume_order"`
    
    // IsPublished 是否发布，TRUE-已发布，FALSE-未发布，默认值为FALSE
    IsPublished bool `gorm:"column:is_published;type:BOOLEAN;default:false;index:idx_novel_volumes_is_published" json:"is_published"`
    
    // PublishedAt 发布时间，可选字段
    PublishedAt *time.Time `gorm:"column:published_at;type:TIMESTAMPTZ" json:"published_at,omitempty"`
    
    // CreatedAt 创建时间，自动设置为当前时间
    CreatedAt time.Time `gorm:"column:created_at;type:TIMESTAMPTZ;default:CURRENT_TIMESTAMP" json:"created_at"`
    
    // UpdatedAt 更新时间，每次更新时自动更新
    UpdatedAt time.Time `gorm:"column:updated_at;type:TIMESTAMPTZ;default:CURRENT_TIMESTAMP" json:"updated_at"`
    
    // 关联关系
    // Novel 小说，多对一关系
    Novel *Novel `gorm:"foreignKey:NovelID;references:ID" json:"novel,omitempty"`
}
```

### 5. 注意事项

- **字段类型映射**：确保GORM标签中的类型与PostgreSQL数据库类型一致
- **唯一索引**：novel_id + volume_order 使用uniqueIndex标签定义复合唯一索引，确保同一小说的分卷排序唯一
- **索引定义**：在GORM标签中定义索引，包括唯一索引和普通索引
- **外键关联**：使用GORM的foreignKey和references标签定义关联关系
- **JSON标签**：所有字段都应有JSON标签，可选字段使用omitempty
- **指针类型**：可选字段（如VolumeDescription、PublishedAt）使用指针类型，支持NULL值
- **时间字段**：使用time.Time类型，支持PostgreSQL的TIMESTAMPTZ类型
- **默认值**：在GORM标签中设置默认值，与数据库设计保持一致
- **唯一约束**：novel_id + volume_order 的复合唯一约束需要在应用层进行验证
