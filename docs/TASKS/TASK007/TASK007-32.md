### 3. 实现AudioFileRepository接口

在同一个文件中实现AudioFileRepository接口：

```go
// audioFileRepository AudioFileRepository的实现
type audioFileRepository struct {
    db *gorm.DB
}

// NewAudioFileRepository 创建AudioFileRepository实例
func NewAudioFileRepository(db *gorm.DB) AudioFileRepository {
    return &audioFileRepository{db: db}
}

// Create 创建音频文件
func (r *audioFileRepository) Create(ctx context.Context, audioFile *models.AudioFile) error {
    if err := r.db.WithContext(ctx).Create(audioFile).Error; err != nil {
        return fmt.Errorf("创建音频文件失败: %w", err)
    }
    return nil
}

// GetByID 根据ID获取音频文件
func (r *audioFileRepository) GetByID(ctx context.Context, id int64) (*models.AudioFile, error) {
    var audioFile models.AudioFile
    if err := r.db.WithContext(ctx).Where("id = ?", id).First(&audioFile).Error; err != nil {
        if errors.Is(err, gorm.ErrRecordNotFound) {
            return nil, errors.New("音频文件不存在")
        }
        return nil, fmt.Errorf("查询音频文件失败: %w", err)
    }
    return &audioFile, nil
}

// GetByEpisodeID 根据分集ID获取音频文件列表
func (r *audioFileRepository) GetByEpisodeID(ctx context.Context, episodeID int64) ([]*models.AudioFile, error) {
    var audioFiles []*models.AudioFile
    if err := r.db.WithContext(ctx).
        Where("episode_id = ?", episodeID).
        Order("id ASC").
        Find(&audioFiles).Error; err != nil {
        return nil, fmt.Errorf("查询音频文件列表失败: %w", err)
    }
    return audioFiles, nil
}

// Update 更新音频文件
func (r *audioFileRepository) Update(ctx context.Context, audioFile *models.AudioFile) error {
    if err := r.db.WithContext(ctx).Save(audioFile).Error; err != nil {
        return fmt.Errorf("更新音频文件失败: %w", err)
    }
    return nil
}

// Delete 删除音频文件
func (r *audioFileRepository) Delete(ctx context.Context, id int64) error {
    result := r.db.WithContext(ctx).Delete(&models.AudioFile{}, id)
    if result.Error != nil {
        return fmt.Errorf("删除音频文件失败: %w", result.Error)
    }
    if result.RowsAffected == 0 {
        return errors.New("音频文件不存在")
    }
    return nil
}
```
