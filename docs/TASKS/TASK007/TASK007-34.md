### 3. 实现VideoSeasonRepository接口

在同一个文件中实现VideoSeasonRepository接口：

```go
// videoSeasonRepository VideoSeasonRepository的实现
type videoSeasonRepository struct {
    db *gorm.DB
}

// NewVideoSeasonRepository 创建VideoSeasonRepository实例
func NewVideoSeasonRepository(db *gorm.DB) VideoSeasonRepository {
    return &videoSeasonRepository{db: db}
}

// Create 创建视频分季
func (r *videoSeasonRepository) Create(ctx context.Context, season *models.VideoSeason) error {
    if err := r.db.WithContext(ctx).Create(season).Error; err != nil {
        return fmt.Errorf("创建视频分季失败: %w", err)
    }
    return nil
}

// GetByID 根据ID获取视频分季
func (r *videoSeasonRepository) GetByID(ctx context.Context, id int64) (*models.VideoSeason, error) {
    var season models.VideoSeason
    if err := r.db.WithContext(ctx).Where("id = ?", id).First(&season).Error; err != nil {
        if errors.Is(err, gorm.ErrRecordNotFound) {
            return nil, errors.New("视频分季不存在")
        }
        return nil, fmt.Errorf("查询视频分季失败: %w", err)
    }
    return &season, nil
}

// GetByVideoID 根据视频ID获取分季列表
func (r *videoSeasonRepository) GetByVideoID(ctx context.Context, videoID int64) ([]*models.VideoSeason, error) {
    var seasons []*models.VideoSeason
    if err := r.db.WithContext(ctx).
        Where("video_id = ?", videoID).
        Order("sort_order ASC, id ASC").
        Find(&seasons).Error; err != nil {
        return nil, fmt.Errorf("查询视频分季列表失败: %w", err)
    }
    return seasons, nil
}

// Update 更新视频分季
func (r *videoSeasonRepository) Update(ctx context.Context, season *models.VideoSeason) error {
    if err := r.db.WithContext(ctx).Save(season).Error; err != nil {
        return fmt.Errorf("更新视频分季失败: %w", err)
    }
    return nil
}

// Delete 删除视频分季
func (r *videoSeasonRepository) Delete(ctx context.Context, id int64) error {
    result := r.db.WithContext(ctx).Delete(&models.VideoSeason{}, id)
    if result.Error != nil {
        return fmt.Errorf("删除视频分季失败: %w", result.Error)
    }
    if result.RowsAffected == 0 {
        return errors.New("视频分季不存在")
    }
    return nil
}

// UpdateStatus 更新视频分季状态
func (r *videoSeasonRepository) UpdateStatus(ctx context.Context, id int64, status string) error {
    result := r.db.WithContext(ctx).Model(&models.VideoSeason{}).
        Where("id = ?", id).
        Update("status", status)
    if result.Error != nil {
        return fmt.Errorf("更新视频分季状态失败: %w", result.Error)
    }
    if result.RowsAffected == 0 {
        return errors.New("视频分季不存在")
    }
    return nil
}
```
