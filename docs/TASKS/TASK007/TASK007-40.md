### 1. 创建NovelService接口文件

在 `internal/service/` 目录下创建 `novel_service.go` 文件，定义NovelService接口：

```go
package service

import (
    "context"
    "yourproject/internal/models"
)

// NovelService 小说服务接口
// 提供小说管理的业务逻辑
type NovelService interface {
    // GetNovel 根据ID获取小说
    GetNovel(ctx context.Context, id int64) (*models.Novel, error)
    
    // GetNovels 获取小说列表（支持分页、筛选、排序）
    GetNovels(ctx context.Context, params *NovelQueryParams) ([]*models.Novel, int64, error)
    
    // CreateNovel 创建小说
    CreateNovel(ctx context.Context, novel *models.Novel) error
    
    // UpdateNovel 更新小说
    UpdateNovel(ctx context.Context, novel *models.Novel) error
    
    // DeleteNovel 删除小说
    DeleteNovel(ctx context.Context, id int64) error
}

// NovelQueryParams 小说查询参数
type NovelQueryParams struct {
    Page          int     // 页码，从1开始
    Limit         int     // 每页数量
    Title         string  // 标题筛选（模糊匹配）
    AuthorID      *int64  // 作者ID筛选
    CategoryIDs   []int64 // 分类ID筛选
    TagIDs        []int64 // 标签ID筛选
    Status        string  // 状态筛选
    SerialStatus  string  // 连载状态筛选
    IsActive      *bool   // 是否启用筛选
    OrderBy       string  // 排序字段（title、created_at、updated_at、word_count、rating）
    OrderDir      string  // 排序方向（asc、desc）
}
```

### 2. 实现NovelService接口

在同一个文件中实现NovelService接口：

```go
// novelService NovelService的实现
type novelService struct {
    novelRepo    repository.NovelRepository
    categorySvc  CategoryService
    tagSvc       TagService
}

// NewNovelService 创建NovelService实例
func NewNovelService(
    novelRepo repository.NovelRepository,
    categorySvc CategoryService,
    tagSvc TagService,
) NovelService {
    return &novelService{
        novelRepo:   novelRepo,
        categorySvc: categorySvc,
        tagSvc:      tagSvc,
    }
}

// GetNovel 根据ID获取小说
func (s *novelService) GetNovel(ctx context.Context, id int64) (*models.Novel, error) {
    if id <= 0 {
        return nil, errors.New("小说ID无效")
    }
    
    novel, err := s.novelRepo.GetByID(ctx, id)
    if err != nil {
        return nil, fmt.Errorf("获取小说失败: %w", err)
    }
    
    return novel, nil
}

// GetNovels 获取小说列表
func (s *novelService) GetNovels(ctx context.Context, params *NovelQueryParams) ([]*models.Novel, int64, error) {
    // 构建查询参数
    queryParams := &repository.NovelQueryParams{
        Page:         params.Page,
        Limit:        params.Limit,
        Title:        params.Title,
        AuthorID:     params.AuthorID,
        CategoryIDs:  params.CategoryIDs,
        TagIDs:       params.TagIDs,
        Status:       params.Status,
        SerialStatus: params.SerialStatus,
        IsActive:     params.IsActive,
        OrderBy:      params.OrderBy,
        OrderDir:     params.OrderDir,
    }
    
    // 验证和设置默认值
    if queryParams.Page < 1 {
        queryParams.Page = 1
    }
    if queryParams.Limit < 1 {
        queryParams.Limit = 20
    }
    if queryParams.Limit > 100 {
        queryParams.Limit = 100
    }
    if queryParams.OrderBy == "" {
        queryParams.OrderBy = "created_at"
    }
    if queryParams.OrderDir == "" {
        queryParams.OrderDir = "desc"
    }
    
    novels, total, err := s.novelRepo.GetAll(ctx, queryParams)
    if err != nil {
        return nil, 0, fmt.Errorf("获取小说列表失败: %w", err)
    }
    
    return novels, total, nil
}

// CreateNovel 创建小说
func (s *novelService) CreateNovel(ctx context.Context, novel *models.Novel) error {
    // 业务验证
    if err := s.validateNovel(novel); err != nil {
        return err
    }
    
    // 验证分类ID
    if len(novel.CategoryIDs) > 0 {
        for _, categoryID := range novel.CategoryIDs {
            category, err := s.categorySvc.GetCategory(ctx, categoryID)
            if err != nil {
                return fmt.Errorf("分类ID %d 不存在", categoryID)
            }
            // 验证分类类型
            if category.CategoryType != "novel" {
                return fmt.Errorf("分类ID %d 不是小说分类", categoryID)
            }
        }
    }
    
    // 验证标签ID
    if len(novel.TagIDs) > 0 {
        for _, tagID := range novel.TagIDs {
            _, err := s.tagSvc.GetTag(ctx, tagID)
            if err != nil {
                return fmt.Errorf("标签ID %d 不存在", tagID)
            }
        }
    }
    
    // 设置默认值
    if novel.Status == "" {
        novel.Status = "draft"
    }
    if novel.SerialStatus == "" {
        novel.SerialStatus = "serializing"
    }
    if novel.WordCount == 0 {
        novel.WordCount = 0
    }
    if novel.Rating == 0 {
        novel.Rating = 0
    }
    if novel.IsActive == nil {
        active := true
        novel.IsActive = &active
    }
    
    // 创建小说
    if err := s.novelRepo.Create(ctx, novel); err != nil {
        return fmt.Errorf("创建小说失败: %w", err)
    }
    
    return nil
}

// UpdateNovel 更新小说
func (s *novelService) UpdateNovel(ctx context.Context, novel *models.Novel) error {
    // 业务验证
    if err := s.validateNovel(novel); err != nil {
        return err
    }
    
    // 检查小说是否存在
    existing, err := s.novelRepo.GetByID(ctx, novel.ID)
    if err != nil {
        return errors.New("小说不存在")
    }
    
    // 验证分类ID
    if len(novel.CategoryIDs) > 0 {
        for _, categoryID := range novel.CategoryIDs {
            category, err := s.categorySvc.GetCategory(ctx, categoryID)
            if err != nil {
                return fmt.Errorf("分类ID %d 不存在", categoryID)
            }
            // 验证分类类型
            if category.CategoryType != "novel" {
                return fmt.Errorf("分类ID %d 不是小说分类", categoryID)
            }
        }
    }
    
    // 验证标签ID
    if len(novel.TagIDs) > 0 {
        for _, tagID := range novel.TagIDs {
            _, err := s.tagSvc.GetTag(ctx, tagID)
            if err != nil {
                return fmt.Errorf("标签ID %d 不存在", tagID)
            }
        }
    }
    
    // 更新小说
    if err := s.novelRepo.Update(ctx, novel); err != nil {
        return fmt.Errorf("更新小说失败: %w", err)
    }
    
    return nil
}

// DeleteNovel 删除小说
func (s *novelService) DeleteNovel(ctx context.Context, id int64) error {
    if id <= 0 {
        return errors.New("小说ID无效")
    }
    
    // 检查小说是否存在
    novel, err := s.novelRepo.GetByID(ctx, id)
    if err != nil {
        return errors.New("小说不存在")
    }
    
    // 删除小说
    if err := s.novelRepo.Delete(ctx, id); err != nil {
        return fmt.Errorf("删除小说失败: %w", err)
    }
    
    return nil
}

// validateNovel 验证小说数据
func (s *novelService) validateNovel(novel *models.Novel) error {
    if novel.Title == "" {
        return errors.New("小说标题不能为空")
    }
    if len(novel.Title) > 200 {
        return errors.New("小说标题长度不能超过200个字符")
    }
    
    if novel.AuthorID <= 0 {
        return errors.New("作者ID无效")
    }
    
    // 验证状态
    validStatuses := []string{"draft", "reviewing", "published", "archived"}
    if novel.Status != "" {
        isValid := false
        for _, validStatus := range validStatuses {
            if novel.Status == validStatus {
                isValid = true
                break
            }
        }
        if !isValid {
            return errors.New("无效的小说状态")
        }
    }
    
    // 验证连载状态
    validSerialStatuses := []string{"serializing", "completed", "suspended"}
    if novel.SerialStatus != "" {
        isValid := false
        for _, validStatus := range validSerialStatuses {
            if novel.SerialStatus == validStatus {
                isValid = true
                break
            }
        }
        if !isValid {
            return errors.New("无效的连载状态")
        }
    }
    
    return nil
}
```
