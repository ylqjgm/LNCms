### 1. 创建ComicService接口文件

在 `internal/service/` 目录下创建 `comic_service.go` 文件，定义ComicService接口：

```go
package service

import (
    "context"
    "yourproject/internal/models"
)

// ComicService 漫画服务接口
// 提供漫画管理的业务逻辑
type ComicService interface {
    // GetComic 根据ID获取漫画
    GetComic(ctx context.Context, id int64) (*models.Comic, error)
    
    // GetComics 获取漫画列表（支持分页、筛选、排序）
    GetComics(ctx context.Context, params *ComicQueryParams) ([]*models.Comic, int64, error)
    
    // CreateComic 创建漫画
    CreateComic(ctx context.Context, comic *models.Comic) error
    
    // UpdateComic 更新漫画
    UpdateComic(ctx context.Context, comic *models.Comic) error
    
    // DeleteComic 删除漫画
    DeleteComic(ctx context.Context, id int64) error
}

// ComicQueryParams 漫画查询参数
type ComicQueryParams struct {
    Page         int     // 页码，从1开始
    Limit        int     // 每页数量
    Title        string  // 标题筛选（模糊匹配）
    AuthorID     *int64  // 作者ID筛选
    RegionID     *int64  // 地区ID筛选
    TagIDs       []int64 // 标签ID筛选
    Status       string  // 状态筛选
    SerialStatus string  // 连载状态筛选
    IsActive     *bool   // 是否启用筛选
    OrderBy      string  // 排序字段（title、created_at、updated_at、rating）
    OrderDir     string  // 排序方向（asc、desc）
}
```

### 2. 实现ComicService接口

在同一个文件中实现ComicService接口：

```go
// comicService ComicService的实现
type comicService struct {
    comicRepo  repository.ComicRepository
    regionSvc  RegionService
    tagSvc     TagService
}

// NewComicService 创建ComicService实例
func NewComicService(
    comicRepo repository.ComicRepository,
    regionSvc RegionService,
    tagSvc TagService,
) ComicService {
    return &comicService{
        comicRepo: comicRepo,
        regionSvc: regionSvc,
        tagSvc:    tagSvc,
    }
}

// GetComic 根据ID获取漫画
func (s *comicService) GetComic(ctx context.Context, id int64) (*models.Comic, error) {
    if id <= 0 {
        return nil, errors.New("漫画ID无效")
    }
    
    comic, err := s.comicRepo.GetByID(ctx, id)
    if err != nil {
        return nil, fmt.Errorf("获取漫画失败: %w", err)
    }
    
    return comic, nil
}

// GetComics 获取漫画列表
func (s *comicService) GetComics(ctx context.Context, params *ComicQueryParams) ([]*models.Comic, int64, error) {
    // 构建查询参数
    queryParams := &repository.ComicQueryParams{
        Page:         params.Page,
        Limit:        params.Limit,
        Title:        params.Title,
        AuthorID:     params.AuthorID,
        RegionID:     params.RegionID,
        TagIDs:       params.TagIDs,
        Status:       params.Status,
        SerialStatus: params.SerialStatus,
        IsActive:     params.IsActive,
        OrderBy:      params.OrderBy,
        OrderDir:     params.OrderDir,
    }
    
    // 验证和设置默认值
    if queryParams.Page < 1 {
        queryParams.Page = 1
    }
    if queryParams.Limit < 1 {
        queryParams.Limit = 20
    }
    if queryParams.Limit > 100 {
        queryParams.Limit = 100
    }
    if queryParams.OrderBy == "" {
        queryParams.OrderBy = "created_at"
    }
    if queryParams.OrderDir == "" {
        queryParams.OrderDir = "desc"
    }
    
    comics, total, err := s.comicRepo.GetAll(ctx, queryParams)
    if err != nil {
        return nil, 0, fmt.Errorf("获取漫画列表失败: %w", err)
    }
    
    return comics, total, nil
}

// CreateComic 创建漫画
func (s *comicService) CreateComic(ctx context.Context, comic *models.Comic) error {
    // 业务验证
    if err := s.validateComic(comic); err != nil {
        return err
    }
    
    // 验证地区ID
    if comic.RegionID != nil {
        _, err := s.regionSvc.GetRegion(ctx, *comic.RegionID)
        if err != nil {
            return fmt.Errorf("地区ID %d 不存在", *comic.RegionID)
        }
    }
    
    // 验证标签ID
    if len(comic.TagIDs) > 0 {
        for _, tagID := range comic.TagIDs {
            _, err := s.tagSvc.GetTag(ctx, tagID)
            if err != nil {
                return fmt.Errorf("标签ID %d 不存在", tagID)
            }
        }
    }
    
    // 设置默认值
    if comic.Status == "" {
        comic.Status = "draft"
    }
    if comic.SerialStatus == "" {
        comic.SerialStatus = "serializing"
    }
    if comic.Rating == 0 {
        comic.Rating = 0
    }
    if comic.IsActive == nil {
        active := true
        comic.IsActive = &active
    }
    
    // 创建漫画
    if err := s.comicRepo.Create(ctx, comic); err != nil {
        return fmt.Errorf("创建漫画失败: %w", err)
    }
    
    return nil
}

// UpdateComic 更新漫画
func (s *comicService) UpdateComic(ctx context.Context, comic *models.Comic) error {
    // 业务验证
    if err := s.validateComic(comic); err != nil {
        return err
    }
    
    // 检查漫画是否存在
    existing, err := s.comicRepo.GetByID(ctx, comic.ID)
    if err != nil {
        return errors.New("漫画不存在")
    }
    
    // 验证地区ID
    if comic.RegionID != nil {
        _, err := s.regionSvc.GetRegion(ctx, *comic.RegionID)
        if err != nil {
            return fmt.Errorf("地区ID %d 不存在", *comic.RegionID)
        }
    }
    
    // 验证标签ID
    if len(comic.TagIDs) > 0 {
        for _, tagID := range comic.TagIDs {
            _, err := s.tagSvc.GetTag(ctx, tagID)
            if err != nil {
                return fmt.Errorf("标签ID %d 不存在", tagID)
            }
        }
    }
    
    // 更新漫画
    if err := s.comicRepo.Update(ctx, comic); err != nil {
        return fmt.Errorf("更新漫画失败: %w", err)
    }
    
    return nil
}

// DeleteComic 删除漫画
func (s *comicService) DeleteComic(ctx context.Context, id int64) error {
    if id <= 0 {
        return errors.New("漫画ID无效")
    }
    
    // 检查漫画是否存在
    comic, err := s.comicRepo.GetByID(ctx, id)
    if err != nil {
        return errors.New("漫画不存在")
    }
    
    // 删除漫画
    if err := s.comicRepo.Delete(ctx, id); err != nil {
        return fmt.Errorf("删除漫画失败: %w", err)
    }
    
    return nil
}

// validateComic 验证漫画数据
func (s *comicService) validateComic(comic *models.Comic) error {
    if comic.Title == "" {
        return errors.New("漫画标题不能为空")
    }
    if len(comic.Title) > 200 {
        return errors.New("漫画标题长度不能超过200个字符")
    }
    
    if comic.AuthorID <= 0 {
        return errors.New("作者ID无效")
    }
    
    // 验证状态
    validStatuses := []string{"draft", "reviewing", "published", "archived"}
    if comic.Status != "" {
        isValid := false
        for _, validStatus := range validStatuses {
            if comic.Status == validStatus {
                isValid = true
                break
            }
        }
        if !isValid {
            return errors.New("无效的漫画状态")
        }
    }
    
    // 验证连载状态
    validSerialStatuses := []string{"serializing", "completed", "suspended"}
    if comic.SerialStatus != "" {
        isValid := false
        for _, validStatus := range validSerialStatuses {
            if comic.SerialStatus == validStatus {
                isValid = true
                break
            }
        }
        if !isValid {
            return errors.New("无效的连载状态")
        }
    }
    
    return nil
}
```
