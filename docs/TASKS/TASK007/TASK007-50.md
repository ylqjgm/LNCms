```go
package service

import (
    "context"
    "yourproject/internal/models"

// AudioEpisodeService 音频分集服务接口
// 提供音频分集管理的业务逻辑
type AudioEpisodeService interface {
    // GetEpisode 根据ID获取分集
    GetEpisode(ctx context.Context, id int64) (*models.AudioEpisode, error)
    
    // GetEpisodes 获取分集列表（支持分页、排序）
    GetEpisodes(ctx context.Context, params *AudioEpisodeQueryParams) ([]*models.AudioEpisode, int64, error)
    
    // CreateEpisode 创建分集
    CreateEpisode(ctx context.Context, episode *models.AudioEpisode) error
    
    // UpdateEpisode 更新分集
    UpdateEpisode(ctx context.Context, episode *models.AudioEpisode) error
    
    // DeleteEpisode 删除分集
    DeleteEpisode(ctx context.Context, id int64) error
    
    // UpdateStatus 更新分集状态
    UpdateStatus(ctx context.Context, id int64, status string) error
}

// AudioEpisodeQueryParams 音频分集查询参数
type AudioEpisodeQueryParams struct {
    AudioID  int64  // 音频ID（必填）
    SeasonID *int64 // 分季ID（可选）
    Page     int    // 页码，从1开始
    Limit    int    // 每页数量
    Status   string // 状态筛选
    IsVip    *bool  // 是否VIP筛选
    OrderBy  string // 排序字段（sort_order、created_at、updated_at）
    OrderDir string // 排序方向（asc、desc）
}
```

### 2. 实现AudioEpisodeService接口

在同一个文件中实现AudioEpisodeService接口：

```go
// audioEpisodeService AudioEpisodeService的实现
type audioEpisodeService struct {
    episodeRepo repository.AudioEpisodeRepository
    audioFileRepo repository.AudioFileRepository
    audioRepo    repository.AudioRepository
    seasonRepo   repository.AudioSeasonRepository
    audioSvc     AudioService
}

// NewAudioEpisodeService 创建AudioEpisodeService实例
func NewAudioEpisodeService(
    episodeRepo repository.AudioEpisodeRepository,
    audioFileRepo repository.AudioFileRepository,
    audioRepo repository.AudioRepository,
    seasonRepo repository.AudioSeasonRepository,
    audioSvc AudioService,
) AudioEpisodeService {
    return &audioEpisodeService{
        episodeRepo:  episodeRepo,
        audioFileRepo: audioFileRepo,
        audioRepo:    audioRepo,
        seasonRepo:   seasonRepo,
        audioSvc:     audioSvc,
    }
}

// GetEpisode 根据ID获取分集
func (s *audioEpisodeService) GetEpisode(ctx context.Context, id int64) (*models.AudioEpisode, error) {
    if id <= 0 {
        return nil, errors.New("分集ID无效")
    }
    
    episode, err := s.episodeRepo.GetByID(ctx, id)
    if err != nil {
        return nil, fmt.Errorf("获取分集失败: %w", err)
    }
    
    return episode, nil
}

// GetEpisodes 获取分集列表
func (s *audioEpisodeService) GetEpisodes(ctx context.Context, params *AudioEpisodeQueryParams) ([]*models.AudioEpisode, int64, error) {
    if params.AudioID <= 0 {
        return nil, 0, errors.New("音频ID无效")
    }
    
    // 验证音频是否存在
    _, err := s.audioRepo.GetByID(ctx, params.AudioID)
    if err != nil {
        return nil, 0, errors.New("音频不存在")
    }
    
    // 构建查询参数
    queryParams := &repository.AudioEpisodeQueryParams{
        Page:     params.Page,
        Limit:    params.Limit,
        Status:   params.Status,
        IsVip:    params.IsVip,
        OrderBy:  params.OrderBy,
        OrderDir: params.OrderDir,
    }
    
    // 验证和设置默认值
    if queryParams.Page < 1 {
        queryParams.Page = 1
    }
    if queryParams.Limit < 1 {
        queryParams.Limit = 20
    }
    if queryParams.Limit > 100 {
        queryParams.Limit = 100
    }
    if queryParams.OrderBy == "" {
        queryParams.OrderBy = "sort_order"
    }
    if queryParams.OrderDir == "" {
        queryParams.OrderDir = "asc"
    }
    
    var episodes []*models.AudioEpisode
    var total int64
    var err2 error
    
    if params.SeasonID != nil {
        episodes, total, err2 = s.episodeRepo.GetBySeasonID(ctx, *params.SeasonID, queryParams)
    } else {
        episodes, total, err2 = s.episodeRepo.GetByAudioID(ctx, params.AudioID, queryParams)
    }
    
    if err2 != nil {
        return nil, 0, fmt.Errorf("获取分集列表失败: %w", err2)
    }
    
    return episodes, total, nil
}

// CreateEpisode 创建分集
func (s *audioEpisodeService) CreateEpisode(ctx context.Context, episode *models.AudioEpisode) error {
    // 业务验证
    if err := s.validateEpisode(episode); err != nil {
        return err
    }
    
    // 验证音频是否存在
    _, err := s.audioRepo.GetByID(ctx, episode.AudioID)
    if err != nil {
        return errors.New("音频不存在")
    }
    
    // 验证分季是否存在（如果指定了分季）
    if episode.SeasonID != nil {
        _, err := s.seasonRepo.GetByID(ctx, *episode.SeasonID)
        if err != nil {
            return errors.New("分季不存在")
        }
    }
    
    // 设置默认值
    if episode.Status == "" {
        episode.Status = "draft"
    }
    if episode.SortOrder == 0 {
        // 获取当前最大排序值
        episodes, _, err := s.episodeRepo.GetByAudioID(ctx, episode.AudioID, &repository.AudioEpisodeQueryParams{
            Page:     1,
            Limit:    10000,
            OrderBy:  "sort_order",
            OrderDir: "desc",
        })
        if err == nil && len(episodes) > 0 {
            episode.SortOrder = episodes[0].SortOrder + 1
        } else {
            episode.SortOrder = 1
        }
    }
    
    // 创建分集
    if err := s.episodeRepo.Create(ctx, episode); err != nil {
        return fmt.Errorf("创建分集失败: %w", err)
    }
    
    // 如果分集已发布，更新音频总时长
    if episode.Status == "published" {
        if err := s.audioSvc.UpdateDuration(ctx, episode.AudioID); err != nil {
            // 记录错误但不中断流程
        }
    }
    
    return nil
}

// UpdateEpisode 更新分集
func (s *audioEpisodeService) UpdateEpisode(ctx context.Context, episode *models.AudioEpisode) error {
    // 业务验证
    if err := s.validateEpisode(episode); err != nil {
        return err
    }
    
    // 检查分集是否存在
    existing, err := s.episodeRepo.GetByID(ctx, episode.ID)
    if err != nil {
        return errors.New("分集不存在")
    }
    
    // 验证音频ID是否改变（不允许改变）
    if existing.AudioID != episode.AudioID {
        return errors.New("不能修改分集所属的音频")
    }
    
    // 更新分集
    if err := s.episodeRepo.Update(ctx, episode); err != nil {
        return fmt.Errorf("更新分集失败: %w", err)
    }
    
    // 如果状态变为已发布或从已发布变为其他状态，更新音频总时长
    if (episode.Status == "published" && existing.Status != "published") || 
       (existing.Status == "published" && episode.Status != "published") {
        if err := s.audioSvc.UpdateDuration(ctx, episode.AudioID); err != nil {
            // 记录错误但不中断流程
        }
    }
    
    return nil
}

// DeleteEpisode 删除分集
func (s *audioEpisodeService) DeleteEpisode(ctx context.Context, id int64) error {
    if id <= 0 {
        return errors.New("分集ID无效")
    }
    
    // 检查分集是否存在
    episode, err := s.episodeRepo.GetByID(ctx, id)
    if err != nil {
        return errors.New("分集不存在")
    }
    
    audioID := episode.AudioID
    
    // 删除分集
    if err := s.episodeRepo.Delete(ctx, id); err != nil {
        return fmt.Errorf("删除分集失败: %w", err)
    }
    
    // 如果分集已发布，更新音频总时长
    if episode.Status == "published" {
        if err := s.audioSvc.UpdateDuration(ctx, audioID); err != nil {
            // 记录错误但不中断流程
        }
    }
    
    return nil
}

// UpdateStatus 更新分集状态
func (s *audioEpisodeService) UpdateStatus(ctx context.Context, id int64, status string) error {
    if id <= 0 {
        return errors.New("分集ID无效")
    }
    
    // 验证状态
    validStatuses := []string{"draft", "reviewing", "published", "archived"}
    isValid := false
    for _, validStatus := range validStatuses {
        if status == validStatus {
            isValid = true
            break
        }
    }
    if !isValid {
        return errors.New("无效的分集状态")
    }
    
    // 检查分集是否存在
    episode, err := s.episodeRepo.GetByID(ctx, id)
    if err != nil {
        return errors.New("分集不存在")
    }
    
    oldStatus := episode.Status
    
    // 更新状态
    if err := s.episodeRepo.UpdateStatus(ctx, id, status); err != nil {
        return fmt.Errorf("更新分集状态失败: %w", err)
    }
    
    // 如果状态变为已发布或从已发布变为其他状态，更新音频总时长
    if (status == "published" && oldStatus != "published") || 
       (oldStatus == "published" && status != "published") {
        if err := s.audioSvc.UpdateDuration(ctx, episode.AudioID); err != nil {
            // 记录错误但不中断流程
        }
    }
    
    return nil
}

// validateEpisode 验证分集数据
func (s *audioEpisodeService) validateEpisode(episode *models.AudioEpisode) error {
    if episode.AudioID <= 0 {
        return errors.New("音频ID无效")
    }
    
    if episode.EpisodeTitle == "" {
        return errors.New("分集标题不能为空")
    }
    if len(episode.EpisodeTitle) > 200 {
        return errors.New("分集标题长度不能超过200个字符")
    }
    
    // 验证状态
    if episode.Status != "" {
        validStatuses := []string{"draft", "reviewing", "published", "archived"}
        isValid := false
        for _, validStatus := range validStatuses {
            if episode.Status == validStatus {
                isValid = true
                break
            }
        }
        if !isValid {
            return errors.New("无效的分集状态")
        }
    }
    
    return nil
}
```
