```go
func TestConfigHandler_GetAudioAttachmentConfig(t *testing.T) {
    tests := []struct {
        name           string
        setupService   func() service.ConfigService
        expectedStatus int
        expectedError  string
    }{
        {
            name: "正常获取音频附件设置",
            setupService: func() service.ConfigService {
                mockService := &MockConfigService{}
                mockService.On("GetConfig", mock.Anything, "attachment").Return(map[string]interface{}{
                    "audio": map[string]interface{}{
                        "enabled":            true,
                        "storage_channel_id": 0,
                        "allowed_types":      []interface{}{"mp3", "wav", "flac"},
                        "max_size":           int64(50 * 1024 * 1024), // 50MB
                        "storage_path":       "/uploads/audio",
                        "convert_format":     "mp3",
                        "transcode_params": map[string]interface{}{
                            "encoder":    "libmp3lame",
                            "sample_rate": 44100,
                            "channels":   2,
                            "bitrate":    128,
                        },
                        "watermark_enabled":  false,
                        "watermark_file":     "",
                        "watermark_position": "",
                    },
                }, nil)
                return mockService
            },
            expectedStatus: http.StatusOK,
        },
        {
            name: "配置不存在",
            setupService: func() service.ConfigService {
                mockService := &MockConfigService{}
                mockService.On("GetConfig", mock.Anything, "attachment").Return(nil, fmt.Errorf("配置类型 attachment 不存在"))
                return mockService
            },
            expectedStatus: http.StatusInternalServerError,
            expectedError:  "获取附件设置失败",
        },
    }
    
    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            router := setupTestRouter(tt.setupService())
            
            req := httptest.NewRequest(http.MethodGet, "/api/v1/system/config/attachment/audio", nil)
            w := httptest.NewRecorder()
            router.ServeHTTP(w, req)
            
            assert.Equal(t, tt.expectedStatus, w.Code)
            
            var response map[string]interface{}
            err := json.Unmarshal(w.Body.Bytes(), &response)
            require.NoError(t, err)
            
            if tt.expectedError != "" {
                errors, ok := response["errors"].([]interface{})
                assert.True(t, ok)
                assert.NotEmpty(t, errors)
            } else {
                data, ok := response["data"].(map[string]interface{})
                assert.True(t, ok)
                assert.NotNil(t, data["enabled"])
            }
        })
    }
}
```

```go
func TestConfigHandler_UpdateAudioAttachmentConfig(t *testing.T) {
    tests := []struct {
        name           string
        setupService   func() service.ConfigService
        requestBody    interface{}
        expectedStatus int
        expectedError  string
    }{
        {
            name: "正常更新音频附件设置",
            setupService: func() service.ConfigService {
                mockService := &MockConfigService{}
                mockService.On("GetConfig", mock.Anything, "attachment").Return(map[string]interface{}{}, nil).Once()
                mockService.On("UpdateConfig", mock.Anything, "attachment", mock.Anything).Return(nil)
                mockService.On("GetConfig", mock.Anything, "attachment").Return(map[string]interface{}{
                    "audio": map[string]interface{}{
                        "enabled":            true,
                        "storage_channel_id": 1,
                        "allowed_types":      []interface{}{"mp3", "wav"},
                        "max_size":           int64(100 * 1024 * 1024),
                        "storage_path":       "/uploads/audio",
                        "convert_format":     "mp3",
                        "transcode_params": map[string]interface{}{
                            "encoder":    "libmp3lame",
                            "sample_rate": 48000,
                            "channels":   2,
                            "bitrate":    192,
                        },
                        "watermark_enabled":  true,
                        "watermark_file":     "/watermark/audio.wav",
                        "watermark_position": "begin",
                    },
                }, nil).Once()
                return mockService
            },
            requestBody: dto.UpdateAudioAttachmentConfigRequest{
                Enabled:          true,
                StorageChannelID: 1,
                AllowedTypes:     []string{"mp3", "wav"},
                MaxSize:          100 * 1024 * 1024,
                StoragePath:      "/uploads/audio",
                ConvertFormat:    "mp3",
                TranscodeParams: dto.AudioTranscodeParams{
                    Encoder:    "libmp3lame",
                    SampleRate: 48000,
                    Channels:   2,
                    Bitrate:    192,
                },
                WatermarkEnabled: true,
                WatermarkFile:    "/watermark/audio.wav",
                WatermarkPosition: "begin",
            },
            expectedStatus: http.StatusOK,
        },
        {
            name: "请求参数验证失败-上传大小为0",
            setupService: func() service.ConfigService {
                return &MockConfigService{}
            },
            requestBody: dto.UpdateAudioAttachmentConfigRequest{
                Enabled:          true,
                StorageChannelID: 0,
                AllowedTypes:     []string{"mp3"},
                MaxSize:          0,
                StoragePath:      "/uploads/audio",
            },
            expectedStatus: http.StatusBadRequest,
            expectedError:  "请求参数错误",
        },
    }
    
    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            router := setupTestRouter(tt.setupService())
            
            requestBodyBytes, err := json.Marshal(tt.requestBody)
            require.NoError(t, err)
            
            req := httptest.NewRequest(http.MethodPut, "/api/v1/system/config/attachment/audio", bytes.NewBuffer(requestBodyBytes))
            req.Header.Set("Content-Type", "application/json")
            w := httptest.NewRecorder()
            router.ServeHTTP(w, req)
            
            assert.Equal(t, tt.expectedStatus, w.Code)
            
            var response map[string]interface{}
            err = json.Unmarshal(w.Body.Bytes(), &response)
            require.NoError(t, err)
            
            if tt.expectedError != "" {
                errors, ok := response["errors"].([]interface{})
                assert.True(t, ok)
                assert.NotEmpty(t, errors)
                errorMsg := errors[0].(map[string]interface{})["message"].(string)
                assert.Contains(t, errorMsg, tt.expectedError)
            } else {
                data, ok := response["data"].(map[string]interface{})
                assert.True(t, ok)
                assert.NotNil(t, data["enabled"])
            }
        })
    }
}
```
