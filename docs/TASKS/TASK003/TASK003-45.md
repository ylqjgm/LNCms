```go
func TestConfigHandler_TestEmailConfig(t *testing.T) {
    tests := []struct {
        name           string
        setupService   func() service.ConfigService
        requestBody    interface{}
        expectedStatus int
        expectedError  string
        expectedSuccess bool
    }{
        {
            name: "正常测试邮箱配置",
            setupService: func() service.ConfigService {
                mockService := &MockConfigService{}
                mockService.On("GetConfig", mock.Anything, "email").Return(map[string]interface{}{
                    "smtp_host":   "smtp.example.com",
                    "smtp_port":   587,
                    "tls_enabled": true,
                    "username":    "user@example.com",
                    "password":    "password123",
                    "from_email":  "noreply@example.com",
                }, nil)
                return mockService
            },
            requestBody: dto.TestEmailConfigRequest{
                TestEmail: "test@example.com",
            },
            expectedStatus:  http.StatusOK,
            expectedSuccess: true,
        },
        {
            name: "请求参数验证失败-测试邮箱地址格式错误",
            setupService: func() service.ConfigService {
                return &MockConfigService{}
            },
            requestBody: dto.TestEmailConfigRequest{
                TestEmail: "invalid-email",
            },
            expectedStatus: http.StatusBadRequest,
            expectedError:  "请求参数错误",
        },
        {
            name: "邮箱配置不完整",
            setupService: func() service.ConfigService {
                mockService := &MockConfigService{}
                mockService.On("GetConfig", mock.Anything, "email").Return(map[string]interface{}{
                    "smtp_host": "smtp.example.com",
                    // 缺少其他必需字段
                }, nil)
                return mockService
            },
            requestBody: dto.TestEmailConfigRequest{
                TestEmail: "test@example.com",
            },
            expectedStatus: http.StatusBadRequest,
            expectedError:  "邮箱配置不完整",
        },
        {
            name: "邮箱配置不存在",
            setupService: func() service.ConfigService {
                mockService := &MockConfigService{}
                mockService.On("GetConfig", mock.Anything, "email").Return(nil, fmt.Errorf("配置类型 email 不存在"))
                return mockService
            },
            requestBody: dto.TestEmailConfigRequest{
                TestEmail: "test@example.com",
            },
            expectedStatus: http.StatusInternalServerError,
            expectedError:  "获取邮箱设置失败",
        },
    }
    
    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            router := setupTestRouter(tt.setupService())
            
            requestBodyBytes, err := json.Marshal(tt.requestBody)
            require.NoError(t, err)
            
            req := httptest.NewRequest(http.MethodPost, "/api/v1/system/config/email/test", bytes.NewBuffer(requestBodyBytes))
            req.Header.Set("Content-Type", "application/json")
            w := httptest.NewRecorder()
            router.ServeHTTP(w, req)
            
            assert.Equal(t, tt.expectedStatus, w.Code)
            
            var response map[string]interface{}
            err = json.Unmarshal(w.Body.Bytes(), &response)
            require.NoError(t, err)
            
            if tt.expectedError != "" {
                errors, ok := response["errors"].([]interface{})
                assert.True(t, ok)
                assert.NotEmpty(t, errors)
                errorMsg := errors[0].(map[string]interface{})["message"].(string)
                assert.Contains(t, errorMsg, tt.expectedError)
            } else {
                data, ok := response["data"].(map[string]interface{})
                assert.True(t, ok)
                if tt.expectedSuccess {
                    success, ok := data["success"].(bool)
                    assert.True(t, ok)
                    assert.True(t, success)
                }
                assert.NotNil(t, data["message"])
            }
        })
    }
}
```

type ConfigHandler struct {

type MockEmailTester struct {
    mock.Mock
}

func (m *MockEmailTester) TestConnection(smtpHost string, smtpPort int, tlsEnabled bool, username, password, fromEmail, testEmail string) (bool, string) {
