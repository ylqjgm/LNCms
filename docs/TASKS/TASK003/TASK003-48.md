func TestConfigHandler_GetShieldConfig(t *testing.T) {
    tests := []struct {
        name           string
        setupService   func() service.ConfigService
        expectedStatus int
        expectedError  string
    }{
        {
            name: "正常获取屏蔽设置",
            setupService: func() service.ConfigService {
                mockService := &MockConfigService{}
                mockService.On("GetConfig", mock.Anything, "shield").Return(map[string]interface{}{
                    "ip_blacklist":     []interface{}{"192.168.1.100", "10.0.0.1"},
                    "user_id_blacklist": []interface{}{float64(1), float64(2), float64(3)},
                    "forbidden_words":  []interface{}{"敏感词1", "敏感词2"},
                }, nil)
                return mockService
            },
            expectedStatus: http.StatusOK,
        },
        {
            name: "配置不存在",
            setupService: func() service.ConfigService {
                mockService := &MockConfigService{}
                mockService.On("GetConfig", mock.Anything, "shield").Return(nil, fmt.Errorf("配置类型 shield 不存在"))
                return mockService
            },
            expectedStatus: http.StatusInternalServerError,
            expectedError:  "获取屏蔽设置失败",
        },
    }
    
    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            router := setupTestRouter(tt.setupService())
            
            req := httptest.NewRequest(http.MethodGet, "/api/v1/system/config/shield", nil)
            w := httptest.NewRecorder()
            router.ServeHTTP(w, req)
            
            assert.Equal(t, tt.expectedStatus, w.Code)
            
            var response map[string]interface{}
            err := json.Unmarshal(w.Body.Bytes(), &response)
            require.NoError(t, err)
            
            if tt.expectedError != "" {
                errors, ok := response["errors"].([]interface{})
                assert.True(t, ok)
                assert.NotEmpty(t, errors)
            } else {
                data, ok := response["data"].(map[string]interface{})
                assert.True(t, ok)
                assert.NotNil(t, data["ip_blacklist"])
            }
        })
    }
}
```

func TestConfigHandler_UpdateShieldConfig(t *testing.T) {
    tests := []struct {
        name           string
        setupService   func() service.ConfigService
        requestBody    interface{}
        expectedStatus int
        expectedError  string
    }{
        {
            name: "正常更新屏蔽设置",
            setupService: func() service.ConfigService {
                mockService := &MockConfigService{}
                mockService.On("GetConfig", mock.Anything, "shield").Return(map[string]interface{}{}, nil).Once()
                mockService.On("UpdateConfig", mock.Anything, "shield", mock.Anything).Return(nil)
                mockService.On("GetConfig", mock.Anything, "shield").Return(map[string]interface{}{
                    "ip_blacklist":     []interface{}{"192.168.1.100", "10.0.0.1"},
                    "user_id_blacklist": []interface{}{float64(1), float64(2)},
                    "forbidden_words":  []interface{}{"敏感词1", "敏感词2", "敏感词3"},
                }, nil).Once()
                return mockService
            },
            requestBody: dto.UpdateShieldConfigRequest{
                IPBlacklist:     []string{"192.168.1.100", "10.0.0.1"},
                UserIDBlacklist: []int64{1, 2},
                ForbiddenWords:  []string{"敏感词1", "敏感词2", "敏感词3"},
            },
            expectedStatus: http.StatusOK,
        },
        {
            name: "正常更新屏蔽设置-空列表",
            setupService: func() service.ConfigService {
                mockService := &MockConfigService{}
                mockService.On("GetConfig", mock.Anything, "shield").Return(map[string]interface{}{}, nil).Once()
                mockService.On("UpdateConfig", mock.Anything, "shield", mock.Anything).Return(nil)
                mockService.On("GetConfig", mock.Anything, "shield").Return(map[string]interface{}{
                    "ip_blacklist":     []interface{}{},
                    "user_id_blacklist": []interface{}{},
                    "forbidden_words":  []interface{}{},
                }, nil).Once()
                return mockService
            },
            requestBody: dto.UpdateShieldConfigRequest{
                IPBlacklist:     []string{},
                UserIDBlacklist: []int64{},
                ForbiddenWords:  []string{},
            },
            expectedStatus: http.StatusOK,
        },
        {
            name: "业务验证失败-IP地址格式无效",
            setupService: func() service.ConfigService {
                return &MockConfigService{}
            },
            requestBody: dto.UpdateShieldConfigRequest{
                IPBlacklist:     []string{"192.168.1.100", "invalid-ip"},
                UserIDBlacklist: []int64{},
                ForbiddenWords:  []string{},
            },
            expectedStatus: http.StatusBadRequest,
            expectedError:  "IP地址格式无效",
        },
    }
    
    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            router := setupTestRouter(tt.setupService())
            
            requestBodyBytes, err := json.Marshal(tt.requestBody)
            require.NoError(t, err)
            
            req := httptest.NewRequest(http.MethodPut, "/api/v1/system/config/shield", bytes.NewBuffer(requestBodyBytes))
            req.Header.Set("Content-Type", "application/json")
            w := httptest.NewRecorder()
            router.ServeHTTP(w, req)
            
            assert.Equal(t, tt.expectedStatus, w.Code)
            
            var response map[string]interface{}
            err = json.Unmarshal(w.Body.Bytes(), &response)
            require.NoError(t, err)
            
            if tt.expectedError != "" {
                errors, ok := response["errors"].([]interface{})
                assert.True(t, ok)
                assert.NotEmpty(t, errors)
                errorMsg := errors[0].(map[string]interface{})["message"].(string)
                assert.Contains(t, errorMsg, tt.expectedError)
            } else {
                data, ok := response["data"].(map[string]interface{})
                assert.True(t, ok)
                assert.NotNil(t, data["ip_blacklist"])
            }
        })
    }
}
```
