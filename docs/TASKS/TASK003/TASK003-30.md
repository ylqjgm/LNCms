### 1. 实现DeleteStorageChannel方法

在ConfigHandler中实现DeleteStorageChannel方法：

```go
// DeleteStorageChannel 删除存储渠道
// @Summary 删除存储渠道
// @Description 删除附件存储渠道，需要管理员权限
// @Tags 系统配置
// @Accept json
// @Produce json
// @Param channel_id path int true "渠道ID"
// @Success 200 {object} response.Envelope "成功响应"
// @Failure 400 {object} response.Envelope "请求参数错误"
// @Failure 401 {object} response.Envelope "未授权"
// @Failure 403 {object} response.Envelope "权限不足"
// @Failure 404 {object} response.Envelope "渠道不存在"
// @Failure 500 {object} response.Envelope "服务器错误"
// @Router /api/v1/system/config/storage/channels/{channel_id} [delete]
// @Security BearerAuth
func (h *ConfigHandler) DeleteStorageChannel(c *gin.Context) {
    ctx := c.Request.Context()
    
    // 权限验证（需要管理员权限）
    // 注意：权限验证逻辑应该在中间件中实现
    
    // 获取路径参数
    channelIDStr := c.Param("channel_id")
    channelID, err := strconv.Atoi(channelIDStr)
    if err != nil {
        response.Error(c, http.StatusBadRequest, "INVALID_REQUEST", "渠道ID格式错误", err)
        return
    }
    
    // 获取当前附件配置
    configData, err := h.configService.GetConfig(ctx, string(model.ConfigTypeAttachment))
    if err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_GET_FAILED", "获取附件设置失败", err)
        return
    }
    
    // 解析当前配置
    configMap, ok := configData.(map[string]interface{})
    if !ok {
        response.Error(c, http.StatusInternalServerError, "CONFIG_PARSE_FAILED", "解析附件设置失败", nil)
        return
    }
    
    // 获取存储渠道列表
    channels, ok := configMap["storage_channels"].([]interface{})
    if !ok {
        response.Error(c, http.StatusNotFound, "CHANNEL_NOT_FOUND", fmt.Sprintf("渠道 %d 不存在", channelID), nil)
        return
    }
    
    // 检查渠道ID范围
    if channelID < 0 || channelID >= len(channels) {
        response.Error(c, http.StatusNotFound, "CHANNEL_NOT_FOUND", fmt.Sprintf("渠道 %d 不存在", channelID), nil)
        return
    }
    
    // 删除渠道（从数组中移除）
    newChannels := make([]interface{}, 0, len(channels)-1)
    for i, channelInterface := range channels {
        if i == channelID {
            continue // 跳过要删除的渠道
        }
        newChannels = append(newChannels, channelInterface)
    }
    
    // 更新配置
    configMap["storage_channels"] = newChannels
    if err := h.configService.UpdateConfig(ctx, string(model.ConfigTypeAttachment), configMap); err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_UPDATE_FAILED", "删除存储渠道失败", err)
        return
    }
    
    // 返回成功响应
    response.Success(c, nil)
}
```
