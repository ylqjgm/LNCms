                }, nil)
                return mockService
            },
            requestBody: dto.AddClusterNodeRequest{
                NodeID:     "node-001",

            expectedError:  "节点标识 node-001 已存在",

        {
            name: "业务验证失败-节点权重超出范围",

                mockService.On("GetConfig", mock.Anything, "cluster").Return(map[string]interface{}{

                return mockService

            expectedStatus: http.StatusBadRequest,
            expectedError:  "节点权重必须在1-100之间",
        },

            router.ServeHTTP(w, req)
            
            assert.Equal(t, tt.expectedStatus, w.Code)
            
            var response map[string]interface{}
            err = json.Unmarshal(w.Body.Bytes(), &response)
            require.NoError(t, err)
            
            if tt.expectedError != "" {
                errors, ok := response["errors"].([]interface{})
                assert.True(t, ok)
                assert.NotEmpty(t, errors)
                errorMsg := errors[0].(map[string]interface{})["message"].(string)
                assert.Contains(t, errorMsg, tt.expectedError)
            } else {
                data, ok := response["data"].(map[string]interface{})
                assert.True(t, ok)
                assert.NotNil(t, data["node_id"])
            }
        })
    }
