### 1. 定义响应结构体

在 `internal/api/v1/dto/config_dto.go` 文件中定义邮箱设置响应结构体：

```go
package dto

// EmailConfigResponse 邮箱设置响应
type EmailConfigResponse struct {
    SMTPHost     string `json:"smtp_host"`      // SMTP服务器
    SMTPPort     int    `json:"smtp_port"`      // SMTP端口
    TLSEnabled   bool   `json:"tls_enabled"`    // TLS开关
    Username     string `json:"username"`       // 登录用户名
    Password     string `json:"password"`       // 登录密码（出于安全考虑，返回空字符串或标记为已设置）
    FromEmail    string `json:"from_email"`     // 发件人邮箱
}
```

### 2. 实现GetEmailConfig方法

在ConfigHandler中实现GetEmailConfig方法：

```go
// GetEmailConfig 获取邮箱设置
// @Summary 获取邮箱设置
// @Description 获取邮箱信息配置，密码字段出于安全考虑不返回明文
// @Tags 系统配置
// @Accept json
// @Produce json
// @Success 200 {object} response.Envelope{data=dto.EmailConfigResponse} "成功响应"
// @Failure 500 {object} response.Envelope "服务器错误"
// @Router /api/v1/system/config/email [get]
func (h *ConfigHandler) GetEmailConfig(c *gin.Context) {
    ctx := c.Request.Context()
    
    // 从服务层获取配置
    configData, err := h.configService.GetConfig(ctx, string(model.ConfigTypeEmail))
    if err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_GET_FAILED", "获取邮箱设置失败", err)
        return
    }
    
    // 将配置数据转换为响应结构体
    var emailConfig dto.EmailConfigResponse
    if err := mapToEmailConfig(configData, &emailConfig); err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_PARSE_FAILED", "解析邮箱设置失败", err)
        return
    }
    
    // 出于安全考虑，密码字段不返回明文
    if emailConfig.Password != "" {
        emailConfig.Password = "***" // 标记为已设置，但不返回实际密码
    }
    
    // 返回成功响应
    response.Success(c, emailConfig)
}

// mapToEmailConfig 将配置数据映射到EmailConfigResponse
func mapToEmailConfig(data interface{}, emailConfig *dto.EmailConfigResponse) error {
    configMap, ok := data.(map[string]interface{})
    if !ok {
        return fmt.Errorf("配置数据格式错误")
    }
    
    // 映射字段
    if v, ok := configMap["smtp_host"].(string); ok {
        emailConfig.SMTPHost = v
    }
    if v, ok := configMap["smtp_port"].(float64); ok {
        emailConfig.SMTPPort = int(v)
    }
    if v, ok := configMap["tls_enabled"].(bool); ok {
        emailConfig.TLSEnabled = v
    }
    if v, ok := configMap["username"].(string); ok {
        emailConfig.Username = v
    }
    if v, ok := configMap["password"].(string); ok {
        emailConfig.Password = v
    }
    if v, ok := configMap["from_email"].(string); ok {
        emailConfig.FromEmail = v
    }
    
    return nil
}
```
