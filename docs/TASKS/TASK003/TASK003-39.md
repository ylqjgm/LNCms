// UpdateVideoAttachmentConfig 更新视频附件设置
// @Summary 更新视频附件设置
// @Description 更新视频附件配置，需要管理员权限
// @Tags 系统配置
// @Accept json
// @Produce json
// @Param request body dto.UpdateVideoAttachmentConfigRequest true "更新视频附件设置请求"
// @Success 200 {object} response.Envelope{data=dto.VideoAttachmentConfigResponse} "成功响应"
// @Failure 400 {object} response.Envelope "请求参数错误"
// @Failure 401 {object} response.Envelope "未授权"
// @Failure 403 {object} response.Envelope "权限不足"
// @Failure 500 {object} response.Envelope "服务器错误"
// @Router /api/v1/system/config/attachment/video [put]
// @Security BearerAuth
func (h *ConfigHandler) UpdateVideoAttachmentConfig(c *gin.Context) {
    ctx := c.Request.Context()
    
    // 权限验证（需要管理员权限）
    // 注意：权限验证逻辑应该在中间件中实现
    
    // 绑定请求参数
    var req dto.UpdateVideoAttachmentConfigRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        response.Error(c, http.StatusBadRequest, "INVALID_REQUEST", "请求参数错误", err)
        return
    }
    
    // 获取当前附件配置
    configData, err := h.configService.GetConfig(ctx, string(model.ConfigTypeAttachment))
    if err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_GET_FAILED", "获取附件设置失败", err)
        return
    }
    
    // 解析当前配置
    configMap, ok := configData.(map[string]interface{})
    if !ok {
        configMap = make(map[string]interface{})
    }
    
    // 构建视频附件配置数据
    videoConfigData := map[string]interface{}{
        "enabled":            req.Enabled,
        "storage_channel_id": req.StorageChannelID,
        "allowed_types":      req.AllowedTypes,
        "max_size":           req.MaxSize,
        "storage_path":       req.StoragePath,
        "convert_format":     req.ConvertFormat,
        "transcode_params": map[string]interface{}{
            "encoder":    req.TranscodeParams.Encoder,
            "frame_rate": req.TranscodeParams.FrameRate,
            "resolution": req.TranscodeParams.Resolution,
            "bitrate":    req.TranscodeParams.Bitrate,
        },
        "watermark_enabled":   req.WatermarkEnabled,
        "watermark_type":      req.WatermarkType,
        "watermark_position":  req.WatermarkPosition,
        "watermark_start_time": req.WatermarkStartTime,
        "watermark_duration":   req.WatermarkDuration,
    }
    
    // 更新配置
    configMap["video"] = videoConfigData
    if err := h.configService.UpdateConfig(ctx, string(model.ConfigTypeAttachment), configMap); err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_UPDATE_FAILED", "更新视频附件设置失败", err)
        return
    }
    
    // 获取更新后的配置并返回
    updatedData, err := h.configService.GetConfig(ctx, string(model.ConfigTypeAttachment))
    if err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_GET_FAILED", "获取更新后的视频附件设置失败", err)
        return
    }
    
    configMap, ok = updatedData.(map[string]interface{})
    if !ok {
        response.Error(c, http.StatusInternalServerError, "CONFIG_PARSE_FAILED", "解析附件设置失败", nil)
        return
    }
    
    videoConfig, ok := configMap["video"].(map[string]interface{})
    if !ok {
        response.Error(c, http.StatusInternalServerError, "CONFIG_PARSE_FAILED", "解析视频附件设置失败", nil)
        return
    }
    
    var videoAttachmentConfig dto.VideoAttachmentConfigResponse
    if err := mapToVideoAttachmentConfig(videoConfig, &videoAttachmentConfig); err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_PARSE_FAILED", "解析视频附件设置失败", err)
        return
    }
    
    // 返回成功响应
    response.Success(c, videoAttachmentConfig)
}
