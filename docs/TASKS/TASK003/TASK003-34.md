func TestConfigHandler_GetImageAttachmentConfig(t *testing.T) {
    tests := []struct {
        name           string
        setupService   func() service.ConfigService
        expectedStatus int
        expectedError  string
    }{
        {
            name: "正常获取图片附件设置",
            setupService: func() service.ConfigService {
                mockService := &MockConfigService{}
                mockService.On("GetConfig", mock.Anything, "attachment").Return(map[string]interface{}{
                    "image": map[string]interface{}{
                        "enabled":            true,
                        "storage_channel_id": 0,
                        "allowed_types":      []interface{}{"jpg", "png", "gif"},
                        "max_size":           int64(5 * 1024 * 1024), // 5MB
                        "storage_path":       "/uploads/images",
                        "convert_format":     "jpg",
                        "quality":            85,
                        "watermark_enabled":  false,
                        "watermark_type":     "",
                        "watermark_position": "",
                        "watermark_opacity":  0.5,
                        "thumbnails":         []interface{}{},
                    },
                }, nil)
                return mockService
            },
            expectedStatus: http.StatusOK,
        },
        {
            name: "配置不存在",
            setupService: func() service.ConfigService {
                mockService := &MockConfigService{}
                mockService.On("GetConfig", mock.Anything, "attachment").Return(nil, fmt.Errorf("配置类型 attachment 不存在"))
                return mockService
            },
            expectedStatus: http.StatusInternalServerError,
            expectedError:  "获取附件设置失败",
        },
    }
    
    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            router := setupTestRouter(tt.setupService())
            
            req := httptest.NewRequest(http.MethodGet, "/api/v1/system/config/attachment/image", nil)
            w := httptest.NewRecorder()
            router.ServeHTTP(w, req)
            
            assert.Equal(t, tt.expectedStatus, w.Code)
            
            var response map[string]interface{}
            err := json.Unmarshal(w.Body.Bytes(), &response)
            require.NoError(t, err)
            
            if tt.expectedError != "" {
                errors, ok := response["errors"].([]interface{})
                assert.True(t, ok)
                assert.NotEmpty(t, errors)
            } else {
                data, ok := response["data"].(map[string]interface{})
                assert.True(t, ok)
                assert.NotNil(t, data["enabled"])
            }
        })
    }
}

func TestConfigHandler_UpdateImageAttachmentConfig(t *testing.T) {
    tests := []struct {
        name           string
        setupService   func() service.ConfigService
        requestBody    interface{}
        expectedStatus int
        expectedError  string
    }{
        {
            name: "正常更新图片附件设置",
            setupService: func() service.ConfigService {
                mockService := &MockConfigService{}
                mockService.On("GetConfig", mock.Anything, "attachment").Return(map[string]interface{}{}, nil).Once()
                mockService.On("UpdateConfig", mock.Anything, "attachment", mock.Anything).Return(nil)
                mockService.On("GetConfig", mock.Anything, "attachment").Return(map[string]interface{}{
                    "image": map[string]interface{}{
                        "enabled":            true,
                        "storage_channel_id": 1,
                        "allowed_types":      []interface{}{"jpg", "png"},
                        "max_size":           int64(10 * 1024 * 1024),
                        "storage_path":       "/uploads/images",
                        "convert_format":     "jpg",
                        "quality":            90,
                        "watermark_enabled":  true,
                        "watermark_type":     "text",
                        "watermark_position": "bottom-right",
                        "watermark_opacity":  0.8,
                        "thumbnails":         []interface{}{},
                    },
                }, nil).Once()
                return mockService
            },
            requestBody: dto.UpdateImageAttachmentConfigRequest{
                Enabled:          true,
                StorageChannelID: 1,
                AllowedTypes:     []string{"jpg", "png"},
                MaxSize:          10 * 1024 * 1024,
                StoragePath:      "/uploads/images",
                ConvertFormat:    "jpg",
                Quality:          90,
                WatermarkEnabled: true,
                WatermarkType:    "text",
                WatermarkPosition: "bottom-right",
                WatermarkOpacity: 0.8,
                Thumbnails:       []dto.ImageThumbnail{},
            },
            expectedStatus: http.StatusOK,
        },
        {
            name: "请求参数验证失败-图片质量超出范围",
            setupService: func() service.ConfigService {
                return &MockConfigService{}
            },
            requestBody: dto.UpdateImageAttachmentConfigRequest{
                Enabled:          true,
                StorageChannelID: 0,
                AllowedTypes:     []string{"jpg"},
                MaxSize:          5 * 1024 * 1024,
                StoragePath:      "/uploads/images",
                Quality:          150,
            },
            expectedStatus: http.StatusBadRequest,
            expectedError:  "图片质量必须在1-100之间",
        },
        {
            name: "请求参数验证失败-水印透明度超出范围",
            setupService: func() service.ConfigService {
                return &MockConfigService{}
            },
            requestBody: dto.UpdateImageAttachmentConfigRequest{
                Enabled:          true,
                StorageChannelID: 0,
                AllowedTypes:     []string{"jpg"},
                MaxSize:          5 * 1024 * 1024,
                StoragePath:      "/uploads/images",
                Quality:          85,
                WatermarkOpacity: 1.5,
            },
            expectedStatus: http.StatusBadRequest,
            expectedError:  "水印透明度必须在0.0-1.0之间",
        },
    }
    
    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            router := setupTestRouter(tt.setupService())
            
            requestBodyBytes, err := json.Marshal(tt.requestBody)
            require.NoError(t, err)
            
            req := httptest.NewRequest(http.MethodPut, "/api/v1/system/config/attachment/image", bytes.NewBuffer(requestBodyBytes))
            req.Header.Set("Content-Type", "application/json")
            w := httptest.NewRecorder()
            router.ServeHTTP(w, req)
            
            assert.Equal(t, tt.expectedStatus, w.Code)
            
            var response map[string]interface{}
            err = json.Unmarshal(w.Body.Bytes(), &response)
            require.NoError(t, err)
            
            if tt.expectedError != "" {
                errors, ok := response["errors"].([]interface{})
                assert.True(t, ok)
                assert.NotEmpty(t, errors)
                errorMsg := errors[0].(map[string]interface{})["message"].(string)
                assert.Contains(t, errorMsg, tt.expectedError)
            } else {
                data, ok := response["data"].(map[string]interface{})
                assert.True(t, ok)
                assert.NotNil(t, data["enabled"])
            }
        })
    }
}
