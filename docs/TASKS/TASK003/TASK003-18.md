### 2. 实现UpdateClusterConfig方法

在ConfigHandler中实现UpdateClusterConfig方法：

```go
// UpdateClusterConfig 更新站群设置
// @Summary 更新站群设置
// @Description 更新网站集群部署配置，需要管理员权限
// @Tags 系统配置
// @Accept json
// @Produce json
// @Param request body dto.UpdateClusterConfigRequest true "更新站群设置请求"
// @Success 200 {object} response.Envelope{data=dto.ClusterConfigResponse} "成功响应"
// @Failure 400 {object} response.Envelope "请求参数错误"
// @Failure 401 {object} response.Envelope "未授权"
// @Failure 403 {object} response.Envelope "权限不足"
// @Failure 500 {object} response.Envelope "服务器错误"
// @Router /api/v1/system/config/cluster [put]
// @Security BearerAuth
func (h *ConfigHandler) UpdateClusterConfig(c *gin.Context) {
    ctx := c.Request.Context()
    
    // 权限验证（需要管理员权限）
    // 注意：权限验证逻辑应该在中间件中实现
    
    // 绑定请求参数
    var req dto.UpdateClusterConfigRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        response.Error(c, http.StatusBadRequest, "INVALID_REQUEST", "请求参数错误", err)
        return
    }
    
    // 业务验证：主站地址必须为有效的URL格式
    if req.MasterSite.MasterURL != "" {
        if _, err := url.Parse(req.MasterSite.MasterURL); err != nil {
            response.Error(c, http.StatusBadRequest, "VALIDATION_ERROR", "主站地址格式无效", err)
            return
        }
    }
    
    // 业务验证：节点标识必须唯一
    nodeIDMap := make(map[string]bool)
    for _, node := range req.ClusterNodes {
        if nodeIDMap[node.NodeID] {
            response.Error(c, http.StatusBadRequest, "VALIDATION_ERROR", fmt.Sprintf("节点标识 %s 重复", node.NodeID), nil)
            return
        }
        nodeIDMap[node.NodeID] = true
        
        // 验证节点地址格式
        if _, err := url.Parse(node.NodeURL); err != nil {
            response.Error(c, http.StatusBadRequest, "VALIDATION_ERROR", fmt.Sprintf("节点地址 %s 格式无效", node.NodeURL), err)
            return
        }
        
        // 验证节点权重范围
        if node.NodeWeight < 1 || node.NodeWeight > 100 {
            response.Error(c, http.StatusBadRequest, "VALIDATION_ERROR", "节点权重必须在1-100之间", nil)
            return
        }
    }
    
    // 构建配置数据
    clusterNodesData := make([]map[string]interface{}, 0, len(req.ClusterNodes))
    for _, node := range req.ClusterNodes {
        clusterNodesData = append(clusterNodesData, map[string]interface{}{
            "node_id":     node.NodeID,
            "node_url":    node.NodeURL,
            "node_status": node.NodeStatus,
            "node_weight": node.NodeWeight,
        })
    }
    
    configData := map[string]interface{}{
        "master_site": map[string]interface{}{
            "master_url": req.MasterSite.MasterURL,
            "master_id":  req.MasterSite.MasterID,
        },
        "cluster_nodes": clusterNodesData,
    }
    
    // 调用服务层更新配置
    if err := h.configService.UpdateConfig(ctx, string(model.ConfigTypeCluster), configData); err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_UPDATE_FAILED", "更新站群设置失败", err)
        return
    }
    
    // 获取更新后的配置并返回
    updatedData, err := h.configService.GetConfig(ctx, string(model.ConfigTypeCluster))
    if err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_GET_FAILED", "获取更新后的站群设置失败", err)
        return
    }
    
    var clusterConfig dto.ClusterConfigResponse
    if err := mapToClusterConfig(updatedData, &clusterConfig); err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_PARSE_FAILED", "解析站群设置失败", err)
        return
    }
    
    // 返回成功响应
    response.Success(c, clusterConfig)
}
```

// RegisterConfigRoutes 注册配置相关路由
func RegisterConfigRoutes(r *gin.RouterGroup, configHandler *handler.ConfigHandler) {
    config := r.Group("/system/config")
    {
        config.GET("/site", configHandler.GetSiteConfig)
        config.PUT("/site", configHandler.UpdateSiteConfig)
        config.GET("/register", configHandler.GetRegisterConfig)
        config.PUT("/register", configHandler.UpdateRegisterConfig)
