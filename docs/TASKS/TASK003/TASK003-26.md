```go
func TestConfigHandler_GetPointsConfig(t *testing.T) {
    tests := []struct {
        name           string
        setupService   func() service.ConfigService
        expectedStatus int
        expectedError  string
    }{
        {
            name: "正常获取积分设置",
            setupService: func() service.ConfigService {
                mockService := &MockConfigService{}
                mockService.On("GetConfig", mock.Anything, "points").Return(map[string]interface{}{
                    "points_name":           "积分",
                    "register_points":       10,
                    "login_points":          5,
                    "comment_points":        2,
                    "recharge_points_ratio": 1.0,
                    "consume_points_ratio":  1.0,
                }, nil)
                return mockService
            },
            expectedStatus: http.StatusOK,
        },
        {
            name: "配置不存在",
            setupService: func() service.ConfigService {
                mockService := &MockConfigService{}
                mockService.On("GetConfig", mock.Anything, "points").Return(nil, fmt.Errorf("配置类型 points 不存在"))
                return mockService
            },
            expectedStatus: http.StatusInternalServerError,
            expectedError:  "获取积分设置失败",
        },
    }
    
    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            router := setupTestRouter(tt.setupService())
            
            req := httptest.NewRequest(http.MethodGet, "/api/v1/system/config/points", nil)
            w := httptest.NewRecorder()
            router.ServeHTTP(w, req)
            
            assert.Equal(t, tt.expectedStatus, w.Code)
            
            var response map[string]interface{}
            err := json.Unmarshal(w.Body.Bytes(), &response)
            require.NoError(t, err)
            
            if tt.expectedError != "" {
                errors, ok := response["errors"].([]interface{})
                assert.True(t, ok)
                assert.NotEmpty(t, errors)
            } else {
                data, ok := response["data"].(map[string]interface{})
                assert.True(t, ok)
                assert.NotNil(t, data["points_name"])
            }
        })
    }
}
```

```go
func TestConfigHandler_UpdatePointsConfig(t *testing.T) {
    tests := []struct {
        name           string
        setupService   func() service.ConfigService
        requestBody    interface{}
        expectedStatus int
        expectedError  string
    }{
        {
            name: "正常更新积分设置",
            setupService: func() service.ConfigService {
                mockService := &MockConfigService{}
                mockService.On("UpdateConfig", mock.Anything, "points", mock.Anything).Return(nil)
                mockService.On("GetConfig", mock.Anything, "points").Return(map[string]interface{}{
                    "points_name":           "更新后的积分",
                    "register_points":       20,
                    "login_points":          10,
                    "comment_points":        5,
                    "recharge_points_ratio": 1.5,
                    "consume_points_ratio":  0.8,
                }, nil)
                return mockService
            },
            requestBody: dto.UpdatePointsConfigRequest{
                PointsName:          "更新后的积分",
                RegisterPoints:      20,
                LoginPoints:         10,
                CommentPoints:       5,
                RechargePointsRatio: 1.5,
                ConsumePointsRatio:  0.8,
            },
            expectedStatus: http.StatusOK,
        },
        {
            name: "请求参数验证失败-积分名称为空",
            setupService: func() service.ConfigService {
                return &MockConfigService{}
            },
            requestBody: dto.UpdatePointsConfigRequest{
                PointsName:          "",
                RegisterPoints:      10,
                LoginPoints:         5,
                CommentPoints:       2,
                RechargePointsRatio: 1.0,
                ConsumePointsRatio:  1.0,
            },
            expectedStatus: http.StatusBadRequest,
            expectedError:  "请求参数错误",
        },
        {
            name: "请求参数验证失败-注册积分为负数",
            setupService: func() service.ConfigService {
                return &MockConfigService{}
            },
            requestBody: dto.UpdatePointsConfigRequest{
                PointsName:          "积分",
                RegisterPoints:      -10,
                LoginPoints:         5,
                CommentPoints:       2,
                RechargePointsRatio: 1.0,
                ConsumePointsRatio:  1.0,
            },
            expectedStatus: http.StatusBadRequest,
            expectedError:  "请求参数错误",
        },
        {
            name: "请求参数验证失败-充值积分比例小于等于0",
            setupService: func() service.ConfigService {
                return &MockConfigService{}
            },
            requestBody: dto.UpdatePointsConfigRequest{
                PointsName:          "积分",
                RegisterPoints:      10,
                LoginPoints:         5,
                CommentPoints:       2,
                RechargePointsRatio: 0,
                ConsumePointsRatio:  1.0,
            },
            expectedStatus: http.StatusBadRequest,
            expectedError:  "请求参数错误",
        },
    }
    
    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            router := setupTestRouter(tt.setupService())
            
            requestBodyBytes, err := json.Marshal(tt.requestBody)
            require.NoError(t, err)
            
            req := httptest.NewRequest(http.MethodPut, "/api/v1/system/config/points", bytes.NewBuffer(requestBodyBytes))
            req.Header.Set("Content-Type", "application/json")
            w := httptest.NewRecorder()
            router.ServeHTTP(w, req)
            
            assert.Equal(t, tt.expectedStatus, w.Code)
            
            var response map[string]interface{}
            err = json.Unmarshal(w.Body.Bytes(), &response)
            require.NoError(t, err)
            
            if tt.expectedError != "" {
                errors, ok := response["errors"].([]interface{})
                assert.True(t, ok)
                assert.NotEmpty(t, errors)
                errorMsg := errors[0].(map[string]interface{})["message"].(string)
                assert.Contains(t, errorMsg, tt.expectedError)
            } else {
                data, ok := response["data"].(map[string]interface{})
                assert.True(t, ok)
                assert.NotNil(t, data["points_name"])
            }
        })
    }
}
```
