```go
package dto

// AudioTranscodeParams 音频转码参数
type AudioTranscodeParams struct {
    Encoder    string `json:"encoder"`     // 编码器
    SampleRate int    `json:"sample_rate"` // 采样率
    Channels   int    `json:"channels"`    // 声道数
    Bitrate    int    `json:"bitrate"`     // 比特率
}

// AudioAttachmentConfigResponse 音频附件设置响应
type AudioAttachmentConfigResponse struct {
    Enabled          bool               `json:"enabled"`            // 附件开关
    StorageChannelID int                `json:"storage_channel_id"` // 存储对象（渠道ID）
    AllowedTypes     []string           `json:"allowed_types"`     // 上传类型
    MaxSize          int64              `json:"max_size"`          // 上传大小，单位字节
    StoragePath      string             `json:"storage_path"`      // 存储路径
    ConvertFormat    string             `json:"convert_format"`    // 转换格式
    TranscodeParams  AudioTranscodeParams `json:"transcode_params"` // 转码参数
    WatermarkEnabled bool               `json:"watermark_enabled"` // 水印开关
    WatermarkFile    string             `json:"watermark_file"`    // 水印音频文件
    WatermarkPosition string            `json:"watermark_position"` // 水印位置
}
```

### 2. 实现GetAudioAttachmentConfig方法

在ConfigHandler中实现GetAudioAttachmentConfig方法：

```go
// GetAudioAttachmentConfig 获取音频附件设置
// @Summary 获取音频附件设置
// @Description 获取音频附件配置信息
// @Tags 系统配置
// @Accept json
// @Produce json
// @Success 200 {object} response.Envelope{data=dto.AudioAttachmentConfigResponse} "成功响应"
// @Failure 500 {object} response.Envelope "服务器错误"
// @Router /api/v1/system/config/attachment/audio [get]
func (h *ConfigHandler) GetAudioAttachmentConfig(c *gin.Context) {
    ctx := c.Request.Context()
    
    // 从服务层获取配置
    configData, err := h.configService.GetConfig(ctx, string(model.ConfigTypeAttachment))
    if err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_GET_FAILED", "获取附件设置失败", err)
        return
    }
    
    // 解析配置数据，获取audio部分
    configMap, ok := configData.(map[string]interface{})
    if !ok {
        response.Error(c, http.StatusInternalServerError, "CONFIG_PARSE_FAILED", "解析附件设置失败", nil)
        return
    }
    
    audioConfig, ok := configMap["audio"].(map[string]interface{})
    if !ok {
        response.Error(c, http.StatusInternalServerError, "CONFIG_PARSE_FAILED", "解析音频附件设置失败", nil)
        return
    }
    
    // 将配置数据转换为响应结构体
    var audioAttachmentConfig dto.AudioAttachmentConfigResponse
    if err := mapToAudioAttachmentConfig(audioConfig, &audioAttachmentConfig); err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_PARSE_FAILED", "解析音频附件设置失败", err)
        return
    }
    
    // 返回成功响应
    response.Success(c, audioAttachmentConfig)
}

// mapToAudioAttachmentConfig 将配置数据映射到AudioAttachmentConfigResponse
func mapToAudioAttachmentConfig(data map[string]interface{}, config *dto.AudioAttachmentConfigResponse) error {
    // 映射基本字段
    if v, ok := data["enabled"].(bool); ok {
        config.Enabled = v
    }
    if v, ok := data["storage_channel_id"].(float64); ok {
        config.StorageChannelID = int(v)
    }
    if v, ok := data["allowed_types"].([]interface{}); ok {
        config.AllowedTypes = make([]string, 0, len(v))
        for _, item := range v {
            if str, ok := item.(string); ok {
                config.AllowedTypes = append(config.AllowedTypes, str)
            }
        }
    }
    if v, ok := data["max_size"].(float64); ok {
        config.MaxSize = int64(v)
    }
    if v, ok := data["storage_path"].(string); ok {
        config.StoragePath = v
    }
    if v, ok := data["convert_format"].(string); ok {
        config.ConvertFormat = v
    }
    if v, ok := data["watermark_enabled"].(bool); ok {
        config.WatermarkEnabled = v
    }
    if v, ok := data["watermark_file"].(string); ok {
        config.WatermarkFile = v
    }
    if v, ok := data["watermark_position"].(string); ok {
        config.WatermarkPosition = v
    }
    
    // 映射转码参数
    if transcodeParams, ok := data["transcode_params"].(map[string]interface{}); ok {
        if v, ok := transcodeParams["encoder"].(string); ok {
            config.TranscodeParams.Encoder = v
        }
        if v, ok := transcodeParams["sample_rate"].(float64); ok {
            config.TranscodeParams.SampleRate = int(v)
        }
        if v, ok := transcodeParams["channels"].(float64); ok {
            config.TranscodeParams.Channels = int(v)
        }
        if v, ok := transcodeParams["bitrate"].(float64); ok {
            config.TranscodeParams.Bitrate = int(v)
        }
    }
    
    return nil
}
```
