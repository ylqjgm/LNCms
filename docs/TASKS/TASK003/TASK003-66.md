### 1. 实现EnableModule方法

在 `internal/api/v1/handler/module_handler.go` 文件中实现EnableModule方法：

```go
// EnableModule 启用模块
// @Summary 启用模块
// @Description 启用指定的模块
// @Tags 模块管理
// @Accept json
// @Produce json
// @Param module_id path string true "模块ID"
// @Success 200 {object} response.Envelope{data=dto.ModuleDetailResponse} "成功响应"
// @Failure 404 {object} response.Envelope "模块不存在"
// @Failure 403 {object} response.Envelope "权限不足"
// @Failure 500 {object} response.Envelope "服务器错误"
// @Router /api/v1/modules/{module_id}/enable [put]
// @Security BearerAuth
func (h *ModuleHandler) EnableModule(c *gin.Context) {
    ctx := c.Request.Context()
    moduleID := c.Param("module_id")
    
    if moduleID == "" {
        response.Error(c, http.StatusBadRequest, "INVALID_PARAM", "模块ID不能为空", nil)
        return
    }
    
    // 权限验证（需要管理员权限）
    // TODO: 实现权限验证中间件
    
    // 从服务层启用模块
    err := h.moduleService.EnableModule(ctx, moduleID)
    if err != nil {
        if errors.Is(err, repository.ErrNotFound) {
            response.Error(c, http.StatusNotFound, "MODULE_NOT_FOUND", "模块不存在", err)
            return
        }
        response.Error(c, http.StatusInternalServerError, "MODULE_ENABLE_FAILED", "启用模块失败", err)
        return
    }
    
    // 获取更新后的模块详情
    module, err := h.moduleService.GetModuleByID(ctx, moduleID)
    if err != nil {
        response.Error(c, http.StatusInternalServerError, "MODULE_GET_FAILED", "获取模块详情失败", err)
        return
    }
    
    // 解析配置数据
    var configData map[string]interface{}
    if len(module.ConfigData) > 0 {
        if err := json.Unmarshal(module.ConfigData, &configData); err != nil {
            configData = make(map[string]interface{})
        }
    }
    
    // 转换为响应结构体
    detail := dto.ModuleDetailResponse{
        ID:              module.ID,
        ModuleID:        module.ModuleID,
        ModuleName:      module.ModuleName,
        ModuleType:      module.ModuleType,
        ModuleURL:       module.ModuleURL,
        ModuleStatus:    module.ModuleStatus,
        HealthCheckURL:  module.HealthCheckURL,
        ConfigData:      configData,
        LastHealthCheck: module.LastHealthCheck.Format("2006-01-02 15:04:05"),
        CreatedAt:       module.CreatedAt.Format("2006-01-02 15:04:05"),
        UpdatedAt:       module.UpdatedAt.Format("2006-01-02 15:04:05"),
    }
    
    // 返回成功响应
    response.Success(c, detail)
}
```

### 2. 注册路由

在 `internal/api/v1/router/router.go` 文件中注册路由：

```go
// RegisterModuleRoutes 注册模块相关路由
func RegisterModuleRoutes(r *gin.RouterGroup, moduleHandler *handler.ModuleHandler) {
    modules := r.Group("/modules")
    {
        modules.GET("", moduleHandler.GetModuleList)
        modules.GET("/:module_id", moduleHandler.GetModuleDetail)
        modules.POST("", moduleHandler.RegisterModule)
        modules.PUT("/:module_id/enable", moduleHandler.EnableModule) // 需要管理员权限
    }
}
```
