### 2. 实现GetPointsConfig方法

在ConfigHandler中实现GetPointsConfig方法：

```go
// GetPointsConfig 获取积分设置
// @Summary 获取积分设置
// @Description 获取网站通用积分配置
// @Tags 系统配置
// @Accept json
// @Produce json
// @Success 200 {object} response.Envelope{data=dto.PointsConfigResponse} "成功响应"
// @Failure 500 {object} response.Envelope "服务器错误"
// @Router /api/v1/system/config/points [get]
func (h *ConfigHandler) GetPointsConfig(c *gin.Context) {
    ctx := c.Request.Context()
    
    // 从服务层获取配置
    configData, err := h.configService.GetConfig(ctx, string(model.ConfigTypePoints))
    if err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_GET_FAILED", "获取积分设置失败", err)
        return
    }
    
    // 将配置数据转换为响应结构体
    var pointsConfig dto.PointsConfigResponse
    if err := mapToPointsConfig(configData, &pointsConfig); err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_PARSE_FAILED", "解析积分设置失败", err)
        return
    }
    
    // 返回成功响应
    response.Success(c, pointsConfig)
}

// mapToPointsConfig 将配置数据映射到PointsConfigResponse
func mapToPointsConfig(data interface{}, pointsConfig *dto.PointsConfigResponse) error {
    configMap, ok := data.(map[string]interface{})
    if !ok {
        return fmt.Errorf("配置数据格式错误")
    }
    
    // 映射字段
    if v, ok := configMap["points_name"].(string); ok {
        pointsConfig.PointsName = v
    }
    if v, ok := configMap["register_points"].(float64); ok {
        pointsConfig.RegisterPoints = int(v)
    }
    if v, ok := configMap["login_points"].(float64); ok {
        pointsConfig.LoginPoints = int(v)
    }
    if v, ok := configMap["comment_points"].(float64); ok {
        pointsConfig.CommentPoints = int(v)
    }
    if v, ok := configMap["recharge_points_ratio"].(float64); ok {
        pointsConfig.RechargePointsRatio = v
    }
    if v, ok := configMap["consume_points_ratio"].(float64); ok {
        pointsConfig.ConsumePointsRatio = v
    }
    
    return nil
}
```

### 3. 注册路由

// RegisterConfigRoutes 注册配置相关路由
func RegisterConfigRoutes(r *gin.RouterGroup, configHandler *handler.ConfigHandler) {
    config := r.Group("/system/config")
    {
        config.GET("/site", configHandler.GetSiteConfig)
        config.PUT("/site", configHandler.UpdateSiteConfig)
        config.GET("/register", configHandler.GetRegisterConfig)
        config.PUT("/register", configHandler.UpdateRegisterConfig)
        config.GET("/cluster", configHandler.GetClusterConfig)
        config.PUT("/cluster", configHandler.UpdateClusterConfig)
        clusterNodes := config.Group("/cluster/nodes")
        {
            clusterNodes.POST("", configHandler.AddClusterNode)
            clusterNodes.PUT("/:node_id", configHandler.UpdateClusterNode)
            clusterNodes.DELETE("/:node_id", configHandler.DeleteClusterNode)
        }
        config.GET("/points", configHandler.GetPointsConfig) // 添加积分设置GET路由
    }
}
