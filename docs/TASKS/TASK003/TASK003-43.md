```go
func TestConfigHandler_GetEmailConfig(t *testing.T) {
    tests := []struct {
        name           string
        setupService   func() service.ConfigService
        expectedStatus int
        expectedError  string
    }{
        {
            name: "正常获取邮箱设置",
            setupService: func() service.ConfigService {
                mockService := &MockConfigService{}
                mockService.On("GetConfig", mock.Anything, "email").Return(map[string]interface{}{
                    "smtp_host":   "smtp.example.com",
                    "smtp_port":   587,
                    "tls_enabled": true,
                    "username":    "user@example.com",
                    "password":    "encrypted_password",
                    "from_email":  "noreply@example.com",
                }, nil)
                return mockService
            },
            expectedStatus: http.StatusOK,
        },
        {
            name: "配置不存在",
            setupService: func() service.ConfigService {
                mockService := &MockConfigService{}
                mockService.On("GetConfig", mock.Anything, "email").Return(nil, fmt.Errorf("配置类型 email 不存在"))
                return mockService
            },
            expectedStatus: http.StatusInternalServerError,
            expectedError:  "获取邮箱设置失败",
        },
    }
    
    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            router := setupTestRouter(tt.setupService())
            
            req := httptest.NewRequest(http.MethodGet, "/api/v1/system/config/email", nil)
            w := httptest.NewRecorder()
            router.ServeHTTP(w, req)
            
            assert.Equal(t, tt.expectedStatus, w.Code)
            
            var response map[string]interface{}
            err := json.Unmarshal(w.Body.Bytes(), &response)
            require.NoError(t, err)
            
            if tt.expectedError != "" {
                errors, ok := response["errors"].([]interface{})
                assert.True(t, ok)
                assert.NotEmpty(t, errors)
            } else {
                data, ok := response["data"].(map[string]interface{})
                assert.True(t, ok)
                assert.NotNil(t, data["smtp_host"])
                // 验证密码字段不返回明文
                password, ok := data["password"].(string)
                assert.True(t, ok)
                assert.NotEqual(t, "encrypted_password", password, "密码字段不应返回明文")
            }
        })
    }
}
```

```go
func TestConfigHandler_UpdateEmailConfig(t *testing.T) {
    tests := []struct {
        name           string
        setupService   func() service.ConfigService
        requestBody    interface{}
        expectedStatus int
        expectedError  string
    }{
        {
            name: "正常更新邮箱设置",
            setupService: func() service.ConfigService {
                mockService := &MockConfigService{}
                mockService.On("GetConfig", mock.Anything, "email").Return(map[string]interface{}{}, nil).Once()
                mockService.On("UpdateConfig", mock.Anything, "email", mock.Anything).Return(nil)
                mockService.On("GetConfig", mock.Anything, "email").Return(map[string]interface{}{
                    "smtp_host":   "smtp.updated.com",
                    "smtp_port":   465,
                    "tls_enabled": true,
                    "username":    "updated@example.com",
                    "password":    "encrypted_new_password",
                    "from_email":  "noreply@updated.com",
                }, nil).Once()
                return mockService
            },
            requestBody: dto.UpdateEmailConfigRequest{
                SMTPHost:   "smtp.updated.com",
                SMTPPort:   465,
                TLSEnabled: true,
                Username:   "updated@example.com",
                Password:   "new_password",
                FromEmail:  "noreply@updated.com",
            },
            expectedStatus: http.StatusOK,
        },
        {
            name: "正常更新邮箱设置-密码为空不更新密码",
            setupService: func() service.ConfigService {
                mockService := &MockConfigService{}
                mockService.On("GetConfig", mock.Anything, "email").Return(map[string]interface{}{
                    "password": "existing_encrypted_password",
                }, nil).Once()
                mockService.On("UpdateConfig", mock.Anything, "email", mock.Anything).Return(nil)
                mockService.On("GetConfig", mock.Anything, "email").Return(map[string]interface{}{
                    "smtp_host":   "smtp.example.com",
                    "smtp_port":   587,
                    "tls_enabled": true,
                    "username":    "user@example.com",
                    "password":    "existing_encrypted_password",
                    "from_email":  "noreply@example.com",
                }, nil).Once()
                return mockService
            },
            requestBody: dto.UpdateEmailConfigRequest{
                SMTPHost:   "smtp.example.com",
                SMTPPort:   587,
                TLSEnabled: true,
                Username:   "user@example.com",
                Password:   "", // 空密码，应该保持原密码
                FromEmail:  "noreply@example.com",
            },
            expectedStatus: http.StatusOK,
        },
        {
            name: "请求参数验证失败-SMTP端口超出范围",
            setupService: func() service.ConfigService {
                return &MockConfigService{}
            },
            requestBody: dto.UpdateEmailConfigRequest{
                SMTPHost:   "smtp.example.com",
                SMTPPort:   70000, // 超出范围
                TLSEnabled: true,
                Username:   "user@example.com",
                FromEmail:  "noreply@example.com",
            },
            expectedStatus: http.StatusBadRequest,
            expectedError:  "请求参数错误",
        },
        {
            name: "请求参数验证失败-发件人邮箱格式错误",
            setupService: func() service.ConfigService {
                return &MockConfigService{}
            },
            requestBody: dto.UpdateEmailConfigRequest{
                SMTPHost:   "smtp.example.com",
                SMTPPort:   587,
                TLSEnabled: true,
                Username:   "user@example.com",
                FromEmail:  "invalid-email", // 邮箱格式错误
            },
            expectedStatus: http.StatusBadRequest,
            expectedError:  "请求参数错误",
        },
    }
    
    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            router := setupTestRouter(tt.setupService())
            
            requestBodyBytes, err := json.Marshal(tt.requestBody)
            require.NoError(t, err)
            
            req := httptest.NewRequest(http.MethodPut, "/api/v1/system/config/email", bytes.NewBuffer(requestBodyBytes))
            req.Header.Set("Content-Type", "application/json")
            w := httptest.NewRecorder()
            router.ServeHTTP(w, req)
            
            assert.Equal(t, tt.expectedStatus, w.Code)
            
            var response map[string]interface{}
            err = json.Unmarshal(w.Body.Bytes(), &response)
            require.NoError(t, err)
            
            if tt.expectedError != "" {
                errors, ok := response["errors"].([]interface{})
                assert.True(t, ok)
                assert.NotEmpty(t, errors)
                errorMsg := errors[0].(map[string]interface{})["message"].(string)
                assert.Contains(t, errorMsg, tt.expectedError)
            } else {
                data, ok := response["data"].(map[string]interface{})
                assert.True(t, ok)
                assert.NotNil(t, data["smtp_host"])
                // 验证密码字段不返回明文
                password, ok := data["password"].(string)
                assert.True(t, ok)
                assert.NotEqual(t, "encrypted_password", password, "密码字段不应返回明文")
            }
        })
    }
}
```
