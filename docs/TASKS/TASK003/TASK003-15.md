type UpdateRegisterConfigRequest struct {

### 2. 实现UpdateRegisterConfig方法

在ConfigHandler中实现UpdateRegisterConfig方法：

```go
// UpdateRegisterConfig 更新注册设置
// @Summary 更新注册设置
// @Description 更新用户注册信息配置，需要管理员权限
// @Tags 系统配置
// @Accept json
// @Produce json
// @Param request body dto.UpdateRegisterConfigRequest true "更新注册设置请求"
// @Success 200 {object} response.Envelope{data=dto.RegisterConfigResponse} "成功响应"
// @Failure 400 {object} response.Envelope "请求参数错误"
// @Failure 401 {object} response.Envelope "未授权"
// @Failure 403 {object} response.Envelope "权限不足"
// @Failure 500 {object} response.Envelope "服务器错误"
// @Router /api/v1/system/config/register [put]
// @Security BearerAuth
func (h *ConfigHandler) UpdateRegisterConfig(c *gin.Context) {
    ctx := c.Request.Context()
    
    // 权限验证（需要管理员权限）
    // 注意：权限验证逻辑应该在中间件中实现
    
    // 绑定请求参数
    var req dto.UpdateRegisterConfigRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        response.Error(c, http.StatusBadRequest, "INVALID_REQUEST", "请求参数错误", err)
        return
    }
    
    // 业务验证：密码最小长度必须小于等于最大长度
    if req.PasswordRequirements.MinLength > req.PasswordRequirements.MaxLength {
        response.Error(c, http.StatusBadRequest, "VALIDATION_ERROR", "密码最小长度必须小于等于最大长度", nil)
        return
    }
    
    // 业务验证：密码长度范围
    if req.PasswordRequirements.MinLength < 6 || req.PasswordRequirements.MinLength > 128 {
        response.Error(c, http.StatusBadRequest, "VALIDATION_ERROR", "密码最小长度必须在6-128之间", nil)
        return
    }
    if req.PasswordRequirements.MaxLength < 8 || req.PasswordRequirements.MaxLength > 128 {
        response.Error(c, http.StatusBadRequest, "VALIDATION_ERROR", "密码最大长度必须在8-128之间", nil)
        return
    }
    
    // 构建配置数据
    configData := map[string]interface{}{
        "register_enabled":      req.RegisterEnabled,
        "captcha_enabled":       req.CaptchaEnabled,
        "password_requirements": map[string]interface{}{
            "min_length":        req.PasswordRequirements.MinLength,
            "max_length":        req.PasswordRequirements.MaxLength,
            "require_uppercase": req.PasswordRequirements.RequireUppercase,
            "require_lowercase": req.PasswordRequirements.RequireLowercase,
            "require_number":    req.PasswordRequirements.RequireNumber,
            "require_special":   req.PasswordRequirements.RequireSpecial,
        },
        "duplicate_register": req.DuplicateRegister,
    }
    
    // 调用服务层更新配置
    if err := h.configService.UpdateConfig(ctx, string(model.ConfigTypeRegister), configData); err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_UPDATE_FAILED", "更新注册设置失败", err)
        return
    }
    
    // 获取更新后的配置并返回
    updatedData, err := h.configService.GetConfig(ctx, string(model.ConfigTypeRegister))
    if err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_GET_FAILED", "获取更新后的注册设置失败", err)
        return
    }
    
    var registerConfig dto.RegisterConfigResponse
    if err := mapToRegisterConfig(updatedData, &registerConfig); err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_PARSE_FAILED", "解析注册设置失败", err)
        return
    }
    
    // 返回成功响应
    response.Success(c, registerConfig)
}
```

func RegisterConfigRoutes(r *gin.RouterGroup, configHandler *handler.ConfigHandler) {
    config := r.Group("/system/config")
    {
        config.GET("/site", configHandler.GetSiteConfig)
        config.PUT("/site", configHandler.UpdateSiteConfig)
