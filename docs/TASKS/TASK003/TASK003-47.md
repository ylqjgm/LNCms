### 2. 实现UpdateShieldConfig方法

在ConfigHandler中实现UpdateShieldConfig方法：

```go
// UpdateShieldConfig 更新屏蔽设置
// @Summary 更新屏蔽设置
// @Description 更新网站信息屏蔽配置，需要管理员权限
// @Tags 系统配置
// @Accept json
// @Produce json
// @Param request body dto.UpdateShieldConfigRequest true "更新屏蔽设置请求"
// @Success 200 {object} response.Envelope{data=dto.ShieldConfigResponse} "成功响应"
// @Failure 400 {object} response.Envelope "请求参数错误"
// @Failure 401 {object} response.Envelope "未授权"
// @Failure 403 {object} response.Envelope "权限不足"
// @Failure 500 {object} response.Envelope "服务器错误"
// @Router /api/v1/system/config/shield [put]
// @Security BearerAuth
func (h *ConfigHandler) UpdateShieldConfig(c *gin.Context) {
    ctx := c.Request.Context()
    
    // 权限验证（需要管理员权限）
    // 注意：权限验证逻辑应该在中间件中实现
    
    // 绑定请求参数
    var req dto.UpdateShieldConfigRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        response.Error(c, http.StatusBadRequest, "INVALID_REQUEST", "请求参数错误", err)
        return
    }
    
    // 业务验证：验证IP地址格式（可选）
    for _, ip := range req.IPBlacklist {
        if ip != "" {
            // 简单的IP地址格式验证（可以使用net包进行更严格的验证）
            if !isValidIP(ip) {
                response.Error(c, http.StatusBadRequest, "VALIDATION_ERROR", fmt.Sprintf("IP地址格式无效: %s", ip), nil)
                return
            }
        }
    }
    
    // 构建配置数据
    configData := map[string]interface{}{
        "ip_blacklist":     req.IPBlacklist,
        "user_id_blacklist": req.UserIDBlacklist,
        "forbidden_words":  req.ForbiddenWords,
    }
    
    // 调用服务层更新配置
    if err := h.configService.UpdateConfig(ctx, string(model.ConfigTypeShield), configData); err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_UPDATE_FAILED", "更新屏蔽设置失败", err)
        return
    }
    
    // 获取更新后的配置并返回
    updatedData, err := h.configService.GetConfig(ctx, string(model.ConfigTypeShield))
    if err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_GET_FAILED", "获取更新后的屏蔽设置失败", err)
        return
    }
    
    var shieldConfig dto.ShieldConfigResponse
    if err := mapToShieldConfig(updatedData, &shieldConfig); err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_PARSE_FAILED", "解析屏蔽设置失败", err)
        return
    }
    
    // 返回成功响应
    response.Success(c, shieldConfig)
}

// isValidIP 验证IP地址格式
func isValidIP(ip string) bool {
    // 使用net包进行IP地址验证
    // 支持IPv4和IPv6
    parsedIP := net.ParseIP(ip)
    return parsedIP != nil
}
```
