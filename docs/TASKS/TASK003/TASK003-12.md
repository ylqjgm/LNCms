package dto

// UpdateSiteConfigRequest 更新网站设置请求
type UpdateSiteConfigRequest struct {
    SiteName          string `json:"site_name" binding:"required,min=1,max=100"`           // 网站名称，必填，1-100字符
    SiteURL           string `json:"site_url" binding:"required,url"`                      // 网站地址，必填，URL格式
    SiteKeywords      string `json:"site_keywords" binding:"max=500"`                      // 网站关键字，可选，最大500字符
    SiteDescription   string `json:"site_description" binding:"max=1000"`                  // 网站描述，可选，最大1000字符
    SiteCopyright     string `json:"site_copyright" binding:"max=500"`                     // 版权申明，可选，最大500字符
    SiteEnabled       bool   `json:"site_enabled"`                                         // 网站开关，必填
    SiteClosedMessage string `json:"site_closed_message" binding:"required_if=SiteEnabled false,max=1000"` // 关闭提示，网站关闭时必填，最大1000字符
}

### 2. 实现UpdateSiteConfig方法

在ConfigHandler中实现UpdateSiteConfig方法：

```go
// UpdateSiteConfig 更新网站设置
// @Summary 更新网站设置
// @Description 更新网站基础信息配置，需要管理员权限
// @Tags 系统配置
// @Accept json
// @Produce json
// @Param request body dto.UpdateSiteConfigRequest true "更新网站设置请求"
// @Success 200 {object} response.Envelope{data=dto.SiteConfigResponse} "成功响应"
// @Failure 400 {object} response.Envelope "请求参数错误"
// @Failure 401 {object} response.Envelope "未授权"
// @Failure 403 {object} response.Envelope "权限不足"
// @Failure 500 {object} response.Envelope "服务器错误"
// @Router /api/v1/system/config/site [put]
// @Security BearerAuth
func (h *ConfigHandler) UpdateSiteConfig(c *gin.Context) {
    ctx := c.Request.Context()
    
    // 权限验证（需要管理员权限）
    // 注意：权限验证逻辑应该在中间件中实现，这里仅作示例说明
    // if !hasAdminPermission(c) {
    //     response.Error(c, http.StatusForbidden, "PERMISSION_DENIED", "权限不足：需要管理员权限", nil)
    //     return
    // }
    
    // 绑定请求参数
    var req dto.UpdateSiteConfigRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        response.Error(c, http.StatusBadRequest, "INVALID_REQUEST", "请求参数错误", err)
        return
    }
    
    // 业务验证：网站关闭时，关闭提示不能为空
    if !req.SiteEnabled && req.SiteClosedMessage == "" {
        response.Error(c, http.StatusBadRequest, "VALIDATION_ERROR", "网站关闭时，关闭提示不能为空", nil)
        return
    }
    
    // 构建配置数据
    configData := map[string]interface{}{
        "site_name":           req.SiteName,
        "site_url":            req.SiteURL,
        "site_keywords":       req.SiteKeywords,
        "site_description":    req.SiteDescription,
        "site_copyright":      req.SiteCopyright,
        "site_enabled":        req.SiteEnabled,
        "site_closed_message": req.SiteClosedMessage,
    }
    
    // 调用服务层更新配置
    if err := h.configService.UpdateConfig(ctx, string(model.ConfigTypeSite), configData); err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_UPDATE_FAILED", "更新网站设置失败", err)
        return
    }
    
    // 获取更新后的配置并返回
    updatedData, err := h.configService.GetConfig(ctx, string(model.ConfigTypeSite))
    if err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_GET_FAILED", "获取更新后的网站设置失败", err)
        return
    }
    
    var siteConfig dto.SiteConfigResponse
    if err := mapToSiteConfig(updatedData, &siteConfig); err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_PARSE_FAILED", "解析网站设置失败", err)
        return
    }
    
    // 返回成功响应
    response.Success(c, siteConfig)
}
```
