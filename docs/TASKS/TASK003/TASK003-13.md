```go
)

// 设置测试环境
func setupTestRouter(configService service.ConfigService) *gin.Engine {
    gin.SetMode(gin.TestMode)
    router := gin.New()
    
    configHandler := NewConfigHandler(configService)
    api := router.Group("/api/v1")
    config := api.Group("/system/config")
    {
        config.GET("/site", configHandler.GetSiteConfig)
        config.PUT("/site", configHandler.UpdateSiteConfig)
    }
    
    return router
}
```

```go
func TestConfigHandler_GetSiteConfig(t *testing.T) {
    tests := []struct {
        name           string
        setupService   func() service.ConfigService
        expectedStatus int
        expectedError  string
    }{
        {
            name: "正常获取网站设置",
            setupService: func() service.ConfigService {
                mockService := &MockConfigService{}
                mockService.On("GetConfig", mock.Anything, "site").Return(map[string]interface{}{
                    "site_name":           "测试网站",
                    "site_url":            "https://example.com",
                    "site_keywords":       "测试,网站",
                    "site_description":    "这是一个测试网站",
                    "site_copyright":      "© 2026 测试网站",
                    "site_enabled":        true,
                    "site_closed_message": "",
                }, nil)
                return mockService
            },
            expectedStatus: http.StatusOK,
        },
        {
            name: "配置不存在",
            setupService: func() service.ConfigService {
                mockService := &MockConfigService{}
                mockService.On("GetConfig", mock.Anything, "site").Return(nil, fmt.Errorf("配置类型 site 不存在"))
                return mockService
            },
            expectedStatus: http.StatusInternalServerError,
            expectedError:  "获取网站设置失败",
        },
    }
    
    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            router := setupTestRouter(tt.setupService())
            
            req := httptest.NewRequest(http.MethodGet, "/api/v1/system/config/site", nil)
            w := httptest.NewRecorder()
            router.ServeHTTP(w, req)
            
            assert.Equal(t, tt.expectedStatus, w.Code)
            
            var response map[string]interface{}
            err := json.Unmarshal(w.Body.Bytes(), &response)
            require.NoError(t, err)
            
            if tt.expectedError != "" {
                errors, ok := response["errors"].([]interface{})
                assert.True(t, ok)
                assert.NotEmpty(t, errors)
            } else {
                data, ok := response["data"].(map[string]interface{})
                assert.True(t, ok)
                assert.NotNil(t, data["site_name"])
            }
        })
    }
}
```

```go
func TestConfigHandler_UpdateSiteConfig(t *testing.T) {
    tests := []struct {
        name           string
        setupService   func() service.ConfigService
        requestBody    interface{}
        expectedStatus int
        expectedError  string
    }{
        {
            name: "正常更新网站设置",
            setupService: func() service.ConfigService {
                mockService := &MockConfigService{}
                mockService.On("UpdateConfig", mock.Anything, "site", mock.Anything).Return(nil)
                mockService.On("GetConfig", mock.Anything, "site").Return(map[string]interface{}{
                    "site_name":           "更新后的网站",
                    "site_url":            "https://updated.com",
                    "site_keywords":       "更新,网站",
                    "site_description":    "更新后的网站描述",
                    "site_copyright":      "© 2026 更新后的网站",
                    "site_enabled":        true,
                    "site_closed_message": "",
                }, nil)
                return mockService
            },
            requestBody: dto.UpdateSiteConfigRequest{
                SiteName:          "更新后的网站",
                SiteURL:           "https://updated.com",
                SiteKeywords:      "更新,网站",
                SiteDescription:   "更新后的网站描述",
                SiteCopyright:     "© 2026 更新后的网站",
                SiteEnabled:       true,
                SiteClosedMessage: "",
            },
            expectedStatus: http.StatusOK,
        },
        {
            name: "请求参数验证失败-网站名称为空",
            setupService: func() service.ConfigService {
                return &MockConfigService{}
            },
            requestBody: dto.UpdateSiteConfigRequest{
                SiteName:          "",
                SiteURL:           "https://example.com",
                SiteEnabled:       true,
                SiteClosedMessage: "",
            },
            expectedStatus: http.StatusBadRequest,
            expectedError:  "请求参数错误",
        },
        {
            name: "业务验证失败-网站关闭时关闭提示为空",
            setupService: func() service.ConfigService {
                return &MockConfigService{}
            },
            requestBody: dto.UpdateSiteConfigRequest{
                SiteName:          "测试网站",
                SiteURL:           "https://example.com",
                SiteEnabled:       false,
                SiteClosedMessage: "",
            },
            expectedStatus: http.StatusBadRequest,
            expectedError:  "网站关闭时，关闭提示不能为空",
        },
        {
            name: "URL格式验证失败",
            setupService: func() service.ConfigService {
                return &MockConfigService{}
            },
            requestBody: dto.UpdateSiteConfigRequest{
                SiteName:          "测试网站",
                SiteURL:           "invalid-url",
                SiteEnabled:       true,
                SiteClosedMessage: "",
            },
            expectedStatus: http.StatusBadRequest,
            expectedError:  "请求参数错误",
        },
    }
    
    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            router := setupTestRouter(tt.setupService())
            
            requestBodyBytes, err := json.Marshal(tt.requestBody)
            require.NoError(t, err)
            
            req := httptest.NewRequest(http.MethodPut, "/api/v1/system/config/site", bytes.NewBuffer(requestBodyBytes))
            req.Header.Set("Content-Type", "application/json")
            w := httptest.NewRecorder()
            router.ServeHTTP(w, req)
            
            assert.Equal(t, tt.expectedStatus, w.Code)
            
            var response map[string]interface{}
            err = json.Unmarshal(w.Body.Bytes(), &response)
            require.NoError(t, err)
            
            if tt.expectedError != "" {
                errors, ok := response["errors"].([]interface{})
                assert.True(t, ok)
                assert.NotEmpty(t, errors)
                // 验证错误消息包含预期内容
                errorMsg := errors[0].(map[string]interface{})["message"].(string)
                assert.Contains(t, errorMsg, tt.expectedError)
            } else {
                data, ok := response["data"].(map[string]interface{})
                assert.True(t, ok)
                assert.NotNil(t, data["site_name"])
            }
        })
    }
}
```

### 4. 创建Mock ConfigService

创建Mock ConfigService用于测试：

```go
import (
    "github.com/stretchr/testify/mock"
    "context"
)

// MockConfigService Mock配置服务
type MockConfigService struct {
    mock.Mock
}

func (m *MockConfigService) GetConfig(ctx context.Context, configType string) (interface{}, error) {
    args := m.Called(ctx, configType)
    return args.Get(0), args.Error(1)
}

func (m *MockConfigService) UpdateConfig(ctx context.Context, configType string, data interface{}) error {
    args := m.Called(ctx, configType, data)
    return args.Error(0)
}
```
