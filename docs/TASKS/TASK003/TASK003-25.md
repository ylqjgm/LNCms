### 2. 实现UpdatePointsConfig方法

在ConfigHandler中实现UpdatePointsConfig方法：

```go
// @Summary 更新积分设置
// @Description 更新网站通用积分配置，需要管理员权限
// @Tags 系统配置
// @Accept json
// @Produce json
// @Param request body dto.UpdatePointsConfigRequest true "更新积分设置请求"
// @Success 200 {object} response.Envelope{data=dto.PointsConfigResponse} "成功响应"
// @Failure 400 {object} response.Envelope "请求参数错误"
// @Failure 401 {object} response.Envelope "未授权"
// @Failure 403 {object} response.Envelope "权限不足"
// @Failure 500 {object} response.Envelope "服务器错误"
// @Router /api/v1/system/config/points [put]
// @Security BearerAuth
func (h *ConfigHandler) UpdatePointsConfig(c *gin.Context) {
    ctx := c.Request.Context()
    
    // 权限验证（需要管理员权限）
    // 注意：权限验证逻辑应该在中间件中实现
    
    // 绑定请求参数
    var req dto.UpdatePointsConfigRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        response.Error(c, http.StatusBadRequest, "INVALID_REQUEST", "请求参数错误", err)
        return
    }
    
    // 构建配置数据
    configData := map[string]interface{}{
        "points_name":           req.PointsName,
        "register_points":       req.RegisterPoints,
        "login_points":          req.LoginPoints,
```

### 3. 注册路由

```go
// RegisterConfigRoutes 注册配置相关路由
func RegisterConfigRoutes(r *gin.RouterGroup, configHandler *handler.ConfigHandler) {
    config := r.Group("/system/config")
    {
        config.GET("/site", configHandler.GetSiteConfig)
        config.PUT("/site", configHandler.UpdateSiteConfig)
        config.GET("/register", configHandler.GetRegisterConfig)
        config.PUT("/register", configHandler.UpdateRegisterConfig)
        config.GET("/cluster", configHandler.GetClusterConfig)
        config.PUT("/cluster", configHandler.UpdateClusterConfig)
        clusterNodes := config.Group("/cluster/nodes")
        {
            clusterNodes.POST("", configHandler.AddClusterNode)
            clusterNodes.PUT("/:node_id", configHandler.UpdateClusterNode)
            clusterNodes.DELETE("/:node_id", configHandler.DeleteClusterNode)
        }
        config.GET("/points", configHandler.GetPointsConfig)
        config.PUT("/points", configHandler.UpdatePointsConfig) // 添加PUT路由
    }
}
```
