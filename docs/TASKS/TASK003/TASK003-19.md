```go
func TestConfigHandler_GetClusterConfig(t *testing.T) {
    tests := []struct {
        name           string
        setupService   func() service.ConfigService
        expectedStatus int
        expectedError  string
    }{
        {
            name: "正常获取站群设置",
            setupService: func() service.ConfigService {
                mockService := &MockConfigService{}
                mockService.On("GetConfig", mock.Anything, "cluster").Return(map[string]interface{}{
                    "master_site": map[string]interface{}{
                        "master_url": "https://master.example.com",
                        "master_id":  "master-001",
                    },
                    "cluster_nodes": []interface{}{
                        map[string]interface{}{
                            "node_id":     "node-001",
                            "node_url":    "https://node1.example.com",
                            "node_status": "active",
                            "node_weight": 50,
                        },
                    },
                }, nil)
                return mockService
            },
            expectedStatus: http.StatusOK,
        },
        {
            name: "配置不存在",
            setupService: func() service.ConfigService {
                mockService := &MockConfigService{}
                mockService.On("GetConfig", mock.Anything, "cluster").Return(nil, fmt.Errorf("配置类型 cluster 不存在"))
                return mockService
            },
            expectedStatus: http.StatusInternalServerError,
            expectedError:  "获取站群设置失败",
        },
    }
    
    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            router := setupTestRouter(tt.setupService())
            
            req := httptest.NewRequest(http.MethodGet, "/api/v1/system/config/cluster", nil)
            w := httptest.NewRecorder()
            router.ServeHTTP(w, req)
            
            assert.Equal(t, tt.expectedStatus, w.Code)
            
            var response map[string]interface{}
            err := json.Unmarshal(w.Body.Bytes(), &response)
            require.NoError(t, err)
            
            if tt.expectedError != "" {
                errors, ok := response["errors"].([]interface{})
                assert.True(t, ok)
                assert.NotEmpty(t, errors)
            } else {
                data, ok := response["data"].(map[string]interface{})
                assert.True(t, ok)
                assert.NotNil(t, data["master_site"])
            }
        })
    }
}
```

### 2. 测试UpdateClusterConfig方法

```go
func TestConfigHandler_UpdateClusterConfig(t *testing.T) {
    tests := []struct {
        name           string
        setupService   func() service.ConfigService
        requestBody    interface{}
        expectedStatus int
        expectedError  string
    }{
        {
            name: "正常更新站群设置",
            setupService: func() service.ConfigService {
                mockService := &MockConfigService{}
                mockService.On("UpdateConfig", mock.Anything, "cluster", mock.Anything).Return(nil)
                mockService.On("GetConfig", mock.Anything, "cluster").Return(map[string]interface{}{
                    "master_site": map[string]interface{}{
                        "master_url": "https://updated-master.example.com",
                        "master_id":  "master-002",
                    },
                    "cluster_nodes": []interface{}{},
                }, nil)
                return mockService
            },
            requestBody: dto.UpdateClusterConfigRequest{
                MasterSite: dto.MasterSite{
                    MasterURL: "https://updated-master.example.com",
                    MasterID:  "master-002",
                },
                ClusterNodes: []dto.ClusterNode{},
            },
            expectedStatus: http.StatusOK,
        },
        {
            name: "请求参数验证失败-主站地址格式无效",
            setupService: func() service.ConfigService {
                return &MockConfigService{}
            },
            requestBody: dto.UpdateClusterConfigRequest{
                MasterSite: dto.MasterSite{
                    MasterURL: "invalid-url",
                    MasterID:  "master-001",
                },
                ClusterNodes: []dto.ClusterNode{},
            },
            expectedStatus: http.StatusBadRequest,
            expectedError:  "主站地址格式无效",
        },
        {
            name: "业务验证失败-节点标识重复",
            setupService: func() service.ConfigService {
                return &MockConfigService{}
            },
            requestBody: dto.UpdateClusterConfigRequest{
                MasterSite: dto.MasterSite{
                    MasterURL: "https://master.example.com",
                    MasterID:  "master-001",
                },
                ClusterNodes: []dto.ClusterNode{
                    {NodeID: "node-001", NodeURL: "https://node1.example.com", NodeStatus: "active", NodeWeight: 50},
                    {NodeID: "node-001", NodeURL: "https://node2.example.com", NodeStatus: "active", NodeWeight: 50},
                },
            },
            expectedStatus: http.StatusBadRequest,
            expectedError:  "节点标识 node-001 重复",
        },
        {
            name: "业务验证失败-节点权重超出范围",
            setupService: func() service.ConfigService {
                return &MockConfigService{}
            },
            requestBody: dto.UpdateClusterConfigRequest{
                MasterSite: dto.MasterSite{
                    MasterURL: "https://master.example.com",
                    MasterID:  "master-001",
                },
                ClusterNodes: []dto.ClusterNode{
                    {NodeID: "node-001", NodeURL: "https://node1.example.com", NodeStatus: "active", NodeWeight: 150},
                },
            },
            expectedStatus: http.StatusBadRequest,
            expectedError:  "节点权重必须在1-100之间",
        },
    }
    
    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            router := setupTestRouter(tt.setupService())
            
            requestBodyBytes, err := json.Marshal(tt.requestBody)
            require.NoError(t, err)
            
            req := httptest.NewRequest(http.MethodPut, "/api/v1/system/config/cluster", bytes.NewBuffer(requestBodyBytes))
            req.Header.Set("Content-Type", "application/json")
            w := httptest.NewRecorder()
            router.ServeHTTP(w, req)
            
            assert.Equal(t, tt.expectedStatus, w.Code)
            
            var response map[string]interface{}
            err = json.Unmarshal(w.Body.Bytes(), &response)
            require.NoError(t, err)
            
            if tt.expectedError != "" {
                errors, ok := response["errors"].([]interface{})
                assert.True(t, ok)
                assert.NotEmpty(t, errors)
                errorMsg := errors[0].(map[string]interface{})["message"].(string)
                assert.Contains(t, errorMsg, tt.expectedError)
            } else {
                data, ok := response["data"].(map[string]interface{})
                assert.True(t, ok)
                assert.NotNil(t, data["master_site"])
            }
        })
    }
}
```
