### 2. 实现CreateStorageChannel方法

在ConfigHandler中实现CreateStorageChannel方法：

```go
// CreateStorageChannel 创建存储渠道
// @Summary 创建存储渠道
// @Description 创建附件存储渠道，需要管理员权限
// @Tags 系统配置
// @Accept json
// @Produce json
// @Param request body dto.CreateStorageChannelRequest true "创建存储渠道请求"
// @Success 200 {object} response.Envelope{data=dto.StorageChannelResponse} "成功响应"
// @Failure 400 {object} response.Envelope "请求参数错误"
// @Failure 401 {object} response.Envelope "未授权"
// @Failure 403 {object} response.Envelope "权限不足"
// @Failure 500 {object} response.Envelope "服务器错误"
// @Router /api/v1/system/config/storage/channels [post]
// @Security BearerAuth
func (h *ConfigHandler) CreateStorageChannel(c *gin.Context) {
    ctx := c.Request.Context()
    
    // 权限验证（需要管理员权限）
    // 注意：权限验证逻辑应该在中间件中实现
    
    // 绑定请求参数
    var req dto.CreateStorageChannelRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        response.Error(c, http.StatusBadRequest, "INVALID_REQUEST", "请求参数错误", err)
        return
    }
    
    // 获取当前附件配置
    configData, err := h.configService.GetConfig(ctx, string(model.ConfigTypeAttachment))
    if err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_GET_FAILED", "获取附件设置失败", err)
        return
    }
    
    // 解析当前配置
    configMap, ok := configData.(map[string]interface{})
    if !ok {
        response.Error(c, http.StatusInternalServerError, "CONFIG_PARSE_FAILED", "解析附件设置失败", nil)
        return
    }
    
    // 获取存储渠道列表
    channels, ok := configMap["storage_channels"].([]interface{})
    if !ok {
        channels = []interface{}{}
    }
    
    // 检查渠道名称是否已存在
    for _, channelInterface := range channels {
        channelMap, ok := channelInterface.(map[string]interface{})
        if !ok {
            continue
        }
        if existingChannelName, ok := channelMap["channel_name"].(string); ok && existingChannelName == req.ChannelName {
            response.Error(c, http.StatusBadRequest, "VALIDATION_ERROR", fmt.Sprintf("渠道名称 %s 已存在", req.ChannelName), nil)
            return
        }
    }
    
    // 创建新渠道
    newChannel := map[string]interface{}{
        "channel_name":   req.ChannelName,
        "storage_type":   req.StorageType,
        "storage_config": req.StorageConfig,
    }
    channels = append(channels, newChannel)
    
    // 更新配置
    configMap["storage_channels"] = channels
    if err := h.configService.UpdateConfig(ctx, string(model.ConfigTypeAttachment), configMap); err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_UPDATE_FAILED", "创建存储渠道失败", err)
        return
    }
    
    // 构建响应
    channelResponse := dto.StorageChannelResponse{
        ChannelID:     len(channels) - 1,
        ChannelName:   req.ChannelName,
        StorageType:   req.StorageType,
        StorageConfig: req.StorageConfig,
    }
    
    // 返回成功响应
    response.Success(c, channelResponse)
}
```
