### 1. 定义响应结构体

在 `internal/api/v1/dto/config_dto.go` 文件中定义站群设置响应结构体：

```go
package dto

// MasterSite 主站设置
type MasterSite struct {
    MasterURL string `json:"master_url"` // 主站地址
    MasterID  string `json:"master_id"`  // 主站标识
}

// ClusterNode 集群节点
type ClusterNode struct {
    NodeID     string `json:"node_id"`     // 节点标识
    NodeURL    string `json:"node_url"`    // 节点地址
    NodeStatus string `json:"node_status"` // 节点状态：active、inactive
    NodeWeight int    `json:"node_weight"` // 节点权重，范围1-100
}

// ClusterConfigResponse 站群设置响应
type ClusterConfigResponse struct {
    MasterSite   MasterSite    `json:"master_site"`   // 主站设置
    ClusterNodes []ClusterNode `json:"cluster_nodes"` // 集群节点列表
}
```

### 2. 实现GetClusterConfig方法

在ConfigHandler中实现GetClusterConfig方法：

```go
// GetClusterConfig 获取站群设置
// @Summary 获取站群设置
// @Description 获取网站集群部署配置
// @Tags 系统配置
// @Accept json
// @Produce json
// @Success 200 {object} response.Envelope{data=dto.ClusterConfigResponse} "成功响应"
// @Failure 500 {object} response.Envelope "服务器错误"
// @Router /api/v1/system/config/cluster [get]
func (h *ConfigHandler) GetClusterConfig(c *gin.Context) {
    ctx := c.Request.Context()
    
    // 从服务层获取配置
    configData, err := h.configService.GetConfig(ctx, string(model.ConfigTypeCluster))
    if err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_GET_FAILED", "获取站群设置失败", err)
        return
    }
    
    // 将配置数据转换为响应结构体
    var clusterConfig dto.ClusterConfigResponse
    if err := mapToClusterConfig(configData, &clusterConfig); err != nil {
        response.Error(c, http.StatusInternalServerError, "CONFIG_PARSE_FAILED", "解析站群设置失败", err)
        return
    }
    
    // 返回成功响应
    response.Success(c, clusterConfig)
}

// mapToClusterConfig 将配置数据映射到ClusterConfigResponse
func mapToClusterConfig(data interface{}, clusterConfig *dto.ClusterConfigResponse) error {
    configMap, ok := data.(map[string]interface{})
    if !ok {
        return fmt.Errorf("配置数据格式错误")
    }
    
    // 映射主站设置
    if masterSite, ok := configMap["master_site"].(map[string]interface{}); ok {
        if v, ok := masterSite["master_url"].(string); ok {
            clusterConfig.MasterSite.MasterURL = v
        }
        if v, ok := masterSite["master_id"].(string); ok {
            clusterConfig.MasterSite.MasterID = v
        }
    }
    
    // 映射集群节点列表
    if clusterNodes, ok := configMap["cluster_nodes"].([]interface{}); ok {
        clusterConfig.ClusterNodes = make([]dto.ClusterNode, 0, len(clusterNodes))
        for _, nodeInterface := range clusterNodes {
            nodeMap, ok := nodeInterface.(map[string]interface{})
            if !ok {
                continue
            }
            
            node := dto.ClusterNode{}
            if v, ok := nodeMap["node_id"].(string); ok {
                node.NodeID = v
            }
            if v, ok := nodeMap["node_url"].(string); ok {
                node.NodeURL = v
            }
            if v, ok := nodeMap["node_status"].(string); ok {
                node.NodeStatus = v
            }
            if v, ok := nodeMap["node_weight"].(float64); ok {
                node.NodeWeight = int(v)
            }
            clusterConfig.ClusterNodes = append(clusterConfig.ClusterNodes, node)
        }
    }
    
    return nil
}
```
